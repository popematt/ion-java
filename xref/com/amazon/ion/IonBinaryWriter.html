16  package com.amazon.ion;
17  
18  import java.io.IOException;
19  import java.io.OutputStream;
20  
21  /**
22   * An {@link IonWriter} that encodes Ion binary data.
23   *
24   * @deprecated Use {@link IonSystem#newBinaryWriter(OutputStream, SymbolTable...)} instead.
25   */
26  @Deprecated
27  public interface IonBinaryWriter
28      extends IonWriter
29  {
30  
31      /**
32       * Gets the size in bytes of this binary data.
33       * This is generally needed before calling {@link #getBytes()} or
34       * {@link #getBytes(byte[], int, int)}.
35       *
36       * @return the size in bytes.
37       */
38      @Deprecated
39      public int byteSize();
40  
41      /**
42       * Copies the current contents of this writer as a new byte array holding
43       * Ion binary-encoded data.
44       * This allocates an array of the size needed to exactly
45       * hold the output and copies the entire value to it.
46       *
47       * @return the byte array with the writers output
48       * @throws IOException
49       */
50      @Deprecated
51      public byte[] getBytes()
52          throws IOException;
53  
54  
55      /**
56       * Copies the current contents of the writer to a given byte array
57       * array.  This starts writing to the array at offset and writes
58       * up to maxlen bytes.
59       * If this writer is not able to stop in the middle of its
60       * work this may overwrite the array and later throw and exception.
61       *
62       * @param bytes users byte array to write into
63       * @param offset initial offset in the array to write into
64       * @param maxlen maximum number of bytes to write
65       * @return number of bytes written
66       * @throws IOException
67       */
68      @Deprecated
69      public int getBytes(byte[] bytes, int offset, int maxlen)
70          throws IOException;
71  
72      /**
73       * Writes the current contents of the writer to the output
74       * stream.  This is only valid if the writer is not in the
75       * middle of writing a container.
76       *
77       * @param userstream OutputStream to write the bytes to
78       * @return int length of bytes written
79       * @throws IOException
80       */
81      @Deprecated
82      public int writeBytes(OutputStream userstream)
83          throws IOException;
84  
85  }
