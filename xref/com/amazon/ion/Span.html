18  import com.amazon.ion.facet.Faceted;

21  /**
22   * An immutable reference to a consecutive sequence of values (perhaps
23   * including large hierarchies) within some base source of Ion data.
24   * <p>
25   * <b>WARNING:</b> This interface should not be implemented or extended by
26   * code outside of this library.
27   * <p>
28   * A span is conceptually comprised of two abstract <em>positions</em> or
29   * offsets within the base source.  The <em>start</em> position denotes the
30   * leftmost edge of the span, and the <em>finish</em> position denotes the
31   * rightmost edge.
32   * Positions lie <em>between</em> values, and when the start and finish
33   * positions are equal, the span is said to be <em>empty</em>.
34   * <p>
35   * A span is said to <em>cover</em> the values that lie within its edges.
36   * <p>
37   * A span is <em>balanced</em> if it starts and finishes within the same
38   * container, otherwise it is <em>unbalanced</em>.  This library currently
39   * does not support unbalanced spans.
40   * <p>
41   * Since different source types require different positioning techniques,
42   * spans is {@link Faceted} to expose the position implementation.
43   *
44   * <h2>Acknowledgements</h2>
45   * This design and terminology is heavily based on Wilfred J. Hansen's work on
46   * subsequence references.
47   *
48   * @see SpanProvider
49   * @see Spans
50   * @see TextSpan
51   * @see OffsetSpan
52   *
53   * @see <a href="http://portal.acm.org/citation.cfm?id=133234">Subsequence
54   * References: First-Class Values for Substrings</a>
55   *
56   */
57  public interface Span
58      extends Faceted
59  {
60      /**
61       * Gets a span covering the container of this span.
62       *
63       * @throws IonException if the current span is at top-level.
64       */
65  //    public Span containerSpan();  // TODO later.  Move to a facet?
66  
67  
68      /**
69       * Gets a span covering all the children of this span, which must cover a
70       * single container.
71       *
72       * @throws IonException if this span covers anything other than a
73       * single container.
74       */
75  //    public Span contentSpan();  // TODO later.  Move to a facet?
76  }
