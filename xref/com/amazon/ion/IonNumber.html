16  package com.amazon.ion;
17  
18  import java.math.BigDecimal;
19  
20  /**
21   * Common functionality of Ion {@code int}, {@code decimal}, and {@code float}
22   * types.
23   * <p>
24   * <b>WARNING:</b> This interface should not be implemented or extended by
25   * code outside of this library.
26   */
27  public interface IonNumber
28      extends IonValue
29  {
30      /**
31       * Gets the value of this Ion number as a {@link BigDecimal}.
32       *
33       * <p>This method will throw an exception for non-null, non-numeric values.
34       * It's recommended to call {@link #isNumericValue()} before calling
35       * this method.</p>
36       *
37       * <p>Negative zero is supported by {@link IonDecimal} and {@link IonFloat},
38       * but not by {@link BigDecimal}. If you need to distinguish positive and
39       * negative zero, you should call {@link IonDecimal#decimalValue()} or
40       * {@link IonFloat#doubleValue()} after casting to the appropriate type.</p>
41       *
42       * @return the {@link BigDecimal} value, or {@code null} if
43       * {@code this.isNullValue()}.
44       *
45       * @throws NumberFormatException if this value is {@code nan},
46       * {@code +inf}, or {@code -inf}, because {@link BigDecimal} cannot
47       * represent those values.
48       */
49      public BigDecimal bigDecimalValue();
50  
51      /**
52       * Determines whether this value is numeric. Returns true if this value
53       * is none of {@code null}, {@code nan}, {@code +inf}, and {@code -inf},
54       * and false if it is any of them.
55       *
56       * @return a checked condition whether this value is numeric.
57       */
58      public boolean isNumericValue();
59  }
