34  /**
35   * Tokenizer for the Ion text parser in IonTextIterator. This
36   * reads bytes and returns the interesting tokens it recognizes
37   * or an error.  While, currently, this does UTF-8 decoding
38   * as it goes that is unnecessary.  The main entry point is
39   * lookahead(n) which gets the token type n tokens ahead (0
40   * is the next token).  The tokens type, its starting offset
41   * in the input stream and its ending offset in the input stream
42   * are cached, so lookahead() can be called repeatedly with
43   * little overhead.  This supports a 7 token lookahead and requires
44   * a "recompile" to change this limit.  (this could be "fixed"
45   * but seems unnecessary at this time - the limit is in
46   * IonTextTokenizer._token_lookahead_size which is 1 larger than
47   * the size of the lookahead allowed)  Tokens are consumed by
48   * a call to consumeToken, or the helper consumeTokenAsString.
49   * The informational interfaces - getValueStart(), getValueEnd()
50   * getValueAsString() can be used to get the contents of the
51   * value once the caller has decided how to use it.
52   *
53   *  This is a copy and paste from IonTextTokenize on the introduction of
54   *  the new input abstraction IonInputStream as the source of characters
55   *  and bytes for the reader.
56   *
57   *  This variation does NOT make local copies of the tokens.  It does
58   *  start "marking" at the beginning of the token and the end.  The stream
59   *  will buffer the input until the mark is released.
60   *
61   *  The result is that only the most recent token is available to the
62   *  calling reader.
63   *
64   */

183     protected final int read_string_char(ProhibitedCharacters prohibitedCharacters) throws IOException
184     {
185         int c = _stream.read();
186         if (prohibitedCharacters.includes(c)) {
187             error("invalid character [" + printCodePointAsString(c) + "]");
188         }
189         // the c == '\\' clause will cause us to eat ALL slash-newlines

196     private final void unread_char(int c)
197     {
198         if (c < 0) {
199             switch (c) {
200             case CharacterSequence.CHAR_SEQ_NEWLINE_SEQUENCE_1:
201                 line_count_unread(c);
202                 _stream.unread('\n');
203                 break;
204             case CharacterSequence.CHAR_SEQ_NEWLINE_SEQUENCE_2:
205                 line_count_unread(c);
206                 _stream.unread('\r');
207                 break;
208             case CharacterSequence.CHAR_SEQ_NEWLINE_SEQUENCE_3:
209                 line_count_unread(c);
210                 _stream.unread('\n');
211                 _stream.unread('\r');
212                 break;
213             case CharacterSequence.CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_1:
214                 _stream.unread('\n');
215                 _stream.unread('\\');
216                 break;
217             case CharacterSequence.CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_2:
218                 _stream.unread('\r');
219                 _stream.unread('\\');
220                 break;
221             case CharacterSequence.CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_3:
222                 _stream.unread('\n');
223                 _stream.unread('\r');
224                 _stream.unread('\\');
225                 break;
226             case UnifiedInputStreamX.EOF:
227                 _stream.unread(UnifiedInputStreamX.EOF);
228                 break;
229             default:
230                 assert false
231                     : "INVALID SPECIAL CHARACTER ENCOUNTERED: " + c;
232             }
233         }
234         else  {
235             _stream.unread(c);
236         }
237     }

239     private final int line_count_unread(int c) {
240         assert( c == CharacterSequence.CHAR_SEQ_NEWLINE_SEQUENCE_1
241              || c == CharacterSequence.CHAR_SEQ_NEWLINE_SEQUENCE_2
242              || c == CharacterSequence.CHAR_SEQ_NEWLINE_SEQUENCE_3
243              || c == CharacterSequence.CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_1
244              || c == CharacterSequence.CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_2
245              || c == CharacterSequence.CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_3
246         );
247         if (_line_count_has_cached) {
248             _line_count = _line_count_cached;
249             _line_starting_position = _line_offset_cached;
250             _line_count_has_cached = false;
251         }
252         return c;
253     }
254     private final int line_count(int c) throws IOException
255     {
256         // check for the slash new line case (and we'l
257         // consume both here it that's what we find
258         switch (c) {
259         case '\\':
260             {
261                 int c2 = _stream.read();
262                 switch (c2) {
263                 case '\r':  // DOS <cr><lf>  or old Mac <cr>
264                     int c3 = _stream.read();
265                     if (c3 != '\n') {
266                         unread_char(c3);
267                         c = CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_2;
268                     }
269                     else {
270                         c = CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_3;
271                     }
272                     break;
273                 case '\n':
274                     // Unix and new Mac (also Unix) <lf>
275                     c = CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_1;
276                     break;
277                 default:
278                     // not a slash new line, so we'll just return the slash
279                     // leave it to be handled elsewhere
280                     unread_char(c2);
281                     return c;
282                 }
283             }
284             break;
285         case '\r':
286             {
287                 // convert '\r' or '\r\n' into the appropriate CHAR_SEQ
288                 // pseudo character
289                 int c2 = _stream.read();
290                 if (c2 == '\n') {
291                     c = CHAR_SEQ_NEWLINE_SEQUENCE_3;
292                 }
293                 else {
294                     unread_char(c2);
295                     c = CHAR_SEQ_NEWLINE_SEQUENCE_2;
296                 }
297             }
298             break;
299         case '\n':
300             c = CHAR_SEQ_NEWLINE_SEQUENCE_1;
301             break;
302         default:
303             throw new IllegalStateException();
304         }
305 
306         // before we adjust the line count we save it so that
307         // we can recover from a unread of a line terminator
308         // note that we can only recover from a single line
309         // terminator unread, but that should be enough.  We
310         // only unread whitespace if it's a delimiter, and
311         // then we only have to unread a single instance.
312         _line_count_cached = _line_count;
313         _line_offset_cached = _line_starting_position;
314         _line_count_has_cached = true;
315 
316         // anything else (and that should only be either a new line
317         // of IonTokenConsts.ESCAPED_NEWLINE_SEQUENCE passed in) we will
318         // return the char unchanged and line count
319         _line_count++;
320         // since we want the first character of the line to be 1, not 0:
321         _line_starting_position = _stream.getPosition() - 1;
322 
323         return c;
324     }

326     /**
327      * peeks into the input stream to see if the next token
328      * would be a double colon.  If indeed this is the case
329      * it skips the two colons and returns true.  If not
330      * it unreads the 1 or 2 real characters it read and
331      * return false.
332      * It always consumes any preceding whitespace.
333      * @return true if the next token is a double colon, false otherwise
334      * @throws IOException
335      */
336     public final boolean skipDoubleColon() throws IOException
337     {
338         int c = skip_over_whitespace();
339         if (c != ':') {
340             unread_char(c);
341             return false;
342         }
343         c = read_char();
344         if (c != ':') {
345             unread_char(c);
346             unread_char(':');
347             return false;
348         }
349         return true;
350     }

353     /**
354      * peeks into the input stream to see if we have an
355      * unquoted symbol that resolves to one of the ion
356      * types.  If it does it consumes the input and
357      * returns the type keyword id.  If not is unreads
358      * the non-whitespace characters and the dot, which
359      * the input argument 'c' should be.
360      */
361     public final int peekNullTypeSymbol() throws IOException
362     {
363         // the '.' has to follow the 'null' immediately
364         int c = read_char();
365         if (c != '.') {
366             unread_char(c);
367             return IonTokenConstsX.KEYWORD_none;
368         }
369 
370         // we have a dot, start reading through the following non-whitespace
371         // and we'll collect it so that we can unread it in the event
372         // we don't actually see a type name
373         int[] read_ahead = new int[IonTokenConstsX.TN_MAX_NAME_LENGTH + 1];
374         int read_count = 0;
375         int possible_names = IonTokenConstsX.KW_ALL_BITS;
376 
377         while (read_count < IonTokenConstsX.TN_MAX_NAME_LENGTH + 1) {
378             c = read_char();
379             read_ahead[read_count++] = c;
380             int letter_idx = IonTokenConstsX.typeNameLetterIdx(c);
381             if (letter_idx < 1) {
382                 if (IonTokenConstsX.isValidTerminatingCharForInf(c)) {
383                     // it's not a letter we care about but it is
384                     // a valid end of const, so maybe we have a keyword now
385                     // we always exit the loop here since we look
386                     // too far so any letter is invalid at pos 10
387                     break;
388                 }
389                 return peekNullTypeSymbolUndo(read_ahead, read_count);
390             }
391             int mask = IonTokenConstsX.typeNamePossibilityMask(read_count - 1, letter_idx);
392             possible_names &= mask;
393             if (possible_names == 0) {
394                 // in this case it can't be a valid keyword since
395                 // it has identifier chars (letters) at 1 past the
396                 // last possible end (at least)
397                 return peekNullTypeSymbolUndo(read_ahead, read_count);
398             }
399         }
400         // now lets get the keyword value from our bit mask
401         // at this point we can fail since we may have hit
402         // a valid terminator before we're done with all key
403         // words.  We even have to check the length.
404         // for example "in)" matches both letters to the
405         // typename int and terminates validly - but isn't
406         // long enough, but with length we have enough to be sure
407         // with the actual type names we're using in 1.0
408         int kw = IonTokenConstsX.typeNameKeyWordFromMask(possible_names, read_count-1);
409         if (kw == IonTokenConstsX.KEYWORD_unrecognized) {
410             peekNullTypeSymbolUndo(read_ahead, read_count);
411         }
412         else {
413             // since we're accepting the rest we aren't unreading anything
414             // else - but we still have to unread the character that stopped us
415             unread_char(c);
416         }
417         return kw;
418     }
419     private final int peekNullTypeSymbolUndo(int[] read_ahead, int read_count)
420     {
421         String type_error = "";
422         for (int ii=0; ii<read_count; ii++) {
423             // this (string concatenation) is horrible, but we're about throw anyway
424             type_error += (char)read_ahead[ii];
425         }
426 
427         String message = "invalid type name on a typed null value";
428         error(message); // this throws so we won't actually return
429         return IonTokenConstsX.KEYWORD_unrecognized;
430     }

495     private final int skip_to_end(SavePoint sp)  throws IOException
496     {
497         int c;
498 
499         // FIXME lots of inconsistency here!
500         // Sometimes the token's first character is still on the stream,
501         // sometimes it's already been consumed.
502 
503         switch (_token) {
504         case IonTokenConstsX.TOKEN_UNKNOWN_NUMERIC:
505             c = skip_over_number(sp);
506             break;
507         case IonTokenConstsX.TOKEN_INT:
508             c = skip_over_int(sp);
509             break;
510         case IonTokenConstsX.TOKEN_HEX:
511             c = skipOverRadix(sp, Radix.HEX);
512             break;
513         case IonTokenConstsX.TOKEN_BINARY:
514             c = skipOverRadix(sp, Radix.BINARY);
515             break;
516         case IonTokenConstsX.TOKEN_DECIMAL:
517             c = skip_over_decimal(sp);
518             break;
519         case IonTokenConstsX.TOKEN_FLOAT:
520             c = skip_over_float(sp);
521             break;
522         case IonTokenConstsX.TOKEN_TIMESTAMP:
523             c = skip_over_timestamp(sp);
524             break;
525         case IonTokenConstsX.TOKEN_SYMBOL_IDENTIFIER:
526             c = skip_over_symbol_identifier(sp);
527             break;
528         case IonTokenConstsX.TOKEN_SYMBOL_QUOTED:
529             // Initial single-quote has been consumed!
530             assert(!is_2_single_quotes_helper());
531             c = skip_single_quoted_string(sp);
532             break;
533         case IonTokenConstsX.TOKEN_SYMBOL_OPERATOR:
534             // Initial operator char has NOT been consumed
535             c = skip_over_symbol_operator(sp);
536             break;
537         case IonTokenConstsX.TOKEN_STRING_DOUBLE_QUOTE:
538             skip_double_quoted_string_helper(); // FIXME Why no sp here?
539             c = skip_over_whitespace();
540             break;
541         case IonTokenConstsX.TOKEN_STRING_TRIPLE_QUOTE:
542             skip_triple_quoted_string(sp);
543             c = skip_over_whitespace();
544             break;
545 
546         case IonTokenConstsX.TOKEN_OPEN_DOUBLE_BRACE:
547             // works just like a pair of nested structs
548             // since "skip_over" doesn't care about formal
549             // syntax (like requiring field names);
550             skip_over_blob(sp);
551             c = read_char();
552             break;
553         case IonTokenConstsX.TOKEN_OPEN_BRACE:
554             assert( sp == null ); // you can't save point a scanned struct (right now anyway)
555             skip_over_struct();
556             c = read_char();
557             break;
558         case IonTokenConstsX.TOKEN_OPEN_PAREN:
559             skip_over_sexp(); // you can't save point a scanned sexp (right now anyway)
560             c = read_char();
561             break;
562         case IonTokenConstsX.TOKEN_OPEN_SQUARE:
563             skip_over_list();  // you can't save point a scanned list (right now anyway)
564             c = read_char();
565             break;
566         case IonTokenConstsX.TOKEN_DOT:
567         case IonTokenConstsX.TOKEN_COMMA:
568         case IonTokenConstsX.TOKEN_COLON:
569         case IonTokenConstsX.TOKEN_DOUBLE_COLON:
570         case IonTokenConstsX.TOKEN_CLOSE_PAREN:
571         case IonTokenConstsX.TOKEN_CLOSE_BRACE:
572         case IonTokenConstsX.TOKEN_CLOSE_SQUARE:
573         case IonTokenConstsX.TOKEN_CLOSE_DOUBLE_BRACE:
574         case IonTokenConstsX.TOKEN_ERROR:
575         case IonTokenConstsX.TOKEN_EOF:
576         default:
577             c = -1; // makes eclipse happy
578             error("token "+IonTokenConstsX.getTokenName(_token)+
579                   " unexpectedly encounterd as \"unfinished\"");
580             break;
581         }
582         if (IonTokenConstsX.isWhitespace(c)) {
583             c = skip_over_whitespace();
584         }
585         _unfinished_token = false;
586         return c;
587     }

603     public final int nextToken() throws IOException
604     {
605         int t = -1;
606         int c, c2;
607 
608         if (_unfinished_token) {
609             c = skip_to_end(null);
610         }
611         else {
612             c = skip_over_whitespace();
613         }
614         _unfinished_token = true;
615 
616         switch (c) {
617         case -1:
618             return next_token_finish(IonTokenConstsX.TOKEN_EOF, true);
619         case ':':
620             c2 = read_char();
621             if (c2 != ':') {
622                 unread_char(c2);
623                 return next_token_finish(IonTokenConstsX.TOKEN_COLON, true);
624             }
625             return next_token_finish(IonTokenConstsX.TOKEN_DOUBLE_COLON, true);
626         case '{':
627             c2 = read_char();
628             if (c2 != '{') {
629                 unread_char(c2);
630                 return next_token_finish(IonTokenConstsX.TOKEN_OPEN_BRACE, true); // CAS: 9 nov 2009
631             }
632             return next_token_finish(IonTokenConstsX.TOKEN_OPEN_DOUBLE_BRACE, true);
633         case '}':
634             // detection of double closing braces is done
635             // in the parser in the blob and clob handling
636             // state - it's otherwise ambiguous with closing
637             // two structs together. see tryForDoubleBrace() below
638             return next_token_finish(IonTokenConstsX.TOKEN_CLOSE_BRACE, false);
639         case '[':
640             return next_token_finish(IonTokenConstsX.TOKEN_OPEN_SQUARE, true); // CAS: 9 nov 2009
641         case ']':
642             return next_token_finish(IonTokenConstsX.TOKEN_CLOSE_SQUARE, false);
643         case '(':
644             return next_token_finish(IonTokenConstsX.TOKEN_OPEN_PAREN, true); // CAS: 9 nov 2009
645         case ')':
646             return next_token_finish(IonTokenConstsX.TOKEN_CLOSE_PAREN, false);
647         case ',':
648             return next_token_finish(IonTokenConstsX.TOKEN_COMMA, false);
649         case '.':
650             c2 = read_char();
651             unread_char(c2);
652             if (IonTokenConstsX.isValidExtendedSymbolCharacter(c2)) {
653                 unread_char('.');
654                 return next_token_finish(IonTokenConstsX.TOKEN_SYMBOL_OPERATOR, true);
655             }
656             return next_token_finish(IonTokenConstsX.TOKEN_DOT, false);
657         case '\'':
658             if (is_2_single_quotes_helper()) {
659                 return next_token_finish(IonTokenConstsX.TOKEN_STRING_TRIPLE_QUOTE, true);
660             }
661             // unread_char(c);
662             return next_token_finish(IonTokenConstsX.TOKEN_SYMBOL_QUOTED, true);
663         case '+':
664             if (peek_inf_helper(c)) // this will consume the inf if it succeeds
665             {
666                 return next_token_finish(IonTokenConstsX.TOKEN_FLOAT_INF, false);
667             }
668             unread_char(c);
669             return next_token_finish(IonTokenConstsX.TOKEN_SYMBOL_OPERATOR, true);
670         case '#': case '/':
671         case '<': case '>': case '*': case '=': case '^': case '&': case '|':
672         case '~': case ';': case '!': case '?': case '@': case '%': case '`':
673             unread_char(c);
674             return next_token_finish(IonTokenConstsX.TOKEN_SYMBOL_OPERATOR, true);
675         case '"':
676             return next_token_finish(IonTokenConstsX.TOKEN_STRING_DOUBLE_QUOTE, true);
677         case 'a': case 'b': case 'c': case 'd': case 'e': case 'f':
678         case 'g': case 'h': case 'i': case 'j': case 'k': case 'l':
679         case 'm': case 'n': case 'o': case 'p': case 'q': case 'r':
680         case 's': case 't': case 'u': case 'v': case 'w': case 'x':
681         case 'y': case 'z':
682         case 'A': case 'B': case 'C': case 'D': case 'E': case 'F':
683         case 'G': case 'H': case 'I': case 'J': case 'K': case 'L':
684         case 'M': case 'N': case 'O': case 'P': case 'Q': case 'R':
685         case 'S': case 'T': case 'U': case 'V': case 'W': case 'X':
686         case 'Y': case 'Z':
687         case '$': case '_':
688             unread_char(c);
689             return next_token_finish(IonTokenConstsX.TOKEN_SYMBOL_IDENTIFIER, true);
690         case '0': case '1': case '2': case '3': case '4':
691         case '5': case '6': case '7': case '8': case '9':
692             t = scan_for_numeric_type(c);
693             unread_char(c);
694             return next_token_finish(t, true);
695         case '-':
696             // see if we have a number or what might be an extended symbol
697             c2 = read_char();
698             unread_char(c2);
699             if (IonTokenConstsX.isDigit(c2)) {
700                 t = scan_negative_for_numeric_type(c);
701                 unread_char(c);
702                 return next_token_finish(t, true);
703             }
704             else if (peek_inf_helper(c)) // this will consume the inf if it succeeds
705             {
706                 return next_token_finish(IonTokenConstsX.TOKEN_FLOAT_MINUS_INF, false);
707             }
708             else {
709                 unread_char(c);
710                 return next_token_finish(IonTokenConstsX.TOKEN_SYMBOL_OPERATOR, true);
711             }
712         default:
713             bad_token_start(c); // throws
714         }
715         throw new IonException("invalid state: next token switch shouldn't exit");
716     }

1008     /**
1009      * we encountered a character that starts a number,
1010      * a digit or a dash (minus).  Now we'll scan a little
1011      * ways ahead to spot some of the numeric types.
1012      *
1013      * this only looks far enough (2 or 6 chars) to identify
1014      * hex and timestamps
1015      * it might encounter a decimal or a 'd' or an 'e' and
1016      * decide this token is float or decimal (or int if we
1017      * hit a non-numeric char) but it may return TOKEN_UNKNOWN_NUMERIC;
1018      *
1019      * if will unread everything it's read, and the character
1020      * passed in as the first digit encountered
1021      *
1022      * @param c first char of number read by caller
1023      * @return numeric token type
1024      * @throws IOException
1025      */
1026     private final int scan_for_numeric_type(int c1) throws IOException
1027     {
1028         int   t = IonTokenConstsX.TOKEN_UNKNOWN_NUMERIC;
1029         int[] read_chars = new int[6];
1030         int   read_char_count = 0;
1031         int   c;
1032 
1033         if (!IonTokenConstsX.isDigit(c1)) {
1034             error(String.format("Expected digit, got U+%04X", c1));
1035         }
1036 
1037         // the caller needs to unread this if they want to: read_chars[read_char_count++] = c1;
1038 
1039         c = read_char();
1040         read_chars[read_char_count++] = c;
1041 
1042         if (c1 == '0') {
1043             // check for hex
1044             switch(c) {
1045             case 'x':
1046             case 'X':
1047                 t = IonTokenConstsX.TOKEN_HEX;
1048                 break;
1049             case 'd':
1050             case 'D':
1051                 t = IonTokenConstsX.TOKEN_DECIMAL;
1052                 break;
1053             case 'e':
1054             case 'E':
1055                 t = IonTokenConstsX.TOKEN_FLOAT;
1056                 break;
1057             case 'b':
1058             case 'B':
1059                 t = IonTokenConstsX.TOKEN_BINARY;
1060                 break;
1061             case '.':
1062                 // the decimal might have an 'e' somewhere down the line so we
1063                 // don't really know the type here
1064                 break;
1065             default:
1066                 if (is_value_terminating_character(c)) {
1067                     t = IonTokenConstsX.TOKEN_INT;
1068                 }
1069                 break;
1070             }
1071         }
1072         if (t == IonTokenConstsX.TOKEN_UNKNOWN_NUMERIC) { // oh for goto :(
1073             if (IonTokenConstsX.isDigit(c)) { // 2nd digit
1074                 // it might be a timestamp if we have 4 digits, a dash,
1075                 // and a digit
1076                 c = read_char();
1077                 read_chars[read_char_count++] = c;
1078                 if (IonTokenConstsX.isDigit(c)) { // digit 3
1079                     c = read_char();
1080                     read_chars[read_char_count++] = c;
1081                     if (IonTokenConstsX.isDigit(c)) {
1082                         // last digit of possible year
1083                         c = read_char();
1084                         read_chars[read_char_count++] = c;
1085                         if (c == '-' || c =='T') {
1086                             // we have dddd- or ddddT looks like a timestamp
1087                             // (or invalid input)
1088                             t = IonTokenConstsX.TOKEN_TIMESTAMP;
1089                         }
1090                     }
1091                 }
1092             }
1093         }
1094 
1095         // unread whatever we read, including the passed in char
1096         do {
1097             read_char_count--;
1098             c = read_chars[read_char_count];
1099             unread_char(c);
1100         } while (read_char_count > 0);
1101 
1102         return t;
1103     }

1109

1121         case CharacterSequence.CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_1:
1122         case CharacterSequence.CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_2:
1123         case CharacterSequence.CHAR_SEQ_ESCAPED_NEWLINE_SEQUENCE_3:

1207     {
1208         assert( terminator == '}' || terminator == ']' || terminator == ')' );
1209         int c;
1210 
1211         for (;;) {
1212             c = skip_over_whitespace();
1213             switch (c) {
1214             case -1:
1215                 unexpected_eof();
1216             case '}':
1217             case ']':
1218             case ')':
1219                 if (c == terminator) { // no point is checking this on every char
1220                     return;
1221                 }
1222                 break;
1223             case '"':
1224                 skip_double_quoted_string_helper();
1225                 break;
1226             case '\'':
1227                 if (is_2_single_quotes_helper()) {
1228                     skip_triple_quoted_string(null);
1229                 }
1230                 else {
1231                     c = skip_single_quoted_string(null);
1232                     unread_char(c);
1233                 }
1234                 break;
1235             case '(':
1236                 skip_over_container(')');
1237                 break;
1238             case '[':
1239                 skip_over_container(']');
1240                 break;
1241             case '{':
1242                 // this consumes lobs as well since the double
1243                 // braces count correctly and the contents
1244                 // of either clobs or blobs will be just content
1245                 c = read_char();
1246                 if (c == '{') {
1247                     // 2nd '{' - it's a lob of some sort - let's find out what sort
1248                     c = skip_over_lob_whitespace();

1266                     }
1267

1272                     // move on, there's nothing to see here ...

1277                 }

1376     {

1384             }
1385             return skip_over_whitespace(); // prefetch
1386         }
1387         if (c != '-') {
1388             error("invalid timestamp encountered");
1389         }

1391         // yyyy-mm-ddT
1392         // yyyy-mm-ddT+hh:mm
1393         // yyyy-mm-ddThh:mm+hh:mm
1394         // yyyy-mm-ddThh:mm:ss+hh:mm
1395         // yyyy-mm-ddThh:mm:ss.dddd+hh:mm

1398         // yyyy-mm-ddThh:mm:ss.ddddZ

1402             if (sp != null) {
1403                 sp.markEnd(0);
1404             }
1405             return skip_over_whitespace(); // prefetch
1406         }

1411         }

1420             bad_token(c);
1421         }

1434             c = skip_over_digits(c);
1435         }
1436         // yyyy-mm-ddThh:mm:ss.ddd?
1437

1526     protected IonType load_number(StringBuilder sb) throws IOException
1527     {
1528         boolean has_sign = false;
1529         int     t, c;
1530 
1531         // this reads int, float, decimal and timestamp strings
1532         // anything staring with a +, a - or a digit
1533         //case '0': case '1': case '2': case '3': case '4':
1534         //case '5': case '6': case '7': case '8': case '9':
1535         //case '-': case '+':
1536 
1537         //start_pos = _stream.getPosition();
1538         c = read_char();
1539         has_sign = ((c == '-') || (c == '+'));
1540         if (has_sign) {
1541             // if there is a sign character, we just consume it
1542             // here and get whatever is next in line
1543             sb.append((char)c);
1544             c = read_char();
1545         }
1546 
1547         // first leading digit - to look for hex and
1548         // to make sure that there is at least 1 digit (or
1549         // this isn't really a number
1550         if (!IonTokenConstsX.isDigit(c)) {
1551             // if it's not a digit, this isn't a number
1552             // the only non-digit it could have been was a
1553             // sign character, and we'll have read past that
1554             // by now
1555             // TODO this will be a confusing error message,
1556             // but I can't figure out when it will be reached.
1557             bad_token(c);
1558         }
1559 
1560         // the first digit is a special case
1561         boolean starts_with_zero = (c == '0');
1562         if (starts_with_zero) {
1563             // if it's a leading 0 check for a hex value
1564             int c2 = read_char();
1565             if (Radix.HEX.isPrefix(c2)) {
1566                 sb.append((char)c);
1567                 c = loadRadixValue(sb, has_sign, c2, Radix.HEX);
1568                 return load_finish_number(sb, c, IonTokenConstsX.TOKEN_HEX);
1569             } else if (Radix.BINARY.isPrefix(c2)) {
1570                 sb.append((char) c);
1571                 c = loadRadixValue(sb, has_sign, c2, Radix.BINARY);
1572                 return load_finish_number(sb, c, IonTokenConstsX.TOKEN_BINARY);
1573             }
1574             // not a next value, back up and try again
1575             unread_char(c2);
1576         }
1577 
1578         // remaining (after the first, c is the first) leading digits
1579         c = load_digits(sb, c);
1580 
1581         if (c == '-' || c == 'T') {
1582             // this better be a timestamp and it starts with a 4 digit
1583             // year followed by a dash and no leading sign
1584             if (has_sign) {
1585                 error("Numeric value followed by invalid character: "
1586                       + sb + (char)c);
1587             }
1588             int len = sb.length();
1589             if (len != 4) {
1590                 error("Numeric value followed by invalid character: "
1591                       + sb + (char)c);
1592             }
1593             IonType tt = load_timestamp(sb, c);
1594             return tt;
1595         }
1596 
1597         if (starts_with_zero) {
1598             // Ion doesn't allow leading zeros, so make sure our buffer only
1599             // has one character.
1600             int len = sb.length();
1601             if (has_sign) {
1602                 len--; // we don't count the sign
1603             }
1604             if (len != 1) {
1605                 error("Invalid leading zero in number: " + sb);
1606             }
1607         }
1608 
1609         if (c == '.') {
1610             // so if it's a float of some sort
1611             // mark it as at least a DECIMAL
1612             // and read the "fraction" digits
1613             sb.append((char)c);
1614             c = read_char();
1615             c = load_digits(sb, c);
1616             t = IonTokenConstsX.TOKEN_DECIMAL;
1617         }
1618         else {
1619             t = IonTokenConstsX.TOKEN_INT;
1620         }
1621 
1622         // see if we have an exponential as in 2d+3
1623         if (c == 'e' || c == 'E') {
1624             t = IonTokenConstsX.TOKEN_FLOAT;
1625             sb.append((char)c);
1626             c = load_exponent(sb);  // the unused lookahead char
1627         }
1628         else if (c == 'd' || c == 'D') {
1629             t = IonTokenConstsX.TOKEN_DECIMAL;
1630             sb.append((char)c);
1631             c = load_exponent(sb);
1632         }
1633         return load_finish_number(sb, c, t);
1634     }

1728     private final IonType load_timestamp(StringBuilder sb, int c)
1729         throws IOException
1730     {
1731         // we read the year in our caller, we should only be
1732         // here is we read 4 digits and then a dash or a 'T'
1733         assert (c == '-' || c == 'T');
1734 
1735         sb.append((char)c);
1736 
1737         // if it's 'T' we done: yyyyT
1738         if (c == 'T') {
1739             c = read_char(); // because we'll unread it before we return
1740             return load_finish_number(sb, c, IonTokenConstsX.TOKEN_TIMESTAMP);
1741         }
1742 
1743         // read month
1744         load_fixed_digits(sb, 2);
1745 
1746         c = read_char();
1747         if (c == 'T') {
1748             sb.append((char)c);
1749             c = read_char(); // because we'll unread it before we return
1750             return load_finish_number(sb, c, IonTokenConstsX.TOKEN_TIMESTAMP);
1751         }
1752         if (c != '-') bad_token(c);
1753 
1754         // read day
1755         sb.append((char)c);
1756         load_fixed_digits(sb, 2);
1757 
1758         // look for the 'T', otherwise we're done (and happy about it)
1759         c = read_char();
1760         if (c != 'T') {
1761             return load_finish_number(sb, c, IonTokenConstsX.TOKEN_TIMESTAMP);
1762         }
1763 
1764         // so either we're done or we must at least hours and minutes
1765         // hour
1766         sb.append((char)c);
1767         c = read_char();
1768         if (!IonTokenConstsX.isDigit(c)) {
1769             return load_finish_number(sb, c, IonTokenConstsX.TOKEN_TIMESTAMP);
1770         }
1771         sb.append((char)c);
1772         load_fixed_digits(sb,1); // we already read the first digit
1773         c = read_char();
1774         if (c != ':') bad_token(c);
1775 
1776         // minutes
1777         sb.append((char)c);
1778         load_fixed_digits(sb, 2);
1779         c = read_char();
1780         if (c == ':') {
1781             // seconds are optional
1782             // and first we'll have the whole seconds
1783             sb.append((char)c);
1784             load_fixed_digits(sb, 2);
1785             c = read_char();
1786             if (c == '.') {
1787                 sb.append((char)c);
1788                 c = read_char();
1789                 // Per spec and W3C Note http://www.w3.org/TR/NOTE-datetime
1790                 // We require at least one digit after the decimal point.
1791                 if (!IonTokenConstsX.isDigit(c)) {
1792                     expected_but_found("at least one digit after timestamp's decimal point", c);
1793                 }
1794                 c = load_digits(sb,c);
1795             }
1796         }
1797 
1798         // since we have a time, we have to have a timezone of some sort
1799         // the timezone offset starts with a '+' '-' 'Z' or 'z'
1800         if (c == 'z' || c == 'Z') {
1801             sb.append((char)c);
1802             // read ahead since we'll check for a valid ending in a bit
1803             c = read_char();
1804         }
1805         else if (c == '+' || c == '-') {
1806             // then ... hours of time offset
1807             sb.append((char)c);
1808             load_fixed_digits(sb, 2);
1809             c = read_char();
1810             if (c != ':') {
1811                 // those hours need their minutes if it wasn't a 'z'
1812                 // (above) then it has to be a +/- hours { : minutes }
1813                 bad_token(c);
1814             }
1815             // and finally the *not* optional minutes of time offset
1816             sb.append((char)c);
1817             load_fixed_digits(sb, 2);
1818             c = read_char();
1819         }
1820         else {
1821             // some sort of offset is required with a time value
1822             // if it wasn't a 'z' (above) then it has to be a +/- hours { : minutes }
1823             bad_token(c);
1824         }
1825         return load_finish_number(sb, c, IonTokenConstsX.TOKEN_TIMESTAMP);
1826     }

2622         // byte1 uses the 6 bits in char1 + 2 highest bits (out of 6) from char2
2623         if (_debug) assert(decode_base64_length(c1, c2, c3, c4) >= 1);
2624         int b1 = (((c1 << 2) & 0xfc) | ((c2 >> 4) & 0x03));
2625         return b1;
2626     }
2627     private final static int decode_base64_byte2(int c1, int c2, int c3, int c4) {
2628         //convert |= (c2 << 12);  [6:1]+12 => [18:13]
2629         //convert |= (c3 << 6);   [6:1]+6  => [12:7]
2630         //b2 = (char)((convert & 0x0000FF00) >> 8); [32:1] & 0x0000FF00 => [16:9] - 8 => [8:1]
2631         // [18:13] - 8 -> [10:5] or [6:5] from c2
2632         // [12:7] - 8 -> [4:-1] or [6:3] - 2 from c3
2633         //byte2 uses 4 low bits from c2 and 4 high bits from c3
2634         if (_debug) assert(decode_base64_length(c1, c2, c3, c4) >= 2);
2635         int b2 = (((c2 << 4) & 0xf0) | ((c3 >> 2) & 0x0f)) & 0xff;
2636         return b2;
2637     }
2638     private final static int decode_base64_byte3(int c1, int c2, int c3, int c4) {
2639         // convert |= (c3 << 6); [6:1]+6  => [12:7]
2640         // convert |= (c4 << 0); [6:1]+9  => [6:1]
2641         // b3 = (char)((convert & 0x000000FF) >> 0);
2642         // b3 uses low 2 bits from c3 and all 6 bits of c4
2643         if (_debug) assert(decode_base64_length(c1, c2, c3, c4) >= 3);
2644         int b3 = (((c3 & 0x03) << 6) | (c4 & 0x3f)) & 0xff;
2645         return b3;

2849     private int readNumeric(Appendable buffer, Radix radix, NumericState startingState) throws IOException
2850     {
2851         NumericState state = startingState;
2852 
2853         for (;;)
2854         {
2855             int c = read_char();
2856             switch (state)
2857             {
2858                 case START:
2859                     if (radix.isValidDigit(c))
2860                     {
2861                         buffer.append(radix.normalizeDigit((char) c));
2862                         state = NumericState.DIGIT;
2863                     }
2864                     else
2865                     {
2866                         return c;
2867                     }
2868                     break;
2869                 case DIGIT:
2870                     if (radix.isValidDigit(c))
2871                     {
2872                         buffer.append(radix.normalizeDigit((char) c));
2873                         state = NumericState.DIGIT;
2874                     }
2875                     else if (c == '_')
2876                     {
2877                         state = NumericState.UNDERSCORE;
2878                     }
2879                     else
2880                     {
2881                         return c;
2882                     }
2883                     break;
2884                 case UNDERSCORE:
2885                     if (radix.isValidDigit(c))
2886                     {
2887                         buffer.append(radix.normalizeDigit((char) c));
2888                         state = NumericState.DIGIT;
2889                     }
2890                     else
2891                     {
2892                         unread_char(c);
2893                         return '_';
2894                     }
2895                     break;
2896             }
