34  /**
35   *    The text user reader add support for symbols and recognizes,
36   *    and consumes (and processes), the system values $ion_1_0 and
37   *    local symbol tables (tagged with $ion_symbol_table).
38   *
39   *    Should this materialize and "symbolate" all the symbol
40   *    values as they come through?  - No.
41   *
42   *    Probably if we want the symbol id's to be the same for this
43   *    reader as it is for other variants.  Hmmm, that's expensive
44   *    when you don't need it (which is most of the time).
45   *
46   *    This will not auto-populate a symbol table.  In the event
47   *    a symbol is a '$<digits>' symbol id symbol this will return
48   *    that value.  If the string is present in the current symbol
49   *    table it will return the id, which would be true if the
50   *    symbol is a system symbol or if there is a local symbol
51   *    table in the input stream.  Otherwise it return the
52   *    undefined symbol value.
53   *
54   */

56      extends IonReaderTextSystemX
57      implements _Private_ReaderWriter

92      /**
93       * this looks forward to see if there is an upcoming value
94       * and if there is it returns true.  It may have to clean up
95       * any value that's partially complete (for example a
96       * collection whose annotation has been read and loaded
97       * but the use has chosen not to step into the collection).
98       *
99       * The user reader variant of hasNext also looks for system
100      * values to process.  System values are the Ion version
101      * marker ($ion_1_0) and local symbol tables.  If either of
102      * these is encountered the symbol table processing will be
103      * handled and the value will be "skipped".
104      *
105      * @return true if more data remains, false on eof
106      */
107     @Override
108     public boolean hasNext()
109     {
110         boolean has_next = has_next_user_value();
111         return has_next;
112     }
113     private final boolean has_next_user_value()
114     {
115         // clear out our previous value
116         clear_system_value_stack();
117 
118         // changed to 'while' since consumed
119         // values will not be counted
120         while (!_has_next_called)
121         {
122             // first move to the next value regardless of whether
123             // it's a system value or a user value
124             has_next_raw_value();
125 
126             // system values are only at the datagram level
127             // we don't care about them if they're buried
128             // down in some other value - note that _value_type
129             // will be null at eof and on as yet undetermined
130             // numeric types (which are never system values)
131             if (_value_type != null && !isNullValue() && IonType.DATAGRAM.equals(getContainerType())) {
132                 switch (_value_type) {
133                 case STRUCT:
134                     if (_annotation_count > 0 && ION_SYMBOL_TABLE.equals(_annotations[0].getText())) {
135                         _symbols = _lstFactory.newLocalSymtab(_catalog,
136                                                               this,
137                                                               true);
138                         push_symbol_table(_symbols);
139                         _has_next_called = false;
140                     }
141                     break;
142                 case SYMBOL:
143                     if (_annotation_count == 0)
144                     {
145                         // $ion_1_0 is read as an IVM only if it is not annotated
146                         String version = symbolValue().getText();
147                         if (isIonVersionMarker(version))
148                         {
149                             if (ION_1_0.equals(version))
150                             {
151                                 if (_value_keyword != IonTokenConstsX.KEYWORD_sid)
152                                 {
153                                     symbol_table_reset();
154                                     push_symbol_table(_system_symtab);
155                                 }
156                                 _has_next_called = false;
157                             }
158                             else
159                             {
160                                 throw new UnsupportedIonVersionException(version);
161                             }
162                         }
163                     }
164                     break;
165                 default:
166                     break;
167                 }
168             }
169         }
170         return (_eof != true);
171     }

173     private static boolean isIonVersionMarker(String text)
174     {
175         return text != null && ION_VERSION_MARKER_REGEX.matcher(text).matches();
176     }

178     private final void symbol_table_reset()
179     {
180         IonType t = next();
181         assert( IonType.SYMBOL.equals(t) );
182         _symbols = _system_symtab;
183         return;
184     }

186     private void validateSymbolToken(SymbolToken symbol) {
187         if (symbol != null) {
188             if (symbol.getText() == null && symbol.getSid() > getSymbolTable().getMaxId()) {
189                 throw new UnknownSymbolException(symbol.getSid());
190             }
191         }

224     //
225     //  This code handles the skipped symbol table
226     //  support - it is cloned in IonReaderTreeUserX
227     //  and IonReaderBinaryUserX
228     //
229     //  SO ANY FIXES HERE WILL BE NEEDED IN THOSE
230     //  TWO LOCATIONS AS WELL.
231     //
232     private int _symbol_table_top = 0;
233     private SymbolTablebolTable[] _symbol_table_stack = new SymbolTable[3]; // 3 is rare, IVM followed by a local sym tab with open content
234     private void clear_system_value_stack()
235     {
236         while (_symbol_table_top > 0) {
237             _symbol_table_top--;
238             _symbol_table_stack[_symbol_table_top] = null;
239         }
240     }
241     private void push_symbol_table(SymbolTable symbols)
242     {
243         assert(symbols != null);
244         if (_symbol_table_top >= _symbol_table_stack.length) {
245             int new_len = _symbol_table_stack.length * 2;
246             SymbolTableymbolTable">SymbolTable[] temp = new SymbolTable[new_len];
247             System.arraycopy(_symbol_table_stack, 0, temp, 0, _symbol_table_stack.length);
248             _symbol_table_stack = temp;
249         }
250         _symbol_table_stack[_symbol_table_top++] = symbols;
251     }
252     @Override
253     public SymbolTable pop_passed_symbol_table()
254     {
255         if (_symbol_table_top <= 0) {
256             return null;
257         }
258         _symbol_table_top--;
259         SymbolTable symbols = _symbol_table_stack[_symbol_table_top];
260         _symbol_table_stack[_symbol_table_top] = null;
261         return symbols;
262     }

276         IonReaderTextSpan(IonReaderTextUserX reader)
277         {
278             // TODO: convert _start_char_offset from a long and data page
279             //       to be an abstract reference into the Unified* data source
280 
281             UnifiedInputStreamX current_stream = reader._scanner.getSourceStream();
282             //
283             // TODO: this page isn't safe, except where we have only a single

286             _data_page = current_stream._buffer.getCurrentPage();
287             _container_type = reader.getContainerType();

349     private void hoistImpl(Span span)
350     {
351         if (!(span instanceof IonReaderTextSpan)) {
352             throw new IllegalArgumentException("position must match the reader");
353         }
354         IonReaderTextSpan text_span = (IonReaderTextSpan)span;
355 
356         UnifiedInputStreamX current_stream = _scanner.getSourceStream();
357         UnifiedDataPageX    curr_page      = text_span.getDataPage();
358         int                 array_offset   = (int)text_span._start_offset + _physical_start_offset;
359         int                 page_limit     = curr_page._page_limit;
360         int                 array_length   = page_limit - array_offset;
361 
362         // we're going to cast this value down.  Since we only support
363         // in memory single buffered chars here this is ok.
364         assert(text_span.getStartOffset() <= Integer.MAX_VALUE);
365 
366         // Now - create a new stream
367         // TODO: this is a pretty expensive way to do this. UnifiedInputStreamX
368         //       needs to have a reset method added that can reset the position
369         //       and length of the input to be some subset of the original source.
370         //       This would avoid a lot of object creation (and wasted destruction.
371         //       But this is a time-to-market solution here.  The change can be
372         //       made as support for streams is added.
373         UnifiedInputStreamX iis;
374         if (current_stream._is_byte_data) {
375             byte[] bytes = current_stream.getByteArray();
376             assert(bytes != null);
377             iis = UnifiedInputStreamX.makeStream(
378                                             bytes
379                                           , array_offset
380                                           , array_length
381                                       );
382         }
383         else {
384             char[] chars = current_stream.getCharArray();
385             assert(chars != null);
386             iis = UnifiedInputStreamX.makeStream(
387                                             chars
388                                           , array_offset
389                                           , array_length
390                                       );
391         }
392         IonType container = text_span.getContainerType();
393         re_init(iis, container, text_span._start_line, text_span._start_column);
394     }

402     {
403         if (facetType == SpanProvider.class)
404         {
405             return facetType.cast(new SpanProviderFacet());
