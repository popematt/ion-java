213     private final static int[] makeHexValueArray() {
214         int[] hex = new int[256];
215         for (int ii=0; ii<256; ii++) {
216             hex[ii] = -1;
217         }
218         for (int ii='0'; ii<='9'; ii++) {
219             hex[ii] = ii - '0';
220         }
221         for (int ii='a'; ii<='f'; ii++) {
222             hex[ii] = ii - 'a' + 10;
223         }
224         for (int ii='A'; ii<='F'; ii++) {
225             hex[ii] = ii - 'A' + 10;
226         }
227         return hex;
228     }

294     private static final int escapeCharactersValues[] = makeEscapeCharacterValuesArray();
295     private static final int[] makeEscapeCharacterValuesArray() {
296         int[] values = new int[256];
297         for (int ii=0; ii<256; ii++) {
298             values[ii] = ESCAPE_NOT_DEFINED;
299         }
300         values['0'] = 0;        //    \u0000  \0  alert NUL
301         values['a'] = 7;        //    \u0007  \a  alert BEL
302         values['b'] = 8;        //    \u0008  \b  backspace BS
303         values['t'] = 9;        //    \u0009  \t  horizontal tab HT
304         values['n'] = '\n';     //    \ u000A  \ n  linefeed LF
305         values['f'] = 0x0c;     //    \u000C  \f  form feed FF
306         values['r'] = '\r';     //    \ u000D  \ r  carriage return CR
307         values['v'] = 0x0b;     //    \u000B  \v  vertical tab VT
308         values['"'] = '"';      //    \u0022  \"  double quote
309         values['\''] = '\'';    //    \u0027  \'  single quote
310         values['?'] = '?';      //    \u003F  \?  question mark
311         values['\\'] = '\\';    //    \u005C  \\  backslash
312         values['/'] = '/';      //    \u002F  \/  forward slash nothing  \NL  escaped NL expands to nothing
313         values['\n'] = ESCAPE_REMOVES_NEWLINE;  // slash-new line the new line eater
314         values['\r'] = ESCAPE_REMOVES_NEWLINE2;  // slash-new line the new line eater
315         values['x'] = ESCAPE_HEX; //    any  \xHH  2-digit hexadecimal unicode character equivalent to \ u00HH
316         values['u'] = ESCAPE_LITTLE_U; //    any  \ uHHHH  4-digit hexadecimal unicode character
317         values['U'] = ESCAPE_BIG_U;
318         return values;

387     public static String escapeSequence(int c) {
388         if (c >= 0 || c <= 0x10FFFF) {
389             if (c < 128) {
390                 return escapeCharacterImage[c];
391             }
392             if (c < 0xFFFF) {
393                 String short_hex = Integer.toHexString(c);
394                 int    short_len = short_hex.length();
395                 if (short_len < 4) {
396                     short_hex = "0000".substring(short_len);
397                 }
398                 return "\\u"+short_hex;
399             }
400             if (c < 0xFFFF) {
401                 String long_hex = Integer.toHexString(c);
402                 int    long_len = long_hex.length();
403                 if (long_len < 4) {
404                     long_hex = "00000000".substring(long_len);
405                 }
406                 return "\\U"+long_hex;
407             }
408         }
409         throw new IllegalArgumentException("the value "+c+" isn't a valid character");
410     }

508     public static int decodeSid(CharSequence sidToken)
509     {
510         assert sidToken.charAt(0) == '$';
511 
512         int length = sidToken.length();
513         assert length > 1;
514 
515         String digits = sidToken.subSequence(1, length).toString();
516 
517         try {
518             return Integer.parseInt(digits);
519         } catch (Exception e) {
520             throw new IonException(String.format("Unable to parse SID %s", digits), e);
521         }
522     }

524     static public int keyword(CharSequence word, int start_word, int end_word)
525     {
526         int c = word.charAt(start_word);
527         int len = end_word - start_word; // +1 but we build that into the constants below
528         switch (c) {
529         case '$':
530             if (len > 1) {
531                 for (int i = start_word + 1; i < end_word; i++) {
532                     if (! isDigit(word.charAt(i))) return -1;
533                 }
534                 return KEYWORD_sid;
535             }
536             return -1;
537         case 'b':
538             if (len == 4) {
539                 if (word.charAt(start_word+1) == 'o'
540                  && word.charAt(start_word+2) == 'o'
541                  && word.charAt(start_word+3) == 'l'
542                 ) {
543                     return KEYWORD_BOOL;
544                 }
545                 if (word.charAt(start_word+1) == 'l'
546                  && word.charAt(start_word+2) == 'o'
547                  && word.charAt(start_word+3) == 'b'
548                 ) {
549                     return KEYWORD_BLOB;
550                 }
551             }
552             return -1;
553         case 'c':
554             if (len == 4) {
555                 if (word.charAt(start_word+1) == 'l'
556                  && word.charAt(start_word+2) == 'o'
557                  && word.charAt(start_word+3) == 'b'
558                 ) {
559                     return KEYWORD_CLOB;
560                 }
561             }
562             return -1;
563         case 'd':
564             if (len == 7) {
565                 if (word.charAt(start_word+1) == 'e'
566                  && word.charAt(start_word+2) == 'c'
567                  && word.charAt(start_word+3) == 'i'
568                  && word.charAt(start_word+4) == 'm'
569                  && word.charAt(start_word+5) == 'a'
570                  && word.charAt(start_word+6) == 'l'
571                 ) {
572                     return KEYWORD_DECIMAL;
573                 }
574             }
575             return -1;
576         case 'f':
577             if (len == 5) {
578                 if (word.charAt(start_word+1) == 'a'
579                  && word.charAt(start_word+2) == 'l'
580                  && word.charAt(start_word+3) == 's'
581                  && word.charAt(start_word+4) == 'e'
582                 ) {
583                     return KEYWORD_FALSE;
584                 }
585                 if (word.charAt(start_word+1) == 'l'
586                  && word.charAt(start_word+2) == 'o'
587                  && word.charAt(start_word+3) == 'a'
588                  && word.charAt(start_word+4) == 't'
589                 ) {
590                     return KEYWORD_FLOAT;
591                 }
592             }
593             return -1;
594         case 'i':
595             if (len == 3) {
596                 if (word.charAt(start_word+1) == 'n') {
597                     if (word.charAt(start_word+2) == 't') {
598                         return KEYWORD_INT;
599                     }
600                 }
601             }
602             return -1;
603         case 'l':
604             if (len == 4) {
605                 if (word.charAt(start_word+1) == 'i'
606                  && word.charAt(start_word+2) == 's'
607                  && word.charAt(start_word+3) == 't'
608                 ) {
609                     return KEYWORD_LIST;
610                 }
611             }
612             return -1;
613         case 'n':
614             if (len == 4) {
615                 if (word.charAt(start_word+1) == 'u'
616                  && word.charAt(start_word+2) == 'l'
617                  && word.charAt(start_word+3) == 'l'
618                 ) {
619                     return KEYWORD_NULL;
620                 }
621             }
622             else if (len == 3) {
623                 if (word.charAt(start_word+1) == 'a'
624                     && word.charAt(start_word+2) == 'n'
625                    ) {
626                        return KEYWORD_NAN;
627                    }
628             }
629             return -1;
630         case 's':
631             if (len == 4) {
632                 if (word.charAt(start_word+1) == 'e'
633                  && word.charAt(start_word+2) == 'x'
634                  && word.charAt(start_word+3) == 'p'
635                 ) {
636                     return KEYWORD_SEXP;
637                 }
638             }
639             else if (len == 6) {
640                 if (word.charAt(start_word+1) == 't'
641                  && word.charAt(start_word+2) == 'r'
642                 ) {
643                     if (word.charAt(start_word+3) == 'i'
644                      && word.charAt(start_word+4) == 'n'
645                      && word.charAt(start_word+5) == 'g'
646                     ) {
647                         return KEYWORD_STRING;
648                     }
649                     if (word.charAt(start_word+3) == 'u'
650                      && word.charAt(start_word+4) == 'c'
651                      && word.charAt(start_word+5) == 't'
652                     ) {
653                         return KEYWORD_STRUCT;
654                     }
655                     return -1;
656                 }
657                 if (word.charAt(start_word+1) == 'y'
658                  && word.charAt(start_word+2) == 'm'
659                  && word.charAt(start_word+3) == 'b'
660                  && word.charAt(start_word+4) == 'o'
661                  && word.charAt(start_word+5) == 'l'
662                 ) {
663                     return KEYWORD_SYMBOL;
664                 }
665             }
666             return -1;
667         case 't':
668             if (len == 4) {
669                 if (word.charAt(start_word+1) == 'r'
670                  && word.charAt(start_word+2) == 'u'
671                  && word.charAt(start_word+3) == 'e'
672                 ) {
673                     return KEYWORD_TRUE;
674                 }
675             }
676             else if (len == 9) {
677                 if (word.charAt(start_word+1) == 'i'
678                  && word.charAt(start_word+2) == 'm'
679                  && word.charAt(start_word+3) == 'e'
680                  && word.charAt(start_word+4) == 's'
681                  && word.charAt(start_word+5) == 't'
682                  && word.charAt(start_word+6) == 'a'
683                  && word.charAt(start_word+7) == 'm'
684                  && word.charAt(start_word+8) == 'p'
685                 ) {
686                     return KEYWORD_TIMESTAMP;
687                 }
688             }
689             return -1;
690         default:
691             return -1;
692         }
693     }

695     /*
696     KW_BIT_BLOB       = 0x0001;
697     KW_BIT_BOOL       = 0x0002;
698     KW_BIT_CLOB       = 0x0004;
699     KW_BIT_DECIMAL    = 0x0008;
700     KW_BIT_FLOAT      = 0x0010;
701     KW_BIT_INT        = 0x0020;
702     KW_BIT_LIST       = 0x0030;
703     KW_BIT_NULL       = 0x0080;
704     KW_BIT_SEXP       = 0x0100;
705     KW_BIT_STRING     = 0x0200;
706     KW_BIT_STRUCT     = 0x0400;
707     KW_BIT_SYMBOL     = 0x0800;
708     KW_BIT_TIMESTAMP  = 0x1000;
709     datagram is not included (since these are types that can appear as a null type)
710 
711     A  1    F  6    N 11    T 16
712     B  2    G  7    O 12    U 17
713     C  3    I  8    P 13    X 18
714     D  4    L  9    R 14    Y 19
715     E  5    M 10    S 15
716 
717     */
718     // this array is [pos][letter_idx] - if this letter is
719     // valid in this position then all the type names that
720     // contain the letter at that position are on.  Each
721     // keyword is assigned a bit value.
722     // however due to the double index expense the index
723     // is represented by a 1 dim int array and fn's used
724     // to access it (since inline-ing final static small methods
725     // is easy for the compiler)
726     static final int  TN_MAX_NAME_LENGTH = "TIMESTAMP".length() + 1; // so anything too long will be 0
727     static final int  TN_LETTER_MAX_IDX = 19;
728 
729     static final int  KW_BIT_BLOB       = 0x0001;
730     static final int  KW_BIT_BOOL       = 0x0002;
731     static final int  KW_BIT_CLOB       = 0x0004;
732     static final int  KW_BIT_DECIMAL    = 0x0008;
733     static final int  KW_BIT_FLOAT      = 0x0010;
734     static final int  KW_BIT_INT        = 0x0020;
735     static final int  KW_BIT_LIST       = 0x0030;
736     static final int  KW_BIT_NULL       = 0x0080;
737     static final int  KW_BIT_SEXP       = 0x0100;
738     static final int  KW_BIT_STRING     = 0x0200;
739     static final int  KW_BIT_STRUCT     = 0x0400;
740     static final int  KW_BIT_SYMBOL     = 0x0800;
741     static final int  KW_BIT_TIMESTAMP  = 0x1000;
742     static final int  KW_ALL_BITS       = 0x1fff;
743 
744     static final int[] typeNameBits = new int[] {
745         KW_BIT_BLOB,      KW_BIT_BOOL,    KW_BIT_CLOB,
746         KW_BIT_DECIMAL,   KW_BIT_FLOAT,   KW_BIT_INT,
747         KW_BIT_LIST,      KW_BIT_NULL,    KW_BIT_SEXP,
748         KW_BIT_STRING,    KW_BIT_STRUCT,  KW_BIT_SYMBOL,
749         KW_BIT_TIMESTAMP
750     };

752     static final String[] typeNameNames = new String[] {
753         "blob",           "bool",         "clob",
754         "decimal",        "float",        "int",
755         "list",           "null",         "sexp",
756         "string",         "struct",       "symbol",
757         "timestamp"
758     };
759     static final int[] typeNameKeyWordIds = new int[] {
760         KEYWORD_BLOB,     KEYWORD_BOOL,   KEYWORD_CLOB,
761         KEYWORD_DECIMAL,  KEYWORD_FLOAT,  KEYWORD_INT,
762         KEYWORD_LIST,     KEYWORD_NULL,   KEYWORD_SEXP,
763         KEYWORD_STRING,   KEYWORD_STRUCT, KEYWORD_SYMBOL,
764         KEYWORD_TIMESTAMP
765     };

767     static final int[] TypeNameBitIndex = makeTypeNameBitIndex();
768     static final int[] makeTypeNameBitIndex() {
769         int[] bits = new int[TN_MAX_NAME_LENGTH*TN_LETTER_MAX_IDX];
770         int   typename_bit;
771         for (int tt=0; tt<typeNameNames.length; tt++) {
772             String typename = typeNameNames[tt];
773             typename_bit = typeNameBits[tt];
774             for (int ii=0; ii<typename.length(); ii++) {
775                 int c = typename.charAt(ii);
776                 int l = typeNameLetterIdx(c);
777                 assert(l > 0);
778                 typename_set_bit(bits, ii, l, typename_bit);
779             }
780         }
781         return bits;
782     }
783     private final static void typename_set_bit(int[] bits, int ii, int l, int typename_bit)
784     {
785         // bits[ii][l]
786         int idx = (ii * TN_LETTER_MAX_IDX) + l - 1;
787         bits[idx] |= typename_bit;
788     }
789     public final static int typeNameLetterIdx(int c) {
790         switch(c) {
791         case 'a': return  1;
792         case 'b': return  2;
793         case 'c': return  3;
794         case 'd': return  4;
795         case 'e': return  5;
796         case 'f': return  6;
797         case 'g': return  7;
798         case 'i': return  8;
799         case 'l': return  9;
800         case 'm': return 10;
801         case 'n': return 11;
802         case 'o': return 12;
803         case 'p': return 13;
804         case 'r': return 14;
805         case 's': return 15;
806         case 't': return 16;
807         case 'u': return 17;
808         case 'x': return 18;
809         case 'y': return 19;
810         default:  return -1;
811         }
812     }
813     public final static int typeNamePossibilityMask(int pos, int letter_idx) {
814         int idx = (pos * TN_LETTER_MAX_IDX) + letter_idx - 1;
815         int mask = TypeNameBitIndex[idx];
816         return mask;
817     }
818     // this can be faster but it's pretty unusual to be called.
819     public final static int typeNameKeyWordFromMask(int possible_names, int length) {
820         int kw = KEYWORD_unrecognized;
821         if (possible_names != IonTokenConstsX.KW_ALL_BITS) {
822             for (int ii=0; ii<typeNameBits.length; ii++) {
823                 int tb = typeNameBits[ii];
824                 if (tb == possible_names) {
825                     if (typeNameNames[ii].length() == length) {
826                         kw = typeNameKeyWordIds[ii];
827                     }
828                     break;
829                 }
830             }
831         }
832         return kw;
833     }
834     public static final String getNullImage(IonType type)
835     {
836         String nullimage = null;
837 
838         switch (type) {
839         case NULL:      nullimage = "null";           break;
840         case BOOL:      nullimage = "null.bool";      break;
841         case INT:       nullimage = "null.int";       break;
842         case FLOAT:     nullimage = "null.float";     break;
843         case DECIMAL:   nullimage = "null.decimal";   break;
844         case TIMESTAMP: nullimage = "null.timestamp"; break;
845         case SYMBOL:    nullimage = "null.symbol";    break;
846         case STRING:    nullimage = "null.string";    break;
847         case BLOB:      nullimage = "null.blob";      break;
848         case CLOB:      nullimage = "null.clob";      break;
849         case SEXP:      nullimage = "null.sexp";      break;
850         case LIST:      nullimage = "null.list";      break;
851         case STRUCT:    nullimage = "null.struct";    break;
852 
853         default: throw new IllegalStateException("unexpected type " + type);
854         }
855 
856         return nullimage;
857     }
858     public static final IonType getNullType(CharSequence s)
859     {
860         IonType type = null;
861         int     c, ii = 0;
862         boolean stop = false;
863         while (!stop && ii<s.length()) {
864             c = s.charAt(ii++);
865             switch (c) {
866             case ' ': case '\t': case '\r': case '\n':
867                 break;
868             case 'n':
869                 stop = true;
870                 break;
871             default:
872                 invalid_null_image(s);
873             }
874         }
875         if (ii>=s.length() || s.charAt(ii++) != 'u') invalid_null_image(s);
876         if (ii>=s.length() || s.charAt(ii++) != 'l') invalid_null_image(s);
877         if (ii>=s.length() || s.charAt(ii++) != 'l') invalid_null_image(s);
878         boolean dot = false;
879         while (!dot && ii<s.length()) {
880             c = s.charAt(ii++);
881             switch (c) {
882             case ' ': case '\t': case '\r': case '\n':
883                 break;
884             case '.':
885                 dot = true;
886                 break;
887             default:
888                 invalid_null_image(s);
889             }
890         }
891         if (dot) {
892             int kw = IonTokenConstsX.keyword(s, ii, s.length());
893             switch (kw) {
894                 case IonTokenConstsX.KEYWORD_NULL:
895                     type = IonType.NULL;
896                     break;
897                 case IonTokenConstsX.KEYWORD_BOOL:
898                     type = IonType.BOOL;
899                     break;
900                 case IonTokenConstsX.KEYWORD_INT:
901                     type = IonType.INT;
902                     break;
903                 case IonTokenConstsX.KEYWORD_FLOAT:
904                     type = IonType.FLOAT;
905                     break;
906                 case IonTokenConstsX.KEYWORD_DECIMAL:
907                     type = IonType.DECIMAL;
908                     break;
909                 case IonTokenConstsX.KEYWORD_TIMESTAMP:
910                     type = IonType.TIMESTAMP;
911                     break;
912                 case IonTokenConstsX.KEYWORD_SYMBOL:
913                     type = IonType.SYMBOL;
914                     break;
915                 case IonTokenConstsX.KEYWORD_STRING:
916                     type = IonType.STRING;
917                     break;
918                 case IonTokenConstsX.KEYWORD_CLOB:
919                     type = IonType.CLOB;
920                     break;
921                 case IonTokenConstsX.KEYWORD_BLOB:
922                     type = IonType.BLOB;
923                     break;
924                 case IonTokenConstsX.KEYWORD_STRUCT:
925                     type = IonType.STRUCT;
926                     break;
927                 case IonTokenConstsX.KEYWORD_LIST:
928                     type = IonType.LIST;
929                     break;
930                 case IonTokenConstsX.KEYWORD_SEXP:
931                     type = IonType.SEXP;
932                     break;
933                 default:
934                     invalid_null_image(s);
935             }
936         }
937 
938         while (ii<s.length()) {
939             c = s.charAt(ii++);
940             switch (c) {
941             case ' ': case '\t': case '\r': case '\n':
942                 break;
943             default:
944                 invalid_null_image(s);
945             }
946         }
947 
948         return type;
949     }
950     private static void invalid_null_image(CharSequence s) {
951         throw new CantConvertException("invalid image "+s.toString());
952     }
