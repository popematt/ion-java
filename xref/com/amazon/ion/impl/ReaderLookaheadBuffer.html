1   package com.amazon.ion.impl;

3   import com.amazon.ion.IonReader;
4   import com.amazon.ion.system.IonReaderBuilder;
5   
6   /**
7    * Interface for lookahead buffers that enable incremental reading of streaming data.
8    */
9   public interface ReaderLookaheadBuffer {
10  
11      /**
12       * If possible, fills the input pipe with enough bytes to enable one more successful call to
13       * {@link IonReader#next()} on the non-incremental reader attached to the pipe. If not enough bytes were available
14       * in the raw input stream to complete the next top-level user value, calling {@link #moreDataRequired()} after this
15       * method returns will return `true`. In this case, this method must be called again before calling
16       * `IonReader.next()` to position the reader on this value. Otherwise, `moreDataRequired()` will return `false` and
17       * a call to `IonReader.next()` may be made to position the reader on this value. Implementations may throw
18       * `IonException` if invalid Ion data is detected. Implementations may define exceptional cases.
19       * @throws Exception if an IOException is thrown by the underlying InputStream.
20       */
21      void fillInput() throws Exception;
22  
23      /**
24       * Indicates whether more data must become present in the raw input stream before a successful call to
25       * {@link IonReader#next()} may be made to position the reader on the most-recently-buffered value.
26       * @return true if more data is required to complete the next top-level user value; otherwise, false.
27       */
28      boolean moreDataRequired();
29  
30      /**
31       * Indicates how many bytes are currently stored in the internal buffer. This can be used to detect whether
32       * calls to fillInput() are successfully retrieving data.
33       * @return The number of bytes waiting in the input buffer.
34       */
35      int available();
36  
37      /**
38       * Builds a reader over the input pipe. NOTE: because IonReader construction consumes bytes from the stream
39       * (to determine whether the stream is binary or text Ion), {@link #fillInput()} must have been called (such that
40       * {@link #moreDataRequired()} returns `false`)  before calling this method. For the same reason, this method must
41       * only be called once per instance of this class.
42       * @param builder the builder containing the reader's configuration.
43       * @return a new IonReader.
44       */
45      IonReader newIonReader(IonReaderBuilder builder);
46  }
