16  package com.amazon.ion.impl;
17  
18  import com.amazon.ion.Decimal;
19  import com.amazon.ion.Timestamp;
20  import java.io.IOException;
21  /**
22   * Interface to read bytes over a variety of sources.
23   * Avoiding the Java standard overhead in so far as possible.
24   * This requires the ability to access any part of the underlying
25   * data source randomly.  However forward access is expected to be
26   * the standard direction.
27   */
28  interface ByteReader
29  {
30      /**
31       * yet another definition of eof of file as -1
32       */
33      public static final int EOF = -1;
34  
35      /**
36       * returns the current position of the reader in the underlying
37       * byte buffer
38       * @return offset of the next byte to read
39       */
40      public int  position();
41      /**
42       * sets the position of the reader so that it will read the
43       * byte at the passed in offset (newPosition) on the next call
44       * to read()
45       * @param newPosition offset to position the reader to
46       */
47      public void position(int newPosition);
48      /**
49       * moves the reader forward length bytes.  This is typically
50       * faster than read as it has no need to actually move any
51       * bytes of data.  It may error if the skip would take the
52       * current position off the end of the buffer.
53       * @param length number of bytes to skip
54       */
55      public void skip(int length);
56  
57      /**
58       * reads a single byte and returns it as an unsigned value,
59       * that is from 0 to 255.  This returns a ByteReader.EOF (-1)
60       * if there is not data to read.
61       * @return next byte of data
62       * @throws IOException
63       */
64      public int  read() throws IOException;
65      /**
66       * reads a number of bytes into the callers supplied buffer (dst).
67       * This will return the number of bytes read, if any.
68       * @param dst callers destination byte array
69       * @param start offset of the first destination array element to write into
70       * @param len maximum number of bytes to copy
71       * @return number of bytes actually copied
72       * @throws IOException
73       */
74      public int  read(byte[] dst, int start, int len) throws IOException;
75  
76      /**
77       * reads the next byte as a typedesc bytes
78       * @return typedesc byte
79       * @throws IOException
80       */
81      public int  readTypeDesc() throws IOException;
82  
83      /**
84       * reads the upcoming bytes as a signed long.  This uses
85       * 7 bits per byte, expects the bytes in bigendian order, and
86       * looks for the high order bit (0x80) to be set to mark the
87       * end of the value. It expects the 2nd bit (0x40 of the first
88       * byte) to be the sign of the value.  The remaining bits are
89       * the absolute value of the orginal value. It
90       * returns the aggregated bytes as a Java long (signed 64 bit
91       * value) correctly signed.
92       * @return long value read
93       * @throws IOException
94       */
95      public long readVarLong() throws IOException;
96      /**
97       * reads the upcoming bytes as a unsigned long.  This uses
98       * 7 bits per byte, expects the bytes in bigendian order, and
99       * looks for the high order bit (0x80) to be set to mark the
100      * end of the value. It does not include a sign bit so all 7 bits
101      * in the first byte are treated as positive. It
102      * returns the aggregated bytes as a Java long (signed 64 bit
103      * value).
104      * @return long value read
105      * @throws IOException
106      */
107     public long readVarULong() throws IOException;
108     /**
109      * reads the next len bytes as a signed long.  This uses
110      * 8 bits per byte and expects the bytes in bigendian order. It
111      * returns the aggregated bytes as a Java long (signed 64 bit
112      * value).
113      * @param len number of bytes to read into the long
114      * @return long value read
115      * @throws IOException
116      */
117     public long readULong(int len) throws IOException;
118 
119     /**
120      * reads the upcoming bytes as a signed int.  This uses
121      * 7 bits per byte, expects the bytes in bigendian order, and
122      * looks for the high order bit (0x80) to be set to mark the
123      * end of the value. It expects the 2nd bit (0x40 of the first
124      * byte) to be the sign of the value.  The remaining bits are
125      * the absolute value of the orginal value. It
126      * returns the aggregated bytes as a Java long (signed 32 bit
127      * value) correctly signed.
128      * @return int value read
129      * @throws IOException
130      */
131     public int  readVarInt() throws IOException;
132     /**
133      * reads the upcoming bytes as a unsigned int.  This uses
134      * 7 bits per byte, expects the bytes in bigendian order, and
135      * looks for the high order bit (0x80) to be set to mark the
136      * end of the value. It does not include a sign bit so all 7 bits
137      * in the first byte are treated as positive. It
138      * returns the aggregated bytes as a Java long (signed 32 bit
139      * value).
140      * @return int value read
141      * @throws IOException
142      */
143     public int  readVarUInt() throws IOException;
144 
145     /**
146      * reads len bytes and treats them as an IEEE 64 binary
147      * floating point value.  Currently the only valid values for
148      * len are 0 (which returns 0.0e1) or 8 which is treated
149      * as a 64 bit binary float and returned.
150      * @param len 0 or 8
151      * @return 0.0 or the double represented by the next 8 bytes
152      * @throws IOException
153      */
154     public double readFloat(int len) throws IOException;
155     /**
156      * reads the next len bytes and converts them to a {@link Decimal} value.
157      * @param len number of bytes over which the value extends
158      * @return Decimal object initialized to the represented value; not null.
159      * @throws IOException
160      */
161     public Decimal readDecimal(int len) throws IOException;
162 
163     /**
164      * Reads the next len bytes and returns them as a {@link Timestamp}
165      * object.  This contains an UTC date and time value and a
166      * timezone offset of the original value.
167      *
168      * @param len
169      * @return a new {@link Timestamp} instance,
170      * or {@code null} if the encoded value is {@code null.timestamp}.
171      * @throws IOException
172      */
173     public Timestamp readTimestamp(int len) throws IOException;
174 
175     /** reads the next len bytes and converts them from UTF-8 into Java
176      * characters and returns those characters as a String.  This may throw
177      * an exception in the event it encounters byte sequences that are not
178      * value UTF-8.  This converts some Unicode values into character pairs
179      * when the value requires the use of surrogate pairs in Unicode 16.
180      * @param len number of bytes to read
181      * @return resultant Java string
182      * @throws IOException
183      */
184     public String readString(int len) throws IOException;
185 }
