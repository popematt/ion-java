8   import java.nio.charset.CharsetEncoder;

11  /**
12   * Encodes {@link String}s to UTF-8. Instances of this class are reusable but are NOT threadsafe.
13   *
14   * Users are strongly encouraged to get instances from {@link Utf8StringEncoderPool#getOrCreateUtf8Encoder()}.
15   * {@link #encode(String)} can be called any number of times. Users are expected to call {@link #close()} when
16   * the encoder is no longer needed.
17   */
18  public class Utf8StringEncoder implements Closeable {
19      // The longest String (as measured by {@link java.lang.String#length()}) that this instance can encode without
20      // requiring additional allocations.
21      private static final int SMALL_STRING_SIZE = 4 * 1024;
22  
23      // Reusable resources for encoding Strings as UTF-8 bytes
24      final Utf8StringEncoderPool utf8StringEncoderPool;
25      final CharsetEncoder utf8Encoder;
26      final ByteBuffer utf8EncodingBuffer;
27      final char[] charArray;
28      final CharBuffer charBuffer;
29  
30      public Utf8StringEncoder(Utf8StringEncoderPool pool) {
31          utf8StringEncoderPool = pool;
32          utf8Encoder = Charset.forName("UTF-8").newEncoder();
33          utf8EncodingBuffer = ByteBuffer.allocate((int) (SMALL_STRING_SIZE * utf8Encoder.maxBytesPerChar()));
34          charArray = new char[SMALL_STRING_SIZE];
35          charBuffer = CharBuffer.wrap(charArray);
36      }
37  
38      public Utf8StringEncoder() {
39          // This instance is not associated with a Utf8StringEncoderPool
40          this(null);
41      }
42  
43      /**
44       * Encodes the provided String's text to UTF-8. Unlike {@link String#getBytes(Charset)}, this method will not
45       * silently replace characters that cannot be encoded with a substitute character. Instead, it will throw
46       * an {@link IllegalArgumentException}.
47       *
48       * Some resources in the returned {@link Result} may be reused across calls to this method. Consequently,
49       * callers should use the Result and discard it immediately.
50       *
51       * @param text A Java String to encode as UTF8 bytes.
52       * @return  A {@link Result} containing a byte array of UTF-8 bytes and encoded length.
53       * @throws IllegalArgumentException if the String cannot be encoded as UTF-8.
54       */
55      public Result encode(String text) {
56          /*
57           This method relies on the standard CharsetEncoder class to encode each String's UTF-16 char[] data into
58           UTF-8 bytes. Strangely, CharsetEncoders cannot operate directly on instances of a String. The CharsetEncoder
59           API requires all inputs and outputs to be specified as instances of java.nio.ByteBuffer and
60           java.nio.CharBuffer, making some number of allocations mandatory. Specifically, for each encoding operation
61           we need to have:
62  
63              1. An instance of a UTF-8 CharsetEncoder.
64              2. A CharBuffer representation of the String's data.
65              3. A ByteBuffer into which the CharsetEncoder may write UTF-8 bytes.
66  
67           To minimize the overhead involved, the Utf8StringEncoder will reuse previously initialized resources wherever
68           possible. However, because CharBuffer and ByteBuffer each have a fixed length, we can only reuse them for
69           Strings that are small enough to fit. This creates two kinds of input String to encode: those that are small
70           enough for us to reuse our buffers ("small strings"), and those which are not ("large strings").
71  
72           The String#getBytes(Charset) method cannot be used for two reasons:
73  
74                 1. It always allocates, so we cannot reuse any resources.
75                 2. If/when it encounters character data that cannot be encoded as UTF-8, it simply replaces that data
76                   with a substitute character[1]. (Sometimes seen in applications as a '?'.) In order
77                   to surface invalid data to the user, the method must be able to detect these events at encoding time.
78  
79              [1] https://en.wikipedia.org/wiki/Substitute_character
80          */
81  
82          CharBuffer stringData;
83          ByteBuffer encodingBuffer;
84  
85          int length = text.length();
86  
87          // While it is technically possible to encode any String using a fixed-size encodingBuffer, we need
88          // to be able to write the length of the complete UTF-8 string to the output stream before we write the string
89          // itself. For simplicity, we reuse or create an encodingBuffer that is large enough to hold the full string.
90  
91          // In order to encode the input String, we need to pass it to CharsetEncoder as an implementation of CharBuffer.
92          // Surprisingly, the intuitive way to achieve this (the CharBuffer#wrap(CharSequence) method) adds a large
93          // amount of CPU overhead to the encoding process. Benchmarking shows that it's substantially faster
94          // to use String#getChars(int, int, char[], int) to copy the String's backing array and then call
95          // CharBuffer#wrap(char[]) on the copy.
96  
97          if (length > SMALL_STRING_SIZE) {
98              // Allocate a new buffer for large strings
99              encodingBuffer = ByteBuffer.allocate((int) (text.length() * utf8Encoder.maxBytesPerChar()));
100             char[] chars = new char[text.length()];
101             text.getChars(0, text.length(), chars, 0);
102             stringData = CharBuffer.wrap(chars);
103         } else {
104             // Reuse our existing buffers for small strings
105             encodingBuffer = utf8EncodingBuffer;
106             encodingBuffer.clear();
107             stringData = charBuffer;
108             text.getChars(0, text.length(), charArray, 0);
109             charBuffer.rewind();
110             charBuffer.limit(text.length());
111         }
112 
113         // Because encodingBuffer is guaranteed to be large enough to hold the encoded string, we can
114         // perform the encoding in a single call to CharsetEncoder#encode(CharBuffer, ByteBuffer, boolean).
115         CoderResult coderResult = utf8Encoder.encode(stringData, encodingBuffer, true);
116 
117         // 'Underflow' is the success state of a CoderResult.
118         if (!coderResult.isUnderflow()) {
119             throw new IllegalArgumentException("Could not encode string as UTF8 bytes: " + text);
120         }
121         encodingBuffer.flip();
122         int utf8Length = encodingBuffer.remaining();
123 
124         // In most usages, the JVM should be able to eliminate this allocation via an escape analysis of the caller.
125         return new Result(utf8Length, encodingBuffer.array());
126     }
127 
128     /**
129      * Attempts to return this instance to the Utf8StringEncoderPool with which it is associated, if any.
130      *
131      * Do not continue to use this encoder after calling this method.
132      */
133     @Override
134     public void close() {
135         if (utf8StringEncoderPool != null) {
136             utf8StringEncoderPool.returnEncoderToPool(this);
137         }
138     }

140     /**
141      * Represents the result of a {@link Utf8StringEncoder#encode(String)} operation.
142      */
143     public static class Result {
144         final private byte[] buffer;
145         final private int encodedLength;
146 
147         public Result(int encodedLength, byte[] buffer) {
148             this.encodedLength = encodedLength;
149             this.buffer = buffer;
150         }
151 
152         /**
153          * Returns a byte array containing the encoded UTF-8 bytes starting at index 0. This byte array is NOT
154          * guaranteed to be the same length as the data it contains. Callers must use {@link #getEncodedLength()}
155          * to determine the number of bytes that should be read from the byte array.
156          *
157          * @return the buffer containing UTF-8 bytes.
158          */
159         public byte[] getBuffer() {
160             return buffer;
161         }
162 
163         /**
164          * @return the number of encoded bytes in the array returned by {@link #getBuffer()}.
165          */
166         public int getEncodedLength() {
167             return encodedLength;
168         }
169     }
