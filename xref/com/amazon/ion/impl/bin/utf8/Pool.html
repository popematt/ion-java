1   package com.amazon.ion.impl.bin.utf8;
2   
3   import java.util.Queue;
4   import java.util.concurrent.ConcurrentLinkedQueue;
5   import java.util.concurrent.atomic.AtomicInteger;
6   
7   abstract class Pool<T extends Poolable<?>> {
8   
9       /**
10       * Allocates objects to be pooled.
11       * @param <T> the type of object.
12       */
13      interface Allocator<T extends Poolable<?>> {
14  
15          /**
16           * Allocate a new object and link it to the given pool.
17           * @param pool the pool to which the new object will be linked.
18           * @return a new instance.
19           */
20          T newInstance(Pool<T> pool);
21      }
22  
23      // The maximum number of objects that can be waiting in the queue before new ones will be discarded.
24      private static final int MAX_QUEUE_SIZE = 128;
25  
26      // A queue of previously initialized objects that can be loaned out.
27      private final Queue<T> objectQueue;
28  
29      // The current size of the queue. Note: some implementations of Queue.size() (including ConcurrentLinkedQueue's)
30      // are not constant-time operations. Tracking the size externally is a performance optimization.
31      private final AtomicInteger size;
32  
33      // Allocator of objects to be pooled.
34      private final Allocator<T> allocator;
35  
36      Pool(Allocator<T> allocator) {
37          this.allocator = allocator;
38          objectQueue = new ConcurrentLinkedQueue<T>();
39          size = new AtomicInteger(0);
40      }
41  
42      /**
43       * If the pool is not empty, removes an object from the pool and returns it;
44       * otherwise, constructs a new object.
45       *
46       * @return An object.
47       */
48      public T getOrCreate() {
49          // The `poll` method does not block. If the queue is empty it returns `null` immediately.
50          T object = objectQueue.poll();
51          if (object == null) {
52              // No objects were available in the pool. Create a new one.
53              object = allocator.newInstance(this);
54          } else {
55              // An object was retrieved from the pool; decrement the pool size.
56              size.decrementAndGet();
57          }
58          return object;
59      }
60  
61      /**
62       * Adds the provided instance to the pool. If the pool is full, the instance will
63       * be discarded.
64       *
65       * Callers MUST NOT use an object after returning it to the pool.
66       *
67       * @param object   An object to add to the pool.
68       */
69      public void returnToPool(T object) {
70          // The `offer` method does not block. If the queue is full, it returns `false` immediately.
71          // If the provided instance cannot be added to the pool, we discard it silently.
72          if (size.getAndIncrement() < MAX_QUEUE_SIZE) {
73              objectQueue.offer(object);
74          } else {
75              // The pool was full. Since the size was optimistically incremented, decrement it now.
76              // Note: there is a race condition here that is deliberately allowed as an optimization.
77              // Under high contention, multiple threads could end up here before the first one
78              // decrements the size, causing objects to be dropped wastefully. This is not harmful
79              // because objects will be re-allocated when necessary; the pool is kept as close as
80              // possible to capacity on a best-effort basis. This race condition should not be "fixed"
81              // without a thorough study of the performance implications.
82              size.decrementAndGet();
83          }
84      }
85  }
