1   package com.amazon.ion.impl.bin.utf8;

3   import com.amazon.ion.IonException;
4   
5   import java.nio.ByteBuffer;
6   import java.nio.CharBuffer;
7   import java.nio.charset.Charset;
8   import java.nio.charset.CharsetDecoder;
9   import java.nio.charset.CoderResult;
10  
11  /**
12   * Decodes {@link String}s from UTF-8. Instances of this class are reusable but are NOT threadsafe.
13   *
14   * Instances are vended by {@link Utf8StringDecoderPool#getOrCreate()}.
15   *
16   * Users are expected to call {@link #close()} when the decoder is no longer needed.
17   *
18   * There are two ways of using this class:
19   * <ol>
20   *     <li>Use {@link #decode(ByteBuffer, int)} to decode the requested number of bytes from the given ByteBuffer in
21   *     a single step. Or,</li>
22   *     <li>Use the following sequence of method calls:
23   *     <ol>
24   *         <li>{@link #prepareDecode(int)} to prepare the decoder to decode the requested number of bytes.</li>
25   *         <li>{@link #partialDecode(ByteBuffer, boolean)} to decode the available bytes from the byte buffer. This may
26   *         be repeated as more bytes are made available in the ByteBuffer, which is the caller's responsibility.</li>
27   *         <li>{@link #finishDecode()} to finish decoding and return the resulting String.</li>
28   *     </ol>
29   *     Note: {@link #decode(ByteBuffer, int)} must not be called between calls to {@link #prepareDecode(int)} and
30   *     {@link #finishDecode()}.
31   *     </li>
32   * </ol>
33   */
34  public class Utf8StringDecoder extends Poolable<Utf8StringDecoder> {
35  
36      // The size of the UTF-8 decoding buffer.
37      private static final int UTF8_BUFFER_SIZE_IN_BYTES = 4 * 1024;
38  
39      private final CharBuffer reusableUtf8DecodingBuffer;
40      private final CharsetDecoder utf8CharsetDecoder;
41      private CharBuffer utf8DecodingBuffer;
42  
43      Utf8StringDecoder(Pool<Utf8StringDecoder> pool) {
44          super(pool);
45          reusableUtf8DecodingBuffer = CharBuffer.allocate(UTF8_BUFFER_SIZE_IN_BYTES);
46          utf8CharsetDecoder = Charset.forName("UTF-8").newDecoder();
47      }
48  
49      /**
50       * Prepares the decoder to decode the given number of UTF-8 bytes.
51       * @param numberOfBytes the number of bytes to decode.
52       */
53      public void prepareDecode(int numberOfBytes) {
54          utf8CharsetDecoder.reset();
55          utf8DecodingBuffer = reusableUtf8DecodingBuffer;
56          if (numberOfBytes > reusableUtf8DecodingBuffer.capacity()) {
57              utf8DecodingBuffer = CharBuffer.allocate(numberOfBytes);
58          }
59      }
60  
61      /**
62       * Decodes the available bytes from the given ByteBuffer.
63       * @param utf8InputBuffer a ByteBuffer containing UTF-8 bytes.
64       * @param endOfInput true if the end of the UTF-8 sequence is expected to occur in the buffer; otherwise, false.
65       */
66      public void partialDecode(ByteBuffer utf8InputBuffer, boolean endOfInput) {
67          CoderResult coderResult = utf8CharsetDecoder.decode(utf8InputBuffer, utf8DecodingBuffer, endOfInput);
68          if (coderResult.isError()) {
69              throw new IonException("Illegal value encountered while validating UTF-8 data in input stream. " + coderResult.toString());
70          }
71      }
72  
73      /**
74       * Finishes decoding and returns the resulting String.
75       * @return the decoded Java String.
76       */
77      public String finishDecode() {
78          utf8DecodingBuffer.flip();
79          return utf8DecodingBuffer.toString();
80      }
81  
82      /**
83       * Decodes the given number of UTF-8 bytes from the given ByteBuffer into a Java String.
84       * @param utf8InputBuffer a ByteBuffer containing UTF-8 bytes.
85       * @param numberOfBytes the number of bytes from the utf8InputBuffer to decode.
86       * @return the decoded Java String.
87       */
88      public String decode(ByteBuffer utf8InputBuffer, int numberOfBytes) {
89          prepareDecode(numberOfBytes);
90  
91          utf8DecodingBuffer.position(0);
92          utf8DecodingBuffer.limit(utf8DecodingBuffer.capacity());
93  
94          partialDecode(utf8InputBuffer, true);
95          return finishDecode();
96      }
97  }
