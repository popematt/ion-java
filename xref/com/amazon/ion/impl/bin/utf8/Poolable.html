3   import java.io.Closeable;
4   
5   /**
6    * Base class for types that may be pooled.
7    * @param <T> the concrete type.
8    */
9   abstract class Poolable<T extends Poolable<T>> implements Closeable {
10  
11      // The pool to which this object is linked.
12      private final Pool<T> pool;
13  
14      /**
15       * @param pool the pool to which the object will be returned upon {@link #close()}.
16       */
17      Poolable(Pool<T> pool) {
18          this.pool = pool;
19      }
20  
21      /**
22       * Attempts to return this instance to the pool with which it is associated, if any.
23       *
24       * Do not continue to use this instance after calling this method.
25       */
26      @Override
27      public void close() {
28          pool.returnToPool((T) this);
29      }
30  }
