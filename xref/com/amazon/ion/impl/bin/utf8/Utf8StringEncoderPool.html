1   package com.amazon.ion.impl.bin.utf8;
2   
3   import java.util.concurrent.ArrayBlockingQueue;
4   
5   /**
6    * A thread-safe shared pool of {@link Utf8StringEncoder}s that can be used for UTF8 encoding and decoding.
7    */
8   public enum Utf8StringEncoderPool {
9       // The only enum variant; a singleton instance.
10      INSTANCE;
11  
12      // The maximum number of Utf8Encoders that can be waiting in the queue before new ones will be discarded.
13      private static final int MAX_QUEUE_SIZE = 128;
14  
15      // A queue of previously initialized encoders that can be loaned out.
16      private final ArrayBlockingQueue<Utf8StringEncoder> bufferQueue;
17  
18      // Do not allow instantiation; all classes should share the singleton instance.
19      private Utf8StringEncoderPool() {
20          bufferQueue = new ArrayBlockingQueue<Utf8StringEncoder>(MAX_QUEUE_SIZE);
21      }
22  
23      /**
24       * @return a threadsafe shared instance of {@link Utf8StringEncoderPool}.
25       */
26      public static Utf8StringEncoderPool getInstance() {
27          return INSTANCE;
28      }
29  
30      /**
31       * If the pool is not empty, removes an instance of {@link Utf8StringEncoder} from the pool and returns it;
32       * otherwise, constructs a new instance.
33       *
34       * @return An instance of {@link Utf8StringEncoder}.
35       */
36      public Utf8StringEncoder getOrCreateUtf8Encoder() {
37          // The `poll` method does not block. If the queue is empty it returns `null` immediately.
38          Utf8StringEncoder encoder = bufferQueue.poll();
39          if (encoder == null) {
40              // No buffers were available in the pool. Create a new one.
41              encoder = new Utf8StringEncoder(this);
42          }
43          return encoder;
44      }
45  
46      /**
47       * Adds the provided instance of {@link Utf8StringEncoder} to the pool. If the pool is full, the instance will
48       * be discarded.
49       *
50       * Callers MUST NOT use an encoder after returning it to the pool.
51       *
52       * @param encoder   A {@link Utf8StringEncoder} to add to the pool.
53       */
54      public void returnEncoderToPool(Utf8StringEncoder encoder) {
55          // The `offer` method does not block. If the queue is full, it returns `false` immediately.
56          // If the provided instance cannot be added to the pool, we discard it silently.
57          bufferQueue.offer(encoder);
58      }
59  
60  }
