16  package com.amazon.ion.impl.bin;
17  
18  import com.amazon.ion.IonWriter;
19  
20  import java.io.ByteArrayOutputStream;
21  import java.io.IOException;
22  
23  /**
24   * NOT FOR APPLICATION USE!
25   *
26   * Exposes {@link IonRawBinaryWriter} functionality for use when creating Ion hashes.
27   */
28  @Deprecated
29  public class _PrivateIon_HashTrampoline
30  {
31      private static final PooledBlockAllocatorProvider ALLOCATOR_PROVIDER = PooledBlockAllocatorProvider.getInstance();
32      
33      public static IonWriter newIonWriter(ByteArrayOutputStream baos) throws IOException
34      {
35          return new IonRawBinaryWriter(
36                  ALLOCATOR_PROVIDER,
37                  _Private_IonManagedBinaryWriterBuilder.DEFAULT_BLOCK_SIZE,
38                  baos,
39                  AbstractIonWriter.WriteValueOptimization.NONE,
40                  IonRawBinaryWriter.StreamCloseMode.CLOSE,
41                  IonRawBinaryWriter.StreamFlushMode.FLUSH,
42                  IonRawBinaryWriter.PreallocationMode.PREALLOCATE_0,
43                  false     // force floats to be encoded as binary64
44          );
45      }
46  }
