79      protected IonWriterSystemText(SymbolTable defaultSystemSymtab,
80                                    _Private_IonTextWriterBuilder options,
81                                    _Private_FastAppendable out)
82      {
83          super(defaultSystemSymtab,
84                options.getInitialIvmHandling(),
85                options.getIvmMinimizing());
86  
87          _output =
88              _Private_IonTextAppender.forFastAppendable(out,
89                                                         options.getCharset());
90          _options = options;
91  
92          _separator_character = _options.topLevelSeparator();

96          _long_string_threshold = threshold;

113     protected IonType getContainer()
114     {
115         IonType container;
116 
117         if (_top < 1) {
118             container = IonType.DATAGRAM;
119         }
120         else {
121             switch(_stack_parent_type[_top-1]) {
122             case _Private_IonConstants.tidDATAGRAM:
123                 container = IonType.DATAGRAM;
124                 break;
125             case _Private_IonConstants.tidSexp:
126                 container = IonType.SEXP;
127                 break;
128             case _Private_IonConstants.tidList:
129                 container = IonType.LIST;
130                 break;
131             case _Private_IonConstants.tidStruct:
132                 container = IonType.STRUCT;
133                 break;
134             default:
135                 throw new IonException("unexpected container in parent stack: "+_stack_parent_type[_top-1]);
136             }
137         }
138         return container;
139     }
140     void push(int typeid)
141     {
142         if (_top + 1 == _stack_parent_type.length) {
143             growStack();
144         }
145         _stack_parent_type[_top] = typeid;
146         _stack_pending_comma[_top] = _pending_separator;
147         switch (typeid) {
148         case _Private_IonConstants.tidSexp:
149             _separator_character = " ";
150             break;
151         case _Private_IonConstants.tidList:
152         case _Private_IonConstants.tidStruct:
153             _separator_character = ",";
154             break;
155         default:
156             _separator_character = _options.lineSeparator();
157         break;
158         }
159         _top++;
160     }
161     void growStack() {
162         int oldlen = _stack_parent_type.length;
163         int newlen = oldlen * 2;
164         int[] temp1 = new int[newlen];
165         boolean[] temp3 = new boolean[newlen];
166 
167         System.arraycopy(_stack_parent_type, 0, temp1, 0, oldlen);
168         System.arraycopy(_stack_pending_comma, 0, temp3, 0, oldlen);
169 
170         _stack_parent_type = temp1;
171         _stack_pending_comma = temp3;
172     }
173     int pop() {
174         _top--;
175         int typeid = _stack_parent_type[_top];  // popped parent
176 
177         int parentid = (_top > 0) ? _stack_parent_type[_top - 1] : -1;
178         switch (parentid) {
179         case -1:
180             _in_struct = false;
181             _separator_character = _options.topLevelSeparator();
182             break;
183         case _Private_IonConstants.tidSexp:
184             _in_struct = false;
185             _separator_character = " ";
186             break;
187         case _Private_IonConstants.tidList:
188             _in_struct = false;
189             _separator_character = ",";
190             break;
191         case _Private_IonConstants.tidStruct:
192             _in_struct = true;
193             _separator_character = ",";
194             break;
195         default:
196             _separator_character = _options.lineSeparator();
197             break;
198         }
199 
200         return typeid;
201     }

211     boolean topPendingComma() {
212         if (_top == 0) return false;

216     private boolean containerIsSexp()
217     {
218         if (_top == 0) return false;
219         int topType = topType();
220         return (topType == tidSexp);
221     }

223     void printLeadingWhiteSpace() throws IOException {
224         for (int ii=0; ii<_top; ii++) {
225             _output.appendAscii(' ');
226             _output.appendAscii(' ');
227         }
228     }
229     void closeCollection(char closeChar) throws IOException {
230        if (_options.isPrettyPrintOn()) {
231            _output.appendAscii(_options.lineSeparator());
232            printLeadingWhiteSpace();
233        }
234        _output.appendAscii(closeChar);
235     }

238     private void writeSidLiteral(int sid)
239         throws IOException
240     {
241         assert sid >= 0;
242 
243         // No extra handling needed for JSON strings, this is already legal.
244 
245         boolean asString = _options._symbol_as_string;
246         if (asString) _output.appendAscii('"');
247 
248         _output.appendAscii('$');
249         _output.printInt(sid);
250 
251         if (asString) _output.appendAscii('"');
252     }

255     /**
256      * @param value must not be null.
257      */
258     private void writeSymbolToken(String value) throws IOException
259     {
260         if (_options._symbol_as_string)
261         {
262             if (_options._string_as_json)
263             {
264                 _output.printJsonString(value);
265             }
266             else
267             {
268                 _output.printString(value);
269             }
270         }
271         else
272         {
273             SymbolVariant variant = IonTextUtils.symbolVariant(value);
274             switch (variant)
275             {
276                 case IDENTIFIER:
277                 {
278                     _output.appendAscii(value);
279                     break;
280                 }
281                 case OPERATOR:
282                 {
283                     if (containerIsSexp())
284                     {
285                         _output.appendAscii(value);
286                         break;
287                     }
288                     // else fall through...
289                 }
290                 case QUOTED:
291                 {
292                     _output.printQuotedSymbol(value);
293                     break;
294                 }
295             }
296         }
297     }

321     void writeAnnotationToken(SymbolToken ann)
322         throws IOException
323     {
324         String name = ann.getText();
325         if (name == null) {
326             _output.appendAscii('$');
327             _output.appendAscii(Integer.toString(ann.getSid()));
328         }
329         else {
330             _output.printSymbol(name);
331         }

334     boolean writeSeparator(boolean followingLongString)
335         throws IOException
336     {
337         if (_options.isPrettyPrintOn()) {
338             if (_pending_separator && !IonTextUtils.isAllWhitespace(_separator_character)) {
339                 // Only bother if the separator is non-whitespace.
340                 _output.appendAscii(_separator_character);
341                 followingLongString = false;
342             }
343             _output.appendAscii(_options.lineSeparator());
344             printLeadingWhiteSpace();
345         }
346         else if (_pending_separator) {
347             _output.appendAscii(_separator_character);
348             if (!IonTextUtils.isAllWhitespace(_separator_character)) followingLongString = false;
349         }
350         return followingLongString;
351     }

353     @Override
354     void startValue() throws IOException
355     {
356         super.startValue();
357 
358         boolean followingLongString = _following_long_string;
359 
360         followingLongString = writeSeparator(followingLongString);
361 
362         // write field name
363         if (_in_struct) {
364             SymbolToken sym = assumeFieldNameSymbol();
365             writeFieldNameToken(sym);
366             _output.appendAscii(':');
367             clearFieldName();
368             followingLongString = false;
369         }
370 
371         // write annotations only if they exist and we're not currently
372         // writing an IVM
373         if (hasAnnotations() && !_is_writing_ivm) {
374             if (! _options._skip_annotations) {
375                 SymbolToken[] annotations = getTypeAnnotationSymbols();
376                 writeAnnotations(annotations);
377                 followingLongString = false;
378             }
379             clearAnnotations();
380         }
381 
382         _following_long_string = followingLongString;
383     }
384 
385     void closeValue()
386         throws IOException
387     {
388         super.endValue();
389         _pending_separator = true;
390         _following_long_string = false;  // Caller overwrites this as needed.
391 
392         // Flush if a top-level-value was written
393         if (getDepth() == 0)
394         {
395             try
396             {
397                 flush();
398             }
399             catch (IOException e)
400             {
401                 throw new IonException(e);
402             }
403         }
404     }

415     @Override
416     void writeLocalSymtab(SymbolTable symtab)
417         throws IOException
418     {
419         SymbolTable[] imports = symtab.getImportedTables();
420 
421         LstMinimizing min = _options.getLstMinimizing();
422         if (min == null)
423         {
424             symtab.writeTo(this);
425         }
426         else if (min == LstMinimizing.LOCALS && imports.length > 0)
427         {
428             // Copy the symtab, but filter out local symbols.
429 
430             IonReader reader = new SymbolTableReader(symtab);
431 
432             // move onto and write the struct header
433             IonType t = reader.next();
434             assert(IonType.STRUCT.equals(t));
435             SymbolToken[] a = reader.getTypeAnnotationSymbols();
436             // you (should) always have the $ion_symbol_table annotation
437             assert(a != null && a.length >= 1);
438 
439             // now we'll start a local symbol table struct
440             // in the underlying system writer
441             setTypeAnnotationSymbols(a);
442             stepIn(IonType.STRUCT);
443 
444             // step into the symbol table struct and
445             // write the values - EXCEPT the symbols field
446             reader.stepIn();
447             for (;;) {
448                 t = reader.next();
449                 if (t == null) break;
450                 // get the field name and skip over 'symbols'
451                 String name = reader.getFieldName();
452                 if (SYMBOLS.equals(name)) {
453                     continue;
454                 }
455                 writeValue(reader);
456             }
457 
458             // we're done step out and move along
459             stepOut();
460         }
461         else  // Collapse to IVM
462         {
463             SymbolTable systemSymtab = symtab.getSystemSymbolTable();
464             writeIonVersionMarker(systemSymtab);
465         }
466 
467         super.writeLocalSymtab(symtab);
468     }

470     public void stepIn(IonType containerType) throws IOException
471     {
472         startValue();
473 
474         int tid;
475         char opener;
476         switch (containerType)
477         {
478             case SEXP:
479                 if (!_options._sexp_as_list) {
480                     tid = tidSexp;
481                     _in_struct = false;
482                     opener = '('; break;
483                 }
484                 // else fall through and act just like list
485             case LIST:
486                 tid = tidList;
487                 _in_struct = false;
488                 opener = '[';
489                 break;
490             case STRUCT:
491                 tid = tidStruct;
492                 _in_struct = true;
493                 opener = '{';
494                 break;
495             default:
496                 throw new IllegalArgumentException();
497         }
498 
499         push(tid);
500         _output.appendAscii(opener);
501         _pending_separator = false;
502         _following_long_string = false;
503     }
504 
505     public void stepOut() throws IOException
506     {
507         if (_top < 1) {
508             throw new IllegalStateException(IonMessages.CANNOT_STEP_OUT);
509         }
510         _pending_separator = topPendingComma();
511         int tid = pop();
512 
513         char closer;
514         switch (tid)
515         {
516             case tidList:   closer = ']'; break;
517             case tidSexp:   closer = ')'; break;
518             case tidStruct: closer = '}'; break;
519             default:
520                 throw new IllegalStateException();
521         }
522         closeCollection(closer);
523         closeValue();
524     }

539     public void writeNull(IonType type) throws IOException
540     {
541         startValue();
542 
543         String nullimage;
544 
545         if (_options._untyped_nulls)
546         {
547             nullimage = "null";
548         }
549         else
550         {
551             switch (type) {
552                 case NULL:      nullimage = "null";           break;
553                 case BOOL:      nullimage = "null.bool";      break;
554                 case INT:       nullimage = "null.int";       break;
555                 case FLOAT:     nullimage = "null.float";     break;
556                 case DECIMAL:   nullimage = "null.decimal";   break;
557                 case TIMESTAMP: nullimage = "null.timestamp"; break;
558                 case SYMBOL:    nullimage = "null.symbol";    break;
559                 case STRING:    nullimage = "null.string";    break;
560                 case BLOB:      nullimage = "null.blob";      break;
561                 case CLOB:      nullimage = "null.clob";      break;
562                 case SEXP:      nullimage = "null.sexp";      break;
563                 case LIST:      nullimage = "null.list";      break;
564                 case STRUCT:    nullimage = "null.struct";    break;
565 
566                 default: throw new IllegalStateException("unexpected type " + type);
567             }
568         }
569 
570         _output.appendAscii(nullimage);
571         closeValue();
572     }

626     public void writeTimestamp(Timestamp value) throws IOException
627     {
628         if (value == null) {
629             writeNull(IonType.TIMESTAMP);
630             return;
631         }
632 
633         startValue();
634 
635         if (_options._timestamp_as_millis)
636         {
637             long millis = value.getMillis();
638             _output.appendAscii(Long.toString(millis));
639         }
640         else if (_options._timestamp_as_string)
641         {
642             // Timestamp is ASCII-safe so this is easy
643             _output.appendAscii('"');
644             _output.appendAscii(value.toString());
645             _output.appendAscii('"');
646         }
647         else
648         {
649             _output.appendAscii(value.toString());
650         }
651 
652         closeValue();
653     }

655     public void writeString(String value)
656         throws IOException
657     {
658         startValue();
659         if (value != null
660             && ! _following_long_string
661             && _long_string_threshold < value.length())
662         {
663             // TODO amzn/ion-java/issues/57 This can lead to mixed newlines in the output.
664             // It assumes NL line separators, but _options could use CR+NL
665             _output.printLongString(value);
666 
667             // This sets _following_long_string = false so we must overwrite
668             closeValue();
669             _following_long_string = true;
670         }
671         else
672         {
673             if (_options._string_as_json)
674             {
675                 _output.printJsonString(value);
676             }
677             else
678             {
679                 _output.printString(value);
680             }
681             closeValue();
682         }
683     }

748     /**
749      * {@inheritDoc}
750      * <p>
751      * The {@link OutputStream} spec is mum regarding the behavior of flush on
752      * a closed stream, so we shouldn't assume that our stream can handle that.
753      */
754     public void flush() throws IOException
755     {
756         if (! _closed) {
757             _output.flush();
758         }
759     }
760 
761     public void close() throws IOException
762     {
763         if (! _closed) {
764             try
765             {
766                 if (getDepth() == 0) {
767                     finish();
768                 }
769             }
770             finally
771             {
772                 // Do this first so we are closed even if the call below throws.
773                 _closed = true;
774 
775                 _output.close();
776             }
777         }
