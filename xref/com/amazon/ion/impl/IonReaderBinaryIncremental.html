38  /**
39   * <p>
40   * This implementation differs from the existing non-incremental binary reader implementation in that if
41   * {@link IonReader#next()} returns {@code null} at the top-level, it indicates that there is not (yet) enough data in
42   * the stream to complete a top-level value. The user may wait for more data to become available in the stream and
43   * call {@link IonReader#next()} again to continue reading. Unlike the non-incremental reader, the incremental reader
44   * will never throw an exception due to unexpected EOF during {@code next()}. If, however, {@link IonReader#close()} is
45   * called when an incomplete value is buffered, an {@link IonException} will be raised.
46   * </p>
47   * <p>
48   * Although the incremental binary reader implementation provides performance superior to the non-incremental reader
49   * implementation for both incremental and non-incremental use cases, there is one caveat: the incremental
50   * implementation must be able to buffer an entire top-level value and any preceding system values (Ion version
51   * marker(s) and symbol table(s)) in memory. This means that each value and preceding system values must be no larger
52   * than any of the following:
53   * <ul>
54   * <li>The configured maximum buffer size of the {@link IonBufferConfiguration}.</li>
55   * <li>The memory available to the JVM.</li>
56   * <li>2GB, because the buffer is held in a Java {@code byte[]}, which is indexed by an {@code int}.</li>
57   * </ul>
58   * This will not be a problem for the vast majority of Ion streams, as it is
59   * rare for a single top-level value or symbol table to exceed a few megabytes in size. However, if the size of the
60   * stream's values risk exceeding the available memory, then this implementation must not be used.
61   * </p>
62   * <p>
63   * To enable this implementation, use {@code IonReaderBuilder.withIncrementalReadingEnabled(true)}.
64   * </p>
65   */
66  class IonReaderBinaryIncremental implements IonReader, _Private_ReaderWriter, _Private_IncrementalReader {

327     /**
328      * Constructor.
329      * @param builder the builder containing the configuration for the new reader.
330      * @param inputStream the InputStream that provides binary Ion data.
331      */
332     IonReaderBinaryIncremental(IonReaderBuilder builder, InputStream inputStream) {
333         this.inputStream = inputStream;
334         this.catalog = builder.getCatalog() == null ? EMPTY_CATALOG : builder.getCatalog();
335         if (builder.isAnnotationIteratorReuseEnabled()) {
336             isAnnotationIteratorReuseEnabled = true;
337             annotationIterator = new AnnotationIterator();
338         } else {
339             isAnnotationIteratorReuseEnabled = false;
340             annotationIterator = null;
341         }
342         IonBufferConfiguration configuration = builder.getBufferConfiguration();
343         if (configuration == null) {
344             configuration = STANDARD_BUFFER_CONFIGURATION;
345             if (inputStream instanceof ByteArrayInputStream) {
346                 // ByteArrayInputStreams are fixed-size streams. Clamp the reader's internal buffer size at the size of
347                 // the stream to avoid wastefully allocating extra space that will never be needed. It is still
348                 // preferable for the user to manually specify the buffer size if it's less than the default, as doing
349                 // so allows this branch to be skipped.
350                 int fixedBufferSize;
351                 try {
352                     fixedBufferSize = inputStream.available();
353                 } catch (IOException e) {
354                     // ByteArrayInputStream.available() does not throw.
355                     throw new IllegalStateException(e);
356                 }
357                 if (configuration.getInitialBufferSize() > fixedBufferSize) {
358                     configuration = FIXED_SIZE_CONFIGURATIONS[logBase2(fixedBufferSize)];
359                 }
360             }
361         }
362         lookahead = new IonReaderLookaheadBuffer(configuration, inputStream);
363         buffer = (ResizingPipedInputStream) lookahead.getPipe();
364         containerStack = new _Private_RecyclingStack<ContainerInfo>(
365             CONTAINER_STACK_INITIAL_CAPACITY,
366             CONTAINER_INFO_FACTORY
367         );
368         annotationSids = new IntList(ANNOTATIONS_LIST_INITIAL_CAPACITY);
369         symbols = new ArrayList<String>(SYMBOLS_LIST_INITIAL_CAPACITY);
370         scalarConverter = new _Private_ScalarConversions.ValueVariant();
371         resetImports();
372     }

630         @Override
631         public SymbolToken find(String text) {
632             SymbolToken token = importedTables.find(text);
633             if (token != null) {
634                 return token;
635             }
636             Integer sid = mapView.get(text);
637             if (sid == null) {
638                 return null;
639             }
640             // The following per-call allocation is intentional. When weighed against the alternative of making
641             // 'mapView' a 'Map<String, SymbolToken>` instead of a `Map<String, Integer>`, the following points should
642             // be considered:
643             // 1. A LocalSymbolTableSnapshot is only created when getSymbolTable() is called on the reader. The reader
644             // does not use the LocalSymbolTableSnapshot internally. There are two cases when getSymbolTable() would be
645             // called: a) when the user calls it, which will basically never happen, and b) when the user uses
646             // IonSystem.iterate over the reader, in which case each top-level value holds a reference to the symbol
647             // table that was in scope when it occurred. In case a), in addition to rarely being called at all, it
648             // would be even rarer for a user to use find() to retrieve each symbol (especially more than once) from the
649             // returned symbol table. Case b) may be called more frequently, but it remains equally rare that a user

792      * Creates a shared symbol table import, resolving it from the catalog if possible.
793      * @param name the name of the shared symbol table.
794      * @param version the version of the shared symbol table.
795      * @param maxId the max_id of the shared symbol table. This value takes precedence over the actual max_id for the
796      *              shared symbol table at the requested version.
797      */
798     private SymbolTable createImport(String name, int version, int maxId) {
799         SymbolTable shared = catalog.getTable(name, version);
800         if (shared == null) {
801             // No match. All symbol IDs that fall within this shared symbol table's range will have unknown text.
802             return new SubstituteSymbolTable(name, version, maxId);
803         } else if (shared.getMaxId() != maxId || shared.getVersion() != version) {
804             // Partial match. If the requested max_id exceeds the actual max_id of the resolved shared symbol table,
805             // symbol IDs that exceed the max_id of the resolved shared symbol table will have unknown text.
806             return new SubstituteSymbolTable(shared, version, maxId);
807         } else {
808             // Exact match; the resolved shared symbol table may be used as-is.
809             return shared;
810         }

848     /**
849      * Creates a SymbolToken representation of the given symbol ID.
850      * @param sid a symbol ID.
851      * @return a SymbolToken.
852      */
853     private SymbolToken getSymbolToken(int sid) {
854         int symbolTableSize = maxSymbolId() + 1;
855         if (symbolTokensById == null) {
856             symbolTokensById = new ArrayList<SymbolToken>(symbolTableSize);
857         }
858         if (symbolTokensById.size() < symbolTableSize) {
859             for (int i = symbolTokensById.size(); i < symbolTableSize; i++) {
860                 symbolTokensById.add(null);
861             }
862         }
863         if (sid >= symbolTableSize) {
864             throw new IonException("Symbol ID exceeds the max ID of the symbol table.");
865         }
866         SymbolToken token = symbolTokensById.get(sid);
867         if (token == null) {
868             String text = getSymbolString(sid, imports, symbols);
869             ImportLocation importLocation = null;
870             if (text == null) {
871                 // Note: this will never be a system symbol.
872                 if (sid > 0 && sid <= imports.getMaxId()) {
873                     importLocation = imports.getImportLocation(sid);
874                 } else {
875                     // All symbols with unknown text in the local symbol range are equivalent to symbol zero.
876                     sid = 0;
877                 }
878             }
879             token = new SymbolTokenImpl(text, sid, importLocation);
880             symbolTokensById.set(sid, token);
881         }
882         return token;
883     }

885     /**
886      * Reads a local symbol table from the buffer.
887      * @param marker marker for the start and end positions of the local symbol table in the buffer.
888      */
889     private void readSymbolTable(IonReaderLookaheadBuffer.Marker marker) {
890         peekIndex = marker.startIndex;
891         boolean isAppend = false;
892         boolean hasSeenImports = false;
893         boolean hasSeenSymbols = false;
894         int symbolsPosition = -1;
895         int symbolsEndPosition = -1;
896         List<SymbolTable> newImports;
897         while (peekIndex < marker.endIndex) {
898             fieldNameSid = readVarUInt();
899             IonTypeID typeID = readTypeId();
900             calculateEndPosition(typeID);
901             int currentValueEndPosition = valueEndPosition;
902             if (fieldNameSid == SystemSymbolIDs.IMPORTS_ID) {
903                 if (hasSeenImports) {
904                     throw new IonException("Symbol table contained multiple imports fields.");
905                 }
906                 if (typeID.type == IonType.SYMBOL) {
907                     isAppend = readUInt(peekIndex, currentValueEndPosition) == SystemSymbolIDs.ION_SYMBOL_TABLE_ID;
908                     peekIndex = currentValueEndPosition;
909                 } else if (typeID.type == IonType.LIST) {
910                     resetImports();
911                     newImports = new ArrayList<SymbolTable>(3);
912                     newImports.add(getSystemSymbolTable());
913                     stepIn();
914                     IonType type = next();
915                     while (type != null) {
916                         String name = null;
917                         int version = -1;
918                         int maxId = -1;
919                         if (type == IonType.STRUCT) {
920                             stepIn();
921                             type = next();
922                             while (type != null) {
923                                 int fieldSid = getFieldId();
924                                 if (fieldSid == SystemSymbolIDs.NAME_ID) {
925                                     if (type == IonType.STRING) {
926                                         name = stringValue();
927                                     }
928                                 } else if (fieldSid == SystemSymbolIDs.VERSION_ID) {
929                                     if (type == IonType.INT) {
930                                         version = intValue();
931                                     }
932                                 } else if (fieldSid == SystemSymbolIDs.MAX_ID_ID) {
933                                     if (type == IonType.INT) {
934                                         maxId = intValue();
935                                     }
936                                 }
937                                 type = next();
938                             }
939                             stepOut();
940                         }
941                         newImports.add(createImport(name, version, maxId));
942                         type = next();
943                     }
944                     stepOut();
945                     imports = new LocalSymbolTableImports(newImports);
946                 }
947                 if (!isAppend) {
948                     // Clear the existing symbols before adding the new imported symbols.
949                     resetSymbolTable();
950                 }
951                 hasSeenImports = true;
952             } else if (fieldNameSid == SystemSymbolIDs.SYMBOLS_ID) {
953                 if (hasSeenSymbols) {
954                     throw new IonException("Symbol table contained multiple symbols fields.");
955                 }
956                 if (typeID.type == IonType.LIST) {
957                     // Just record this position and skip forward. Come back after the imports (if any) are parsed.
958                     symbolsPosition = peekIndex;
959                     symbolsEndPosition = currentValueEndPosition;
960                 }
961                 hasSeenSymbols = true;
962             }
963             peekIndex = currentValueEndPosition;
964         }
965         if (peekIndex > marker.endIndex) {
966             throw new IonException("Malformed symbol table. Child values exceeded the length declared in the header.");
967         }
968         if (!hasSeenImports) {
969             resetSymbolTable();
970             resetImports();
971         }
972         if (symbolsPosition > -1) {
973             peekIndex = symbolsPosition;
974             valueType = IonType.LIST;
975             valueEndPosition = symbolsEndPosition;
976             stepIn();
977             while (next() != null) {
978                 if (valueType != IonType.STRING) {
979                     symbols.add(null);
980                 } else {
981                     symbols.add(stringValue());
982                 }
983             }
984             stepOut();
985             peekIndex = valueEndPosition;
986         }
987     }

989     /**
990      * Advance the reader to the next top-level value. Buffers an entire top-level value, reads any IVMs and/or local
991      * symbol tables that precede the value, and sets the byte positions of important components of the value.
992      */
993     private void nextAtTopLevel() {
994         if (completeValueBuffered) {
995             // There is already data buffered, but the user is choosing to skip it.
996             buffer.seekTo(valueEndPosition);
997             completeValueBuffered = false;
998         }
999         try {
1000             lookahead.fillInput();
1001         } catch (Exception e) {
1002             throw new IonException(e);
1003         }
1004         if (lookahead.moreDataRequired()) {
1005             valueType = null;
1006             valueTypeID = null;
1007             return;
1008         }
1009         completeValueBuffered = true;
1010         if (lookahead.getIvmIndex() > -1) {
1011             peekIndex = lookahead.getIvmIndex();
1012             majorVersion = buffer.peek(peekIndex++);
1013             minorVersion = buffer.peek(peekIndex++);
1014             if (buffer.peek(peekIndex++) != IVM_FINAL_BYTE) {
1015                 throw new IonException("Invalid Ion version marker.");
1016             }
1017             requireSupportedIonVersion();
1018             resetSymbolTable();
1019             resetImports();
1020             lookahead.resetIvmIndex();
1021         } else if (peekIndex < 0) {
1022             // peekIndex is initialized to -1 and only increases. This branch is reached if the IVM does not occur
1023             // first in the stream. This is necessary because currently a binary incremental reader will be created if
1024             // an empty stream is provided to the IonReaderBuilder. If, once bytes appear in the stream, those bytes do
1025             // not represent valid binary Ion, a quick failure is necessary.
1026             throw new IonException("Binary Ion must start with an Ion version marker.");
1027         }
1028         List<IonReaderLookaheadBuffer.Marker> symbolTableMarkers = lookahead.getSymbolTableMarkers();
1029         if (!symbolTableMarkers.isEmpty()) {
1030             // The cached SymbolTable (if any) is a snapshot in time, so it must be cleared whenever a new symbol
1031             // table is read regardless of whether the new LST is an append or a reset.
1032             cachedReadOnlySymbolTable = null;
1033             for (IonReaderLookaheadBuffer.Marker symbolTableMarker : symbolTableMarkers) {
1034                 readSymbolTable(symbolTableMarker);
1035             }
1036             lookahead.resetSymbolTableMarkers();
1037         }
1038         peekIndex = lookahead.getValueStart();
1039         hasAnnotations = lookahead.hasAnnotations();
1040         if (hasAnnotations) {
1041             if (peekIndex >= lookahead.getValueEnd()) {
1042                 throw new IonException("Annotation wrappers without values are invalid.");
1043             }
1044             annotationSids.clear();
1045             IonReaderLookaheadBuffer.Marker annotationSidsMarker = lookahead.getAnnotationSidsMarker();
1046             annotationStartPosition = annotationSidsMarker.startIndex;
1047             annotationEndPosition = annotationSidsMarker.endIndex;
1048             peekIndex = annotationEndPosition;
1049             valueTypeID = IonTypeID.TYPE_IDS[buffer.peek(peekIndex++)];
1050             int wrappedValueLength = valueTypeID.length;
1051             if (valueTypeID.variableLength) {
1052                 wrappedValueLength = readVarUInt();
1053             }
1054             valueType = valueTypeID.type;
1055             if (valueType == IonTypeID.ION_TYPE_ANNOTATION_WRAPPER) {
1056                 throw new IonException("Nested annotations are invalid.");
1057             }
1058             if (peekIndex + wrappedValueLength != lookahead.getValueEnd()) {
1059                 throw new IonException("Mismatched annotation wrapper length.");
1060             }
1061         } else {
1062             valueTypeID = lookahead.getValueTid();
1063             valueType = valueTypeID.type;
1064         }
1065         valueStartPosition = peekIndex;
1066         valueEndPosition = lookahead.getValueEnd();
1067         lookahead.resetNopPadIndex();
1068     }

1111     /**
1112      * Advance the reader to the next value within a container, which must already be buffered.
1113      */
1114     private void nextBelowTopLevel() {
1115         // Seek past the previous value.
1116         if (peekIndex < valueEndPosition) {
1117             peekIndex = valueEndPosition;
1118         }
1119         if (peekIndex >= containerStack.peek().endPosition) {
1120             endContainer();
1121         } else {
1122             if (containerStack.peek().type == IonType.STRUCT) {
1123                 fieldNameSid = readVarUInt();
1124             }
1125             IonTypeID typeID = readTypeId();
1126             while (typeID.isNopPad) {
1127                 calculateEndPosition(typeID);
1128                 peekIndex = valueEndPosition;
1129                 if (peekIndex >= containerStack.peek().endPosition) {
1130                     endContainer();
1131                     return;
1132                 }
1133                 if (containerStack.peek().type == IonType.STRUCT) {
1134                     fieldNameSid = readVarUInt();
1135                 }
1136                 typeID = readTypeId();
1137             }
1138             calculateEndPosition(typeID);
1139             if (valueType == IonTypeID.ION_TYPE_ANNOTATION_WRAPPER) {
1140                 hasAnnotations = true;
1141                 annotationSids.clear();
1142                 int annotationsLength = readVarUInt();
1143                 annotationStartPosition = peekIndex;
1144                 annotationEndPosition = annotationStartPosition + annotationsLength;
1145                 peekIndex = annotationEndPosition;
1146                 typeID = readTypeId();
1147                 if (typeID.isNopPad) {
1148                     throw new IonException(
1149                         "Invalid annotation wrapper: NOP pad may not occur inside an annotation wrapper."
1150                     );
1151                 }
1152                 if (valueType == IonTypeID.ION_TYPE_ANNOTATION_WRAPPER) {
1153                     throw new IonException("Nested annotations are invalid.");
1154                 }
1155                 long annotationWrapperEndPosition = valueEndPosition;
1156                 calculateEndPosition(typeID);
1157                 if (annotationWrapperEndPosition != valueEndPosition) {
1158                     throw new IonException(
1159                         "Invalid annotation wrapper: end of the wrapper did not match end of the value."
1160                     );
1161                 }
1162             } else {
1163                 annotationStartPosition = -1;
1164                 annotationEndPosition = -1;
1165                 hasAnnotations = false;
1166                 if (valueEndPosition > containerStack.peek().endPosition) {
1167                     throw new IonException("Value overflowed its container.");
1168                 }
1169             }
1170             if (!valueTypeID.isValid) {
1171                 throw new IonException("Invalid type ID.");
1172             }
1173             valueStartPosition = peekIndex;
1174         }
1175     }

1265     @Override
1266     public IntegerSize getIntegerSize() {
1267         if (valueType != IonType.INT || isNullValue()) {
1268             return null;
1269         }
1270         if (valueTypeID.length < INT_SIZE_IN_BYTES) {
1271             // Note: this is conservative. Most integers of size 4 also fit in an int, but since exactly the
1272             // same parsing code is used for ints and longs, there is no point wasting the time to determine the
1273             // smallest possible type.
1274             return IntegerSize.INT;
1275         } else if (valueTypeID.length < LONG_SIZE_IN_BYTES) {
1276             return IntegerSize.LONG;
1277         } else if (valueTypeID.length == LONG_SIZE_IN_BYTES) {
1278             // Because creating BigIntegers is so expensive, it is worth it to look ahead and determine exactly
1279             // which 8-byte integers can fit in a long.
1280             if (valueTypeID.isNegativeInt) {
1281                 // The smallest negative 8-byte integer that can fit in a long is -0x80_00_00_00_00_00_00_00.
1282                 int firstByte = buffer.peek(valueStartPosition);
1283                 if (firstByte < MOST_SIGNIFICANT_BYTE_OF_MIN_LONG) {
1284                     return IntegerSize.LONG;
1285                 } else if (firstByte > MOST_SIGNIFICANT_BYTE_OF_MIN_LONG) {
1286                     return IntegerSize.BIG_INTEGER;
1287                 }
1288                 for (int i = valueStartPosition + 1; i < valueEndPosition; i++) {
1289                     if (0x00 != buffer.peek(i)) {
1290                         return IntegerSize.BIG_INTEGER;
1291                     }
1292                 }
1293             } else {
1294                 // The largest positive 8-byte integer that can fit in a long is 0x7F_FF_FF_FF_FF_FF_FF_FF.
1295                 if (buffer.peek(valueStartPosition) > MOST_SIGNIFICANT_BYTE_OF_MAX_LONG) {
1296                     return IntegerSize.BIG_INTEGER;
1297                 }
1298             }
1299             return IntegerSize.LONG;
1300         }
1301         return IntegerSize.BIG_INTEGER;
1302     }

1317     /**
1318      * Reads a VarUInt.
1319      * @return the value.
1320      */
1321     private int readVarUInt() {
1322         int currentByte = 0;
1323         int result = 0;

1652     /**
1653      * Reads a decimal value as a BigDecimal.
1654      * @return the value.
1655      */
1656     private BigDecimal readBigDecimal() {
1657         int length = valueEndPosition - peekIndex;
1658         if (length == 0) {
1659             return BigDecimal.ZERO;
1660         }
1661         int scale = -readVarInt();
1662         BigDecimal value;
1663         if (length < LONG_SIZE_IN_BYTES) {
1664             // No need to allocate a BigInteger to hold the coefficient.
1665             long coefficient = 0;
1666             int sign = 1;
1667             if (peekIndex < valueEndPosition) {
1668                 int firstByte = buffer.peek(peekIndex++);
1669                 sign = (firstByte & HIGHEST_BIT_BITMASK) == 0 ? 1 : -1;
1670                 coefficient = firstByte & LOWER_SEVEN_BITS_BITMASK;
1671             }
1672             while (peekIndex < valueEndPosition) {
1673                 coefficient = (coefficient << VALUE_BITS_PER_UINT_BYTE) | buffer.peek(peekIndex++);
1674             }
1675             value = BigDecimal.valueOf(coefficient * sign, scale);
1676         } else {
1677             // The coefficient may overflow a long, so a BigInteger is required.
1678             value = new BigDecimal(readIntAsBigInteger(valueEndPosition), scale);
1679         }
1680         return value;
1681     }

1683     /**
1684      * Reads a decimal value as a Decimal.
1685      * @return the value.
1686      */
1687     private Decimal readDecimal() {
1688         int length = valueEndPosition - peekIndex;
1689         if (length == 0) {
1690             return Decimal.ZERO;
1691         }
1692         int scale = -readVarInt();
1693         BigInteger coefficient;
1694         length = valueEndPosition - peekIndex;
1695         if (length > 0) {
1696             // NOTE: unfortunately, there is no BigInteger(int signum, byte[] bits, int offset, int length) constructor,
1697             // so copying to scratch space is always required.
1698             byte[] bits = copyBytesToScratch(peekIndex, length);
1699             int signum = getAndClearSignBit(bits);
1700             // NOTE: there is a BigInteger.valueOf(long unscaledValue, int scale) factory method that avoids allocating
1701             // a BigInteger for coefficients that fit in a long. See its use in readBigDecimal() above. Unfortunately,
1702             // it is not possible to use this for Decimal because the necessary BigDecimal constructor is
1703             // package-private. If a compatible BigDecimal constructor is added in a future JDK revision, a
1704             // corresponding factory method should be added to Decimal to enable this optimization.
1705             coefficient = new BigInteger(signum, bits);
1706             if (coefficient.signum() == 0 && signum < 0) {
1707                 return Decimal.negativeZero(scale);
1708             }
1709         }
1710         else {
1711             coefficient = BigInteger.ZERO;
1712         }
1713         return Decimal.valueOf(coefficient, scale);
1714     }

1739         if (isNullValue()) {
1740             return null;
1741         }
1742         peekIndex = valueStartPosition;
1743         int firstByte = buffer.peek(peekIndex++);
1744         Integer offset = null;
1745         if (firstByte != VAR_INT_NEGATIVE_ZERO) {
1746             offset = readVarInt(firstByte);
1747         }
1748         int year = readVarUInt();
1749         int month = 0;
1750         int day = 0;
1751         int hour = 0;
1752         int minute = 0;
1753         int second = 0;
1754         BigDecimal fractionalSecond = null;
1755         Timestamp.Precision precision = Timestamp.Precision.YEAR;
1756         if (peekIndex < valueEndPosition) {
