18  /*
19   * This is a class that supports encoding and decoding binary
20   * data in base 64 encodings.
21   *
22   * <p>
23   * The default encoding is the URL Safe encoding variant specified
24   * in http://tools.ietf.org/html/rfc4648
25   * </p>
26   *
27   * <p>
28   * It's character translation table is:
29   *
30   * <pre>
31   *    Table 2: The "URL and Filename safe" Base 64 Alphabet
32   *
33   * NO LONGER URL/FILENAME SAFE, BACK TO THE ORIGINAL URL CHARACTERS
34   *
35   *   Value Encoding  Value Encoding  Value Encoding  Value Encoding
36   *        0 A            17 R            34 i            51 z
37   *        1 B            18 S            35 j            52 0
38   *        2 C            19 T            36 k            53 1
39   *        3 D            20 U            37 l            54 2
40   *        4 E            21 V            38 m            55 3
41   *        5 F            22 W            39 n            56 4
42   *        6 G            23 X            40 o            57 5
43   *        7 H            24 Y            41 p            58 6
44   *        8 I            25 Z            42 q            59 7
45   *        9 J            26 a            43 r            60 8
46   *       10 K            27 b            44 s            61 9
47   *       11 L            28 c            45 t            62 +
48   *       12 M            29 d            46 u            63 /
49   *       13 N            30 e            47 v
50   *       14 O            31 f            48 w
51   *       15 P            32 g            49 x
52   *       16 Q            33 h            50 y            (pad) =
53   * </pre>
54   * </p>
55   */

103     static private char init64IntToCharTerminator(EL[] els)
104     {
105         for (EL letter : els) {
106             if (letter.value == -1) {
107                 return letter.letter;
108             }
109         }
110         throw new RuntimeException(new IonException("fatal: invalid char map definition - missing terminator"));
111     }
112     static private int[] init64IntToChar(EL[] els)
113     {
114         int[] output = new int[64];
115         for (EL letter : els) {
116             if (letter.value != -1) {
117                 output[letter.value] = letter.letter;
118             }
119         }
120         return output;
121     }
122     static private int[] init64CharToInt(EL[] els)
123     {
124         int[] output = new int[256];
125         for (int ii=0; ii<256; ii++) {
126             output[ii] = -1; // mark everything as invalid to start with
127         }
128         // mark the valid entries with a non -1 value is they're useful
129         for (EL letter : els) {
130             if (letter.letter > 255) {
131                 throw new RuntimeException("fatal base 64 encoding static initializer: letter out of bounds");
132             }
133             else if (letter.value >= 0) {
134                 output[letter.letter] = letter.value;
135             }
136         }
137         return output;
138     }

140     final static boolean isBase64Character(int c) {
141         if (c < 32 || c > 255) return false;
142         return (URLSafe64CharToInt[c] >= 0);
143     }

149      * BinaryStream, reads a text input and decodes the printable characters

192         private int characterToBinary(final int c) throws IOException {
193             int result = -1;
194             if (c >= 0 && c < _chartobin.length) {
195                 result = _chartobin[c];
196             }
197             if (result < 0) {
198                 throw new IonException("invalid base64 character (" + c + ")");
199             }
200             return result;
201         }

203         // Read a buffer from the input stream and prep the output stream
204         private void loadNextBuffer() throws IOException
205         {
206 
207             int inlen = 0;
208             int convert, c = -1, cbin;
209 
210             this._bufEnd = 0;
211             this._bufPos = 0;
212 
213             if (this._state == 1) {
214                 return;
215             }
216 
217             // try to read in 4 convertable text characters from the source stream
218             while (inlen < 4) {
219                 c = this._source.read();
220                 if (c == -1 || c == 65535 || c == this._terminator || c == this._otherTerminator) {
221                     _terminatingChar = c;
222                     break;
223                 }
224                 if (IonTextUtils.isWhitespace(c)) continue;
225                 cbin = characterToBinary(c);
226 
227                 this._buffer[inlen++] = (char)cbin;
228             }
229             if (inlen != 4) {
230                 if (inlen == 0 && c != this._terminator) {
231                     this._state = 1;
232                     return;
233                 }
234                 // read through the trailing '='s
235                 int templen = inlen;
236                 while (c == this._terminator) {
237                     templen++;
238                     c = this._source.read();
239                 }
240                 if (templen != 4) {
241                     throw new IonException("base64 character count must be divisible by 4, using '=' for padding");
242                 }
243                 else if (inlen < 1) {
244                     throw new IonException("base64 character count must be divisible by 4, but using no more than 3 '=' chars for padding");
245                 }
246 
247                 this._terminatingChar = c;
248             }
249 
250             // now convert those characters
251             int ii = 0;
252             for (;;) {
253 
254                 // next usable char:
255                 c = this._buffer[ii++];
256                 convert =  c << 18;
257 
258                 // next usable char:
259                 c = this._buffer[ii++];
260                 convert |= (c << 12);
261                 if (inlen < 3) {
262                     // with 2 input chars (6*2 = 12 bits) we get only 1 output byte (8 bits)
263                     this._buffer[this._bufEnd++] = (char)((convert & 0x00FF0000) >> 16);
264                     this._state = 1; // if we ran out, then we're done
265                     break;
266                 }
267 
268                 // next usable char:
269                 c = this._buffer[ii++];
270                 convert |= (c << 6);
271                 if (inlen < 4) {
272                     // with 3 input chars (6*3 = 18 bits) we get 2 output bytes (8*2 = 16 bits)
273                     this._buffer[this._bufEnd++] = (char)((convert & 0x00FF0000) >> 16);
274                     this._buffer[this._bufEnd++] = (char)((convert & 0x0000FF00) >> 8);
275                     this._state = 1; // if we ran out, then we're done
276                     break;
277                 }
278 
279                 //    next (and final possible out of 4) usable char:
280                 c = this._buffer[ii++];
281                 convert |= (c << 0);
282                 // and with 4 input chars (6*4 = 24 bits) we get the full 3 output byte2 (8*3 = 24 bits)
283                 this._buffer[this._bufEnd++] = (char)((convert & 0x00FF0000) >> 16);
284                 this._buffer[this._bufEnd++] = (char)((convert & 0x0000FF00) >> 8);
285                 this._buffer[this._bufEnd++] = (char)((convert & 0x000000FF) >> 0);
286                 break;
287             }
288             return;
289         }

303         //Read a single character.
304         @Override
305         public int read() throws IOException
306         {
307             int outchar = -1;
308             if (!_ready) {
309                 throw new IOException(this.getClass().getName()+ " is not ready");
310             }
311 
312             // read input until we get a translatable character
313             if (this._bufPos >= this._bufEnd) {
314                 this.loadNextBuffer();
315             }
316             if (this._bufPos < this._bufEnd) {
317                 outchar = this._buffer[this._bufPos++];
318             }
319             return outchar;
320         }

322         // Read characters into an array.
323         @Override
324         public int read(char[] cbuf) throws IOException
325         {
326             int ii = 0;
327             for (ii=0; ii<cbuf.length; ii++) {
328                 int c = read();
329                 if (c == -1) break;
330                 cbuf[ii] = (char)c;
331             }
332             return ii;
333         }

407         // Read a buffer from the binary input stream and prep the output stream
408         //
409         // note that the output is longer than the input as 3 binary
410         // bytes produce 4 ascii characters
411         //
412         @SuppressWarnings("cast")
413         private void loadNextBuffer() throws IOException
414         {
415             this._outBufEnd = 0;
416             this._outBufPos = 0;
417             int inlen = this._source.read(this._inbuf, 0, BUFSIZE_BIN);
418             if (inlen < 0) {
419                 this._state = 1;
420                 return;
421             }
422 
423             int ii = 0;
424             for (;;) {
425                 // next usable char:
426                 if (ii >= inlen) {
427                     break;
428                 }
429                 int c = (((int)this._inbuf[ii++]) & 0xFF);
430                 int convert =  c << 16;
431 
432                 // next usable char
433                 if (ii >= inlen) {
434                     this._outbuf[this._outBufEnd++] = (char)this._bintochar[((convert & 0xFC0000) >> 18)];
435                     this._outbuf[this._outBufEnd++] = (char)this._bintochar[((convert & 0x03F000) >> 12)];
436                     this._outbuf[this._outBufEnd++] = (char)this._padding;
437                     this._outbuf[this._outBufEnd++] = (char)this._padding;
438                     break;
439                 }
440                 c = (((int)this._inbuf[ii++]) & 0xFF);
441                 convert |=  c << 8;
442 
443                 // next usable char
444                 if (ii >= inlen) {
445                     this._outbuf[this._outBufEnd++] = (char)this._bintochar[((convert & 0xFC0000) >> 18)];
446                     this._outbuf[this._outBufEnd++] = (char)this._bintochar[((convert & 0x03F000) >> 12)];
447                     this._outbuf[this._outBufEnd++] = (char)this._bintochar[((convert & 0x000FC0) >>  6)];
448                     this._outbuf[this._outBufEnd++] = (char)this._padding;
449                     break;
450                 }
451                 c = (((int)this._inbuf[ii++]) & 0xFF);
452                 convert |=  c << 0;
453 
454                 this._outbuf[this._outBufEnd++] = (char)this._bintochar[((convert & 0xFC0000) >> 18)];
455                 this._outbuf[this._outBufEnd++] = (char)this._bintochar[((convert & 0x03F000) >> 12)];
456                 this._outbuf[this._outBufEnd++] = (char)this._bintochar[((convert & 0x000FC0) >>  6)];
457                 this._outbuf[this._outBufEnd++] = (char)this._bintochar[((convert & 0x00003F) >>  0)];
458             }
459             return;
460         }
461         //Read a single character.
462         @Override
463         public int read() throws IOException
464         {
465             int outchar = -1;
466             if (!_ready) {
467                 throw new IOException(this.getClass().getName()+ " is closed");
468             }
469             if (_state != 0) return  -1;
470 
471             // read input until we get a translatable character
472             if (this._outBufPos >= this._outBufEnd) {
473                 this.loadNextBuffer();
474             }
475             if (this._outBufPos < this._outBufEnd) {
476                 outchar = this._outbuf[this._outBufPos++];
477             }
478             return outchar;
479         }

481         // Read characters into an array.
482         @Override
483         public int read(char[] cbuf) throws IOException
484         {
485             if (!_ready) {
486                 throw new IOException(this.getClass().getName()+ " is closed");
487             }
488             if (_state != 0) return -1;
489 
490             int dstPos = 0;
491             int needed = cbuf.length;
492 
493             while (needed > 0) {
494                 // read input until we get a translatable character
495                 if (this._outBufPos >= this._outBufEnd) {
496                     this.loadNextBuffer();
497                 }
498                 if (this._outBufPos >= this._outBufEnd) {
499                     break;
500                 }
501                 int xfer = this._outBufEnd - this._outBufPos;
502                 if (xfer > needed) xfer = needed;
503                 System.arraycopy(this._outbuf, this._outBufPos, cbuf, dstPos, xfer);
504                 this._outBufPos += xfer;
505                 needed -= xfer;
506             }
507             return dstPos;
508         }

516             }
517             if (_state != 0) return -1;

519             int dstPos = off;
520             int needed = rlen;
521             while (needed > 0) {
522                 // read input until we get a translatable character

525                 }

554         //Skip characters.
555         @Override
556         public long skip(long n) throws IOException
557         {
558             if (!_ready) {
559                 throw new IOException(this.getClass().getName()+ " is closed");
560             }
561             if (n < 0) {
562                 throw new IllegalArgumentException("error skip only support non-negative a values for n");
563             }
564             long needed = n;
565             int  available;
566 
567             while (needed > 0) {
568                 available = this._outBufEnd - this._outBufPos;
569                 if (available < 1) {
570                     this.loadNextBuffer();
571                     available = this._outBufEnd - this._outBufPos;
572                     if (available < 1) break;
573                 }
574                 if (available > needed) {
575                     this._outBufPos += (int)needed;
576                     needed = 0;
577                     break;
578                 }
579                 needed -= available;
580                 this._outBufPos += available;
581             }
582             return n - needed;
583         }
