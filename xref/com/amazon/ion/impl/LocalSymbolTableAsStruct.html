106     private final SymbolTableStructCache structCache;
107 
108     /**
109      * @param imports           never null
110      * @param symbolsList       may be null or empty
111      */
112     private LocalSymbolTableAsStruct(LocalSymbolTableImports imports,
113                                      List<String> symbolsList,
114                                      IonStruct image)
115     {
116         super(imports, symbolsList);
117         structCache = new SymbolTableStructCache(this, imports.getImportedTablesNoCopy(), image);
118     }
119 
120     @Override
121     int putSymbol(String symbolName)
122     {
123         int sid = super.putSymbol(symbolName);
124         if (structCache.hasStruct())
125         {
126             structCache.addSymbol(symbolName, sid);
127         }
128         return sid;
129     }
