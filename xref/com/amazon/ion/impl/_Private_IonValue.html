16  package com.amazon.ion.impl;
17  
18  import com.amazon.ion.IonDatagram;
19  import com.amazon.ion.IonValue;
20  import com.amazon.ion.SymbolTable;
21  import com.amazon.ion.SymbolToken;
22  import java.io.PrintWriter;
23  
24  /**
25   * NOT FOR APPLICATION USE!
26   */
27  public interface _Private_IonValue
28      extends IonValue
29  {
30  
31      /**
32       * Provides an IonValue's SymbolTable.
33       */
34      public interface SymbolTableProvider {
35          public SymbolTable getSymbolTable();
36      }
37  
38      /**
39       *
40       * @return int the offset of this value in its containers member list
41       */
42      public int         getElementId();
43  
44      /**
45       * Overrides {@link IonValue#getFieldNameSymbol()} for use when there exists
46       * a SymbolTableProvider implementation for this IonValue.
47       * @param symbolTableProvider - provides this IonValue's symbol table
48       * @return the field name SymbolToken
49       * @see IonValue#getFieldNameSymbol()
50       */
51      public SymbolToken getFieldNameSymbol(SymbolTableProvider symbolTableProvider);
52  
53      /**
54       * Overrides {@link IonValue#getTypeAnnotationSymbols()} for use when there exists
55       * a SymbolTableProvider implementation for this IonValue.
56       * @param symbolTableProvider - provides this IonValue's symbol table
57       * @return the type annotation SymbolTokens
58       * @see IonValue#getTypeAnnotationSymbols()
59       */
60      public SymbolToken[] getTypeAnnotationSymbols(SymbolTableProvider symbolTableProvider);
61  
62      /**
63       * Returns the given annotation's index in the value's annotations list, or -1 if not present.
64       * @param annotation the annotation to find.
65       * @return the index or -1.
66       */
67      int findTypeAnnotation(String annotation);
68  
69      /**
70       * Makes this symbol table current for this value.
71       * This may directly apply to this IonValue if this
72       * value is either loose or a top level datagram
73       * member.  Or it may be delegated to the IonContainer
74       * this value is a contained in.
75       * <p>
76       * Assigning null forces any symbol values to be
77       * resolved to strings and any associated symbol
78       * table will be removed.
79       * <p>
80       * @param symbols must be local or system table. May be null.
81       *
82       * @throws UnsupportedOperationException if this is a datagram.
83       */
84      public void setSymbolTable(SymbolTable symbols);
85  
86      /**
87       * Returns the symbol table that is directly associated with this value,
88       * without doing any recursive lookup.
89       * Values that are not top-level will return null as they don't actually
90       * own their own symbol table.
91       *
92       * @throws UnsupportedOperationException if this is an {@link IonDatagram}.
93       */
94      public SymbolTable getAssignedSymbolTable();
95  
96      public void dump(PrintWriter out);
97  
98      public String validate();
99  }
