10  /**
11   * Manages a resizing buffer for production and consumption of data within a <strong>single thread</strong>.
12   * Buffered bytes may be consumed through the InputStream interface. This provides a few benefits over using
13   * a PipedOutputStream/PipedInputStream pair in a single thread:
14   * <ol>
15   *     <li>There is no risk of deadlock. Piped streams, which are intended for producing data in one
16   *     thread and consuming it in another, will block on read when no data is available and block on write
17   *     when the buffer is full. In a single-threaded context, avoiding deadlock on read requires checking
18   *     that bytes are available before every read. Avoiding deadlock on write would require checking that
19   *     the buffer is not full before every write, but there is no built-in, publicly-accessible way of doing
20   *     this with a PipedInputStream/PipedOutputStream.</li>
21   *     <li>The buffer can grow. Piped streams use a fixed-size buffer that causes blocking when full. If
22   *     used in a single-thread, this serves as a hard limit on the amount of data that can be written without
23   *     a matching read. This can require arbitrary limits on data size to be imposed by the application. The
24   *     ResizingPipedInputStream imposes no such limitation, but optionally allows for a maximum buffer
25   *     size to be configured to protect against unbounded growth.</li>
26   * </ol>
27   */
28  public class ResizingPipedInputStream extends InputStream {

30      /**
31       * Handler of notifications provided by the ResizingPipedInputStream.
32       */
33      interface NotificationConsumer {
34  
35          /**
36           * Bytes have been shifted to the start of the buffer in order to make room for additional bytes
37           * to be buffered.
38           * @param leftShiftAmount the amount of the left shift (also: the pre-shift read index of the first shifted
39           *                        byte).
40           */
41          void bytesConsolidatedToStartOfBuffer(int leftShiftAmount);
42      }

44      /**
45       * A NotificationConsumer that does nothing.
46       */
47      private static final NotificationConsumer NO_OP_NOTIFICATION_CONSUMER = new NotificationConsumer() {
48          @Override
49          public void bytesConsolidatedToStartOfBuffer(int leftShiftAmount) {
50              // Do nothing.

54      /**
55       * Mask to isolate a single byte.

57      private static final int SINGLE_BYTE_MASK = 0xFF;

59      /**
60       * The initial size of the buffer and the number of bytes by which the size of the buffer will increase
61       * each time it grows, unless it must grow by a smaller amount to fit within 'maximumBufferSize'.
62       */
63      private final int initialBufferSize;
64  
65      /**
66       * The maximum size of the buffer. If the user attempts to buffer more bytes than this, an exception will be raised.
67       */
68      private final int maximumBufferSize;

70      /**
71       * Whether to use a boundary to limit the number of available bytes. This can be used to buffer
72       * arbitrarily-sized chunks of bytes without making them available for consumption. When true,
73       * the boundary must be manually extended (see {@link #extendBoundary(int)} to make these bytes
74       * available. When false, all buffered bytes will be available to read.
75       */
76      private final boolean useBoundary;

78      /**
79       * The NotificationConsumer currently registered.
80       */
81      private NotificationConsumer notificationConsumer = NO_OP_NOTIFICATION_CONSUMER;

123     /**
124      * Constructor.
125      * @param initialBufferSize the initial size of the buffer. When full, the buffer will grow by this
126      *                          many bytes. The buffer always stores bytes contiguously, so growth requires
127      *                          allocation of a new buffer capable of holding the new capacity and copying of the
128      *                          existing bytes into the new buffer. As such, a size should be chosen carefully
129      *                          such that growth is expected to occur rarely, if ever.
130      */
131     public ResizingPipedInputStream(final int initialBufferSize) {
132         this(initialBufferSize, Integer.MAX_VALUE, false);
133     }
134 
135     /**
136      * Constructor.
137      * @param initialBufferSize the initial size of the buffer. When full, the buffer will grow by this
138      *                          many bytes. The buffer always stores bytes contiguously, so growth requires
139      *                          allocation of a new buffer capable of holding the new capacity and copying of the
140      *                          existing bytes into the new buffer. As such, a size should be chosen carefully
141      *                          such that growth is expected to occur rarely, if ever.
142      * @param maximumBufferSize the maximum size of the buffer. If a call to `receive` attempts to transfer an amount
143      *                          of bytes that would cause the buffer to exceed this size, a
144      *                          {@link BufferOverflowException} will be thrown. Must be greater than or equal to the
145      *                          initial buffer size.
146      * @param useBoundary whether to use a boundary to limit the number of available bytes. This can be used to buffer
147      *                    arbitrarily-sized chunks of bytes without making them available for consumption. When true,
148      *                    the boundary must be manually extended (see {@link #extendBoundary(int)} to make these bytes
149      *                    available. When false, all buffered bytes will be available to read.
150      */
151     ResizingPipedInputStream(final int initialBufferSize, final int maximumBufferSize, final boolean useBoundary) {
152         if (initialBufferSize < 1) {
153             throw new IllegalArgumentException("Initial buffer size must be at least 1.");
154         }
155         if (maximumBufferSize < initialBufferSize) {
156             throw new IllegalArgumentException("Maximum buffer size cannot be less than the initial buffer size.");
157         }
158         this.initialBufferSize = initialBufferSize;
159         this.maximumBufferSize = maximumBufferSize;
160         this.capacity = initialBufferSize;
161         buffer = new byte[initialBufferSize];
162         byteBuffer = ByteBuffer.wrap(buffer, 0, capacity);
163         this.useBoundary = useBoundary;
164     }
165 
166     /**
167      * Moves all buffered (but not yet read) bytes from 'buffer' to the destination buffer. In total, {@link #size()}
168      * bytes will be moved.
169      * @param destinationBuffer the destination buffer, which may be 'buffer' itself or a new buffer.
170      */
171     private void moveBytesToStartOfBuffer(byte[] destinationBuffer) {
172         if (size > 0) {
173             System.arraycopy(buffer, readIndex, destinationBuffer, 0, size);
174         }
175         if (readIndex > 0) {
176             notificationConsumer.bytesConsolidatedToStartOfBuffer(readIndex);
177         }
178         readIndex = 0;
179         boundary = available;
180         writeIndex = size;
181     }
182 
183     /**
184      * @return the number of bytes that can be written at the end of the buffer.
185      */
186     private int freeSpaceAtEndOfBuffer() {
187         return capacity - writeIndex;
188     }
189 
190     /**
191      * Ensures that there is at least 'minimumNumberOfBytesRequired' bytes of free space in the buffer, growing the
192      * buffer if necessary. May consolidate buffered bytes, performing an in-order copy and resetting indices
193      * such that the `readIndex` points to the same byte and the `writeIndex` is positioned after the last
194      * byte that is available to read.
195      * @param minimumNumberOfBytesRequired the minimum amount of free space that needs to be available for writing.
196      */
197     private void ensureSpaceInBuffer(int minimumNumberOfBytesRequired) {
198         if (size < 1 || freeSpaceAtEndOfBuffer() < minimumNumberOfBytesRequired) {
199             int shortfall = minimumNumberOfBytesRequired - freeSpaceAtEndOfBuffer() - readIndex;
200             if (shortfall <= 0) {
201                 // Free up space by moving any unread bytes to the start of the buffer and resetting the indices.
202                 moveBytesToStartOfBuffer(buffer);
203             } else {
204                 // There is not enough space in the buffer even though all available bytes have already been
205                 // moved to the start of the buffer. Growth is required.
206                 int amountToGrow = Math.max(initialBufferSize, shortfall);
207                 if (capacity + amountToGrow > maximumBufferSize) {
208                     amountToGrow = shortfall;
209                     if (capacity + amountToGrow > maximumBufferSize) {
210                         throw new BufferOverflowException();
211                     }
212                 }
213                 byte[] newBuffer = new byte[buffer.length + amountToGrow];
214                 moveBytesToStartOfBuffer(newBuffer);
215                 capacity += amountToGrow;
216                 buffer = newBuffer;
217                 byteBuffer = ByteBuffer.wrap(buffer, readIndex, capacity);
218             }
219         }
220     }

222     /**
223      * Buffers a single additional byte, growing the buffer if it is already full.
224      * @param b the byte to buffer.
225      */
226     public void receive(final int b) {
227         ensureSpaceInBuffer(1);
228         buffer[writeIndex] = (byte) b;
229         writeIndex++;
230         size++;
231         if (!useBoundary) {
232             extendBoundary(1);
233         }
234     }

236     /**
237      * Buffers `len` additional bytes, growing the buffer if it is already full or if it would become full
238      * by writing `len` bytes.
239      * @param b the bytes to buffer.
240      * @param off the offset into `b` that points to the first byte to buffer.
241      * @param len the number of bytes to buffer.
242      */
243     public void receive(final byte[] b, final int off, final int len) {
244         ensureSpaceInBuffer(len);
245         System.arraycopy(b, off, buffer, writeIndex, len);
246         writeIndex += len;
247         size += len;
248         if (!useBoundary) {
249             extendBoundary(len);
250         }
251     }

253     /**
254      * Buffers `b.length` additional bytes.
255      * @see #receive(byte[], int, int)
256      * @param b the bytes to buffer.
257      */
258     public void receive(final byte[] b) {
259         receive(b, 0, b.length);
260     }

262     /**
263      * Buffers up to `len` additional bytes, growing the buffer if it is already full or if it would become full
264      * by writing `len` bytes. This method will block if and only if the given `InputStream`'s
265      * {@link InputStream#read(byte[], int, int)} blocks when trying to read `len` bytes. If this is not desired,
266      * the caller should ensure that the given `InputStream` has at least `len` bytes available before calling
267      * this method or provide an InputStream implementation that does not block.
268      * @param input the source of the bytes.
269      * @param len the number of bytes to attempt to write.
270      * @return the number of bytes actually written, which will only be less than `len` if
271      * {@link InputStream#read(byte[], int, int)} returns less than `len`.
272      * @throws IOException if thrown by the given `InputStream` during read, except for {@link EOFException}. If an
273      *   EOFException is thrown by the `InputStream`, it will be caught and this method will return the number of bytes
274      *   that were received before the exception was thrown.
275      */
276     public int receive(final InputStream input, final int len) throws IOException  {
277         ensureSpaceInBuffer(len);
278         int numberOfBytesRead;
279         try {
280             numberOfBytesRead = input.read(buffer, writeIndex, len);
281         } catch (EOFException e) {
282             // Some InputStream implementations (such as GZIPInputStream) will throw EOFException instead of
283             // returning -1.
284             numberOfBytesRead = -1;
285         }
286         if (numberOfBytesRead > 0) {
287             writeIndex += numberOfBytesRead;
288             size += numberOfBytesRead;
289         } else {
290             numberOfBytesRead = 0;
291         }
292         if (!useBoundary) {
293             extendBoundary(numberOfBytesRead);
294         }
295         return numberOfBytesRead;
296     }

298     /**
299      * {@inheritDoc}
300      * <p>
301      * NOTE: This method adheres to the documented behavior of {@link InputStream#read(byte[], int, int)}
302      * <strong>except</strong> that it never blocks. If a read is attempted before the first write,
303      * this method will return -1.
304      */
305     @Override
306     public int read(final byte[] b, final int off, final int len) {
307         if (b.length == 0 || len == 0) {
308             return 0;
309         }
310         if (available < 1) {
311             return -1;
312         }
313         int bytesToRead = Math.min(available, len);
314         System.arraycopy(buffer, readIndex, b, off, bytesToRead);
315         readIndex += bytesToRead;
316         available -= bytesToRead;
317         size -= bytesToRead;
318         return bytesToRead;
319     }

321     /**
322      * Copies all of the available bytes in the buffer without changing the number of bytes available to subsequent
323      * reads.
324      * @param outputStream stream to which the bytes will be copied.
325      * @throws IOException if thrown by {@link OutputStream#write(byte[], int, int)}.
326      */
327     public void copyTo(final OutputStream outputStream) throws IOException {
328         outputStream.write(buffer, readIndex, available);
329     }

336         int amount = index - readIndex;

356     /**
357      * Rewinds the buffer to the given read index and sets 'available' to the given value. Subsequent
358      * behavior is undefined unless the values resulted from calling {@link #getReadIndex()} and
359      * {@link #available()} in immediate sequence, without any calls to 'receive' since.
360      * @param previousReadIndex the read index value to be set.
361      * @param previousAvailable the available value to be set.
362      */
363     void rewind(final int previousReadIndex, final int previousAvailable) {
364         readIndex = previousReadIndex;
365         available = previousAvailable;
366         boundary = previousReadIndex + previousAvailable;
367         size = writeIndex - readIndex;
368     }

370     /**
371      * Truncates the buffer to the given write index and sets both 'available' and 'size' to the to the given value.
372      * Subsequent behavior is undefined unless the values resulted from calling {@link #getWriteIndex()} and
373      * {@link #available()} in immediate sequence, without any calls to 'read' or 'skip' since. It is the caller's
374      * responsibility to ensure that calling this method will not result in loss of important data beyond the boundary.
375      * @param previousWriteIndex the write index value to be set.
376      * @param previousAvailable the available value to be set.
377      */
378     void truncate(final int previousWriteIndex, final int previousAvailable) {
379         writeIndex = previousWriteIndex;
380         available = previousAvailable;
381         boundary = writeIndex;
382         size = previousAvailable;
383     }

385     /**
386      * Skips up to `n` buffered bytes. Less than `n` bytes will be skipped if less than `n` bytes are
387      * available in the buffer.
388      * @param n the number of bytes to skip.
389      * @return the number of bytes actually skipped.
390      */
391     @Override
392     public long skip(final long n) {
393         if (n < 1 || available < 1) {
394             return 0;
395         }
396         int bytesSkipped = (int) Math.min(available, n);
397         readIndex += bytesSkipped;
398         available -= bytesSkipped;
399         size -= bytesSkipped;
400         return bytesSkipped;
401     }

440     /**
441      * Extends the boundary by the given number of bytes. It is the caller's responsibility to ensure that the
442      * resulting boundary includes only bytes that have been buffered (i.e. that it does not exceed `writeIndex`).
443      * @param numberOfBytes the number of bytes by which the boundary should be extended.
444      */
445     void extendBoundary(int numberOfBytes) {
446         boundary += numberOfBytes;
447         available += numberOfBytes;
448     }

450     /**
451      * {@inheritDoc}
452      * <p>
453      * NOTE: This method adheres to the documented behavior of {@link InputStream#read(byte[], int, int)}
454      * <strong>except</strong> that it never blocks. If a read is attempted before the first write,
455      * this method will return -1.
456      */
457     @Override
458     public int read() {
459         if (available < 1) {
460             return -1;
461         }
462         int b = buffer[readIndex];
463         readIndex++;
464         available--;
465         size--;
466         return b & SINGLE_BYTE_MASK;
467     }

469     /**
470      * Peeks the byte at the given index without modifying any internal indexes. It is the caller's responsibility to
471      * ensure that the given index points to an available byte.
472      * @param index the index of the byte to peek.
473      * @return the byte value.
474      */
475     int peek(int index) {
476         return buffer[index] & SINGLE_BYTE_MASK;
477     }

504     /**
505      * Returns a ByteBuffer view of the underlying buffer.
506      * @param position the start position of the ByteBuffer.
507      * @param limit the limit of the ByteBuffer.
508      * @return a ByteBuffer.
509      */
510     ByteBuffer getByteBuffer(int position, int limit) {
511         // Setting the limit to the capacity first is required because setting the position will fail if the new
512         // position is outside the limit.
513         byteBuffer.limit(capacity);
514         byteBuffer.position(position);
515         byteBuffer.limit(limit);
516         return byteBuffer;
517     }

520      * Copies bytes from the underlying buffer. It is the caller's responsibility to ensure the requested bytes
521      * are available.
522      * @param position the start position from which to read.
523      * @param destination the buffer to copy into.
524      * @param destinationOffset the offset of the buffer to copy into.
525      * @param length the number of bytes to copy.

528         System.arraycopy(buffer, position, destination, destinationOffset, length);
529     }

531     /**
532      * Moves all bytes starting at 'fromPosition' to 'toPosition', overwriting the bytes in-between. It is the caller's
533      * responsibility to ensure that the overwritten bytes are not needed.
534      * @param fromPosition the position to move bytes from. Must be less than or equal to 'writeIndex' and 'boundary'.
535      * @param toPosition the position to move bytes to. Must be greater than or equal to 'readIndex'.
536      */
537     void consolidate(int fromPosition, int toPosition) {
538         if (fromPosition > writeIndex || fromPosition > boundary || toPosition < readIndex) {
539             throw new IllegalArgumentException("Tried to consolidate using an index that violates the constraints.");
540         }
541         int indexShift = fromPosition - toPosition;
542         System.arraycopy(buffer, fromPosition, buffer, toPosition, writeIndex - fromPosition);
543         size -= indexShift;
544         available -= indexShift;
545         writeIndex -= indexShift;
546         boundary -= indexShift;
547         // readIndex does not need to change, because none of the consolidated bytes have been read yet.
548     }
