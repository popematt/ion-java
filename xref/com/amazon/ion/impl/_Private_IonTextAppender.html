56      private static final boolean[] IDENTIFIER_START_CHAR_FLAGS;
57      private static final boolean[] IDENTIFIER_FOLLOW_CHAR_FLAGS;
58      static
59      {
60          IDENTIFIER_START_CHAR_FLAGS = new boolean[256];
61          IDENTIFIER_FOLLOW_CHAR_FLAGS = new boolean[256];
62  
63          for (int ii='a'; ii<='z'; ii++) {
64              IDENTIFIER_START_CHAR_FLAGS[ii]  = true;
65              IDENTIFIER_FOLLOW_CHAR_FLAGS[ii] = true;
66          }
67          for (int ii='A'; ii<='Z'; ii++) {
68              IDENTIFIER_START_CHAR_FLAGS[ii]  = true;
69              IDENTIFIER_FOLLOW_CHAR_FLAGS[ii] = true;
70          }
71          IDENTIFIER_START_CHAR_FLAGS ['_'] = true;
72          IDENTIFIER_FOLLOW_CHAR_FLAGS['_'] = true;
73  
74          IDENTIFIER_START_CHAR_FLAGS ['$'] = true;
75          IDENTIFIER_FOLLOW_CHAR_FLAGS['$'] = true;
76  
77          for (int ii='0'; ii<='9'; ii++) {
78              IDENTIFIER_FOLLOW_CHAR_FLAGS[ii] = true;
79          }
80      }

82      public static boolean isIdentifierStart(int codePoint) {
83          return IDENTIFIER_START_CHAR_FLAGS[codePoint & 0xff] && is8bitValue(codePoint);
84      }

86      public static boolean isIdentifierPart(int codePoint) {
87          return IDENTIFIER_FOLLOW_CHAR_FLAGS[codePoint & 0xff] && is8bitValue(codePoint);
88      }

125      * Escapes for U+00 through U+FF, for use in double-quoted Ion strings.
126      * This includes escapes for all LATIN-1 code points U+80 through U+FF.
127      */
128     private static final String[] STRING_ESCAPE_CODES;
129     static
130     {
131         STRING_ESCAPE_CODES = new String[256];
132         STRING_ESCAPE_CODES[0x00] = "\\0";
133         STRING_ESCAPE_CODES[0x07] = "\\a";
134         STRING_ESCAPE_CODES[0x08] = "\\b";
135         STRING_ESCAPE_CODES['\t'] = "\\t";
136         STRING_ESCAPE_CODES['\n'] = "\\n";
137         STRING_ESCAPE_CODES[0x0B] = "\\v";
138         STRING_ESCAPE_CODES['\f'] = "\\f";
139         STRING_ESCAPE_CODES['\r'] = "\\r";
140         STRING_ESCAPE_CODES['\\'] = "\\\\";
141         STRING_ESCAPE_CODES['\"'] = "\\\"";
142         for (int i = 1; i < 0x20; ++i) {
143             if (STRING_ESCAPE_CODES[i] == null) {
144                 String s = Integer.toHexString(i);
145                 STRING_ESCAPE_CODES[i] = "\\x" + ZERO_PADDING[2 - s.length()] + s;
146             }
147         }
148         for (int i = 0x7F; i < 0x100; ++i) {
149             String s = Integer.toHexString(i);
150             STRING_ESCAPE_CODES[i] = "\\x" + s;
151         }
152     }

398     /**
399      * Determines whether the given text matches one of the Ion identifier
400      * keywords <code>null</code>, <code>true</code>, or <code>false</code>.
401      * <p>
402      * This does <em>not</em> check for non-identifier keywords such as
403      * <code>null.int</code>.
404      *
405      * @param text the symbol to check.
406      * @return <code>true</code> if the text is an identifier keyword.
407      */
408     public static boolean isIdentifierKeyword(CharSequence text)
409     {
410         int pos = 0;
411         int valuelen = text.length();
412 
413         if(valuelen == 0)
414         {
415             return false;
416         }
417 
418         boolean keyword = false;
419 
420         // there has to be at least 1 character or we wouldn't be here
421         switch (text.charAt(pos++)) {
422         case '$':
423             if (valuelen == 1) return false;
424             while (pos < valuelen) {
425                 char c = text.charAt(pos++);
426                 if (! isDecimalDigit(c)) return false;
427             }
428             return true;
429         case 'f':
430             if (valuelen == 5 //      'f'
431              && text.charAt(pos++) == 'a'
432              && text.charAt(pos++) == 'l'
433              && text.charAt(pos++) == 's'
434              && text.charAt(pos++) == 'e'
435             ) {
436                 keyword = true;
437             }
438             break;
439         case 'n':
440             if (valuelen == 4 //      'n'
441              && text.charAt(pos++) == 'u'
442              && text.charAt(pos++) == 'l'
443              && text.charAt(pos++) == 'l'
444             ) {
445                 keyword = true;
446             }
447             else if (valuelen == 3 // 'n'
448              && text.charAt(pos++) == 'a'
449              && text.charAt(pos++) == 'n'
450             ) {
451                 keyword = true;
452             }
453             break;
454         case 't':
455             if (valuelen == 4 //      't'
456              && text.charAt(pos++) == 'r'
457              && text.charAt(pos++) == 'u'
458              && text.charAt(pos++) == 'e'
459             ) {
460                 keyword = true;
461             }
462             break;
463         }
464 
465         return keyword;
466     }

469     /**
470      * Determines whether the text of a symbol requires (single) quotes.
471      *
472      * @param symbol must be a non-empty string.
473      * @param quoteOperators indicates whether the caller wants operators to be
474      * quoted; if <code>true</code> then operator symbols like <code>!=</code>
475      * will return <code>true</code>.
476      * has looser quoting requirements than other containers.
477      * @return <code>true</code> if the given symbol requires quoting.
478      *
479      * @throws NullPointerException
480      *         if <code>symbol</code> is <code>null</code>.
481      */
482     public static boolean symbolNeedsQuoting(CharSequence symbol,
483                                              boolean      quoteOperators)
484     {
485         int length = symbol.length();
486 
487         // If the symbol's text matches an Ion keyword or it's an empty symbol, we must quote it.
488         // Eg, the symbol 'false' and '' must be rendered quoted.
489         if(length == 0 || isIdentifierKeyword(symbol))
490         {
491             return true;
492         }
493 
494         char c = symbol.charAt(0);
495 
496         // Surrogates are neither identifierStart nor operatorPart, so the
497         // first one we hit will fall through and return true.
498         // TODO test that
499 
500         if (!quoteOperators && isOperatorPart(c))
501         {
502             for (int ii = 0; ii < length; ii++) {
503                 c = symbol.charAt(ii);
504                 // We don't need to look for escapes since all
505                 // operator characters are ASCII.
506                 if (!isOperatorPart(c)) {
507                     return true;
508                 }
509             }
510             return false;
511         }
512         else if (isIdentifierStart(c))
513         {
514             for (int ii = 0; ii < length; ii++) {
515                 c = symbol.charAt(ii);
516                 if ((c == '\'' || c < 32 || c > 126)
517                     || !isIdentifierPart(c))
518                 {
519                     return true;
520                 }
521             }
522             return false;
523         }
524 
525         // quote by default
526         return true;
527     }

573     private final void printCodePoints(CharSequence text, String[] escapes)
574         throws IOException
575     {
576         int len = text.length();
577         for (int i = 0; i < len; ++i)
578         {
579             // Find a span of non-escaped ASCII code points so we can write
580             // them as quickly as possible.
581             char c = 0;
582             int j;
583             for (j = i; j < len; ++j) {
584                 c = text.charAt(j);
585                 // The escapes array always includes U+80 through U+FF.
586                 if (c >= 0x100 || escapes[c] != null)
587                 {
588                     // c is escaped and/or outside ASCII range.
589                     if (j > i) {
590                         appendAscii(text, i, j);
591                         i = j;
592                     }
593                     break;
594                 }
595             }
596             if (j == len) {
597                 // we've reached the end of sequence; append it and break
598                 appendAscii(text, i, j);
599                 break;
600             }
601 
602             // We've found a code point that's escaped and/or non-ASCII.
603 
604             if (c < 0x80)
605             {
606                 // An escaped ASCII character.
607                 assert escapes[c] != null;
608                 appendAscii(escapes[c]);
609             }
610             else if (c < 0x100)
611             {
612                 // Non-ASCII LATIN-1; we will have an escape sequence but may
613                 // not use it.
614                 assert escapes[c] != null;
615 
616                 // Always escape the C1 control codes U+80 through U+9F.
617                 if (escapeNonAscii || c <= 0x9F) {
618                     appendAscii(escapes[c]);
619                 } else {
620                     appendUtf16(c);
621                 }
622             }
623             else if (c < 0xD800 || c >= 0xE000)
624             {
625                 // Not LATIN-1, but still in the BMP.
626                 String s = Integer.toHexString(c);
627                 if (escapeNonAscii) {
628                     appendAscii(HEX_4_PREFIX);
629                     appendAscii(ZERO_PADDING[4 - s.length()]);
630                     appendAscii(s);
631                 } else {
632                     appendUtf16(c);
633                 }
634             }
635             else if (isHighSurrogate(c))
636             {
637                 // Outside the BMP! High surrogate must be followed by low.
638                 char c2;
639                 if (++i == len || !isLowSurrogate(c2 = text.charAt(i))) {
640                     String message =
641                         "text is invalid UTF-16. It contains an unmatched " +
642                         "leading surrogate 0x" + Integer.toHexString(c) +
643                         " at index " + (i-1);
644                     throw new IllegalArgumentException(message);
645                 }
646                 if (escapeNonAscii) {
647                     int cp = makeUnicodeScalar(c, c2);
648                     String s = Integer.toHexString(cp);
649                     appendAscii(HEX_8_PREFIX);
650                     appendAscii(ZERO_PADDING[8 - s.length()]);
651                     appendAscii(s);
652                 } else {
653                     appendUtf16Surrogate(c, c2);
654                 }
655             }
656             else
657             {
658                 // unmatched low surrogate
659                 assert isLowSurrogate(c);
660 
661                 String message =
662                     "text is invalid UTF-16. It contains an unmatched " +
663                     "trailing surrogate 0x" + Integer.toHexString(c) +
664                     " at index " + i;
665                 throw new IllegalArgumentException(message);
666             }
667         }
668     }

675     /** ONLY FOR USE BY {@link #printInt(long)}. */
676     private final char[] _fixedIntBuffer = new char[MAX_LONG_TEXT_SIZE];
677 
678     public void printInt(long value)
679         throws IOException
680     {
681         int j = _fixedIntBuffer.length;
682         if (value == 0) {
683             _fixedIntBuffer[--j] = '0';
684         } else {
685             if (value < 0) {
686                 while (value != 0) {
687                     _fixedIntBuffer[--j] = (char)(0x30 - value % 10);
688                     value /= 10;
689                 }
690                 _fixedIntBuffer[--j] = '-';
691             } else {
692                 while (value != 0) {
693                     _fixedIntBuffer[--j] = (char)(0x30 + value % 10);
694                     value /= 10;
695                 }
696             }
697         }
698 
699         // Using CharBuffer avoids copying the _fixedIntBuffer into a String
700         appendAscii(CharBuffer.wrap(_fixedIntBuffer),
701                     j,
702                     _fixedIntBuffer.length);
703     }

719     public void printDecimal(_Private_IonTextWriterBuilder _options,
720                              BigDecimal                    value)
721         throws IOException
722     {
723         if (value == null)
724         {
725             appendAscii("null.decimal");
726             return;
727         }
728 
729         BigInteger unscaled = value.unscaledValue();
730 
731         int signum = value.signum();
732         if (signum < 0)
733         {
734             appendAscii('-');
735             unscaled = unscaled.negate();
736         }
737         else if (value instanceof Decimal
738              && ((Decimal)value).isNegativeZero())
739         {
740             // for the various forms of negative zero we have to
741             // write the sign ourselves, since neither BigInteger
742             // nor BigDecimal recognize negative zero, but Ion does.
743             appendAscii('-');
744         }
745 
746         final String unscaledText = unscaled.toString();
747         final int significantDigits = unscaledText.length();
748 
749         final int scale = value.scale();
750         final int exponent = -scale;
751 
752         if (_options._decimal_as_float)
753         {
754             appendAscii(unscaledText);
755             appendAscii('e');
756             appendAscii(Integer.toString(exponent));
757         }
758         else if (exponent == 0)
759         {
760             appendAscii(unscaledText);
761             appendAscii('.');
762         }
763         else if (exponent < 0)
764         {
765             // Avoid printing small negative exponents using a heuristic
766             // adapted from http://speleotrove.com/decimal/daconvs.html
767 
768             final int adjustedExponent = significantDigits - 1 - scale;
769             if (adjustedExponent >= 0)
770             {
771                 int wholeDigits = significantDigits - scale;
772                 appendAscii(unscaledText, 0, wholeDigits);
773                 appendAscii('.');
774                 appendAscii(unscaledText, wholeDigits,
775                                     significantDigits);
776             }
777             else if (adjustedExponent >= -6)
778             {
779                 appendAscii("0.");
780                 appendAscii("00000", 0, scale - significantDigits);
781                 appendAscii(unscaledText);
782             }
783             else
784             {
785                 appendAscii(unscaledText);
786                 appendAscii("d-");
787                 appendAscii(Integer.toString(scale));
788             }
789         }
790         else // (exponent > 0)
791         {
792             // We cannot move the decimal point to the right, adding
793             // rightmost zeros, because that would alter the precision.
794             appendAscii(unscaledText);
795             appendAscii('d');
796             appendAscii(Integer.toString(exponent));
797         }
798     }

801     public void printFloat(_Private_IonTextWriterBuilder _options, double value)
802         throws IOException
803     {
804         // shortcut zero cases
805         if (value == 0.0)
806         {
807             if (Double.compare(value, 0d) == 0)  // Only matches positive zero
808             {
809                 appendAscii("0e0");
810             }
811             else
812             {
813                 appendAscii("-0e0");
814             }
815         }
816         else if (Double.isNaN(value))
817         {
818             if (_options._float_nan_and_inf_as_null) {
819                 appendAscii("null");
820             } else {
821                 appendAscii("nan");
822             }
823         }
824         else if (value == Double.POSITIVE_INFINITY)
825         {
826             if (_options._float_nan_and_inf_as_null) {
827                 appendAscii("null");
828             } else {
829                 appendAscii("+inf");
830             }
831         }
832         else if (value == Double.NEGATIVE_INFINITY)
833         {
834             if (_options._float_nan_and_inf_as_null) {
835                 appendAscii("null");
836             } else {
837                 appendAscii("-inf");
838             }
839         }
840         else
841         {
842             // Double.toString() forces a digit after the decimal point.
843             // Remove it when it's not meaningful.
844             String str = Double.toString(value);
845             if (str.endsWith(".0"))
846             {
847                 appendAscii(str, 0, str.length() - 2);
848                 appendAscii("e0");
849             }
850             else
851             {
852                 appendAscii(str);
853                 if (str.indexOf('E') == -1)
854                 {
855                     appendAscii("e0");
856                 }
857             }
858         }
859     }

879     public void printBlob(_Private_IonTextWriterBuilder _options,
880                           byte[] value, int start, int len)
881         throws IOException
882     {
883         if (value == null)
884         {
885             appendAscii("null.blob");
886             return;
887         }
888 
889         @SuppressWarnings("resource")
890         TextStream ts =
891             new TextStream(new ByteArrayInputStream(value, start, len));
892 
893         // base64 encoding is 6 bits per char so
894         // it evens out at 3 bytes in 4 characters
895         char[] buf = new char[_options.isPrettyPrintOn() ? 80 : 400];
896         CharBuffer cb = CharBuffer.wrap(buf);
897 
898         if (_options._blob_as_string)
899         {
900             appendAscii('"');
901         }
902         else
903         {
904             appendAscii("{{");
905             if (_options.isPrettyPrintOn())
906             {
907                 appendAscii(' ');
908             }
909         }
910 
911         for (;;)
912         {
913             // TODO is it better to fill up the CharBuffer before outputting?
914             int clen = ts.read(buf, 0, buf.length);
915             if (clen < 1) break;
916             appendAscii(cb, 0, clen);
917         }
918 
919         if (_options._blob_as_string)
920         {
921             appendAscii('"');
922         }
923         else
924         {
925             if (_options.isPrettyPrintOn())
926             {
927                 appendAscii(' ');
928             }
929             appendAscii("}}");
930         }
931     }

934     private void printClobBytes(byte[] value, int start, int end,
935                                 String[] escapes)
936         throws IOException
937     {
938         for (int i = start; i < end; i++) {
939             char c = (char)(value[i] & 0xff);
940             String escapedByte = escapes[c];
941             if (escapedByte != null) {
942                 appendAscii(escapedByte);
943             } else {
944                 appendAscii(c);
945             }
946         }
947     }

950     public void printClob(_Private_IonTextWriterBuilder _options,
951                           byte[] value, int start, int len)
952         throws IOException
953     {
954         if (value == null)
955         {
956             appendAscii("null.clob");
957             return;
958         }
959 
960 
961         final boolean json =
962             _options._clob_as_string && _options._string_as_json;
963 
964         final int threshold = _options.getLongStringThreshold();
965         final boolean longString = (0 < threshold && threshold < value.length);
966 
967         if (!_options._clob_as_string)
968         {
969             appendAscii("{{");
970             if (_options.isPrettyPrintOn())
971             {
972                 appendAscii(' ');
973             }
974         }
975 
976         if (json)
977         {
978             appendAscii('"');
979             printClobBytes(value, start, start + len, JSON_ESCAPE_CODES);
980             appendAscii('"');
981         }
982         else if (longString)
983         {
984             // This may escape more often than is necessary, but doing it
985             // minimally is very tricky. Must be sure to account for
986             // quotes at the end of the content.
987 
988             // TODO Account for NL versus CR+NL streams
989             appendAscii(TRIPLE_QUOTES);
990             printClobBytes(value, start, start + len, LONG_STRING_ESCAPE_CODES);
991             appendAscii(TRIPLE_QUOTES);
992         }
993         else
994         {
995             appendAscii('"');
996             printClobBytes(value, start, start + len, STRING_ESCAPE_CODES);
997             appendAscii('"');
998         }
999 
1000         if (! _options._clob_as_string)
1001         {
1002             if (_options.isPrettyPrintOn())
1003             {
1004                 appendAscii(' ');
1005             }
1006             appendAscii("}}");
1007         }
