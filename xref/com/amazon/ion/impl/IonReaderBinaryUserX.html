105     /**
106      * Determines the abstract position of the reader, such that one can
107      * later {@link #seek} back to it.
108      * <p>
109      * The current implementation only works when the reader is positioned on
110      * a value (not before, between, or after values). In other words, one
111      * should only call this method when {@link #getType()} is non-null.
112      *
113      * @param beforeTid -
114      *          When true, the position returned starts before the
115      *          type/length octet.
116      *          When false, the position returned starts after the
117      *          type/length octet and any optional length octets.
118      *
119      * @return the current position; not null.
120      *
121      * @throws IllegalStateException if the reader doesn't have a current
122      * value.
123      */
124     public Span getCurrentPosition(boolean beforeTid)
125     {
126         if (getType() == null)
127         {
128             throw new IllegalStateException("IonReader isn't positioned on a value");
129         }
130         // Only spans that include the TID octet are seekable.
131         IonReaderBinarySpan pos = new IonReaderBinarySpan(beforeTid);
132         long start = beforeTid ?  _position_start : _value_start;
133         long len = beforeTid ? _position_len : _value_len;
134         pos._offset = start - _physical_start_offset;
135         pos._limit = pos._offset + len;
136         pos._symbol_table = _symbols;
137         pos._state = _state;
138         return pos;
139     }

147     public void seek(IonReaderBinarySpan position)
148     {
149         IonReaderBinarySpan pos = position;
150 
151         if (pos == null)
152         {
153             throw new IllegalArgumentException("Position invalid for binary reader");
154         }
155         if (!(_input instanceof FromByteArray))
156         {
157             throw new UnsupportedOperationException("Binary seek not implemented for non-byte array backed sources");
158         }
159 
160         // TODO test that span is within the bounds of the input byte[]
161 
162         // manually reset the input specific type of input stream
163         FromByteArray input = (FromByteArray)_input;
164         input._pos   = (int) (pos._offset + _physical_start_offset);
165         input._limit = (int) (pos._limit  + _physical_start_offset);
166 
167         // TODO: these (eof and save points) should be put into
168         //       a re-init method on the input stream
169         input._eof = false;
170         for (;;) {
171             SavePoint sp = input._save_points._active_stack;
172             if (sp == null) break;
173             input._save_points.savePointPopActive(sp);
174             sp.free();
175         }
176 
177         // reset the raw reader
178         re_init_raw();
179 
180         // reset the system reader
181         // - nothing to do
182 
183         // reset the user reader
184         init_user(this._catalog);
185 
186         // now we need to set our symbol table
187         _symbols = pos._symbol_table;
188 
189         // and the other misc state variables we had
190         // read past before getPosition gets called
191         //           Don't do this, we'll re-read the data from the stream.
192         //           Otherwise, this reader will be in the wrong state.
193         //           For example, getType() will return non-null but that
194         //           shouldn't happen until the user calls next().
195 //        _state         = pos._state;
196 //        _value_type    = pos._value_type;
197 //        _value_is_null = pos._value_is_null;
198 //        _value_is_true = pos._value_is_true;
199 
200 //        _is_in_struct = false;
201     }

204     @Override
205     public IonType next()
206     {
207         IonType t = null;
208         if (hasNext()) {
209             _has_next_needed = true;
210             t = _value_type;
211         }
212         return t;
213     }
214 
215     @Override
216     public boolean hasNext()
217     {
218         if (!_eof && _has_next_needed) {
219             clear_system_value_stack();
220             try {
221                 while (!_eof && _has_next_needed) {
222                     has_next_helper_user();
223                 }
224             }
225             catch (IOException e) {
226                 error(e);
227             }
228         }
229         return !_eof;
230     }

233     private final void has_next_helper_user() throws IOException
234     {
235         super.hasNext();
236         if (getDepth() == 0 && !_value_is_null) {
237             if (_value_tid == _Private_IonConstants.tidSymbol) {
238                 if (load_annotations() == 0) {
239                     // $ion_1_0 is read as an IVM only if it is not annotated
240                     load_cached_value(AS_TYPE.int_value);
241                     int sid = _v.getInt();
242                     if (sid == ION_1_0_SID) {
243                         _symbols = SharedSymbolTable.getSystemSymbolTable(1);
244                         push_symbol_table(_symbols);
245                         _has_next_needed = true;
246                     }
247                 }
248             }
249             else if (_value_tid == _Private_IonConstants.tidStruct) {
250                 int count = load_annotations();
251                 if (count > 0 && _annotation_ids[0] == ION_SYMBOL_TABLE_SID) {
252                     _symbols = _lstFactory.newLocalSymtab(_catalog, this, false);
253                     push_symbol_table(_symbols);
254                     _has_next_needed = true;
255                 }
256             }
257             else {
258                 assert (_value_tid != _Private_IonConstants.tidTypedecl);
259             }
260         }
261     }

294     //  This code handles the skipped symbol table
295     //  support - it is cloned in IonReaderTextUserX,
296     //  IonReaderBinaryUserX and _Private_IonWriterBase

298     //  SO ANY FIXES HERE WILL BE NEEDED IN THOSE

301     private int _symbol_table_top = 0;
302     private SymbolTablebolTable[] _symbol_table_stack = new SymbolTable[3]; // 3 is rare, IVM followed by a local sym tab with open content
303     private void clear_system_value_stack()
304     {
305         while (_symbol_table_top > 0) {
306             _symbol_table_top--;
307             _symbol_table_stack[_symbol_table_top] = null;
308         }
309     }
310     private void push_symbol_table(SymbolTable symbols)
311     {
312         assert(symbols != null);
313         if (_symbol_table_top >= _symbol_table_stack.length) {
314             int new_len = _symbol_table_stack.length * 2;
315             SymbolTableymbolTable">SymbolTable[] temp = new SymbolTable[new_len];
316             System.arraycopy(_symbol_table_stack, 0, temp, 0, _symbol_table_stack.length);
317             _symbol_table_stack = temp;
318         }
319         _symbol_table_stack[_symbol_table_top++] = symbols;
320     }
321     @Override
322     public SymbolTable pop_passed_symbol_table()
323     {
324         if (_symbol_table_top <= 0) {
325             return null;
326         }
327         _symbol_table_top--;
328         SymbolTable symbols = _symbol_table_stack[_symbol_table_top];
329         _symbol_table_stack[_symbol_table_top] = null;
330         return symbols;
331     }

338     @Override
339     public <T> T asFacet(Class<T> facetType)
340     {
341         if (facetType == SpanProvider.class)
342         {
343             return facetType.cast(new SpanProviderFacet());
344         }
345 
346         // TODO amzn/ion-java/issues/17 support seeking over InputStream
347         if (_input instanceof FromByteArray)
348         {
349             if (facetType == SeekableReader.class)
350             {
351                 return facetType.cast(new SeekableReaderFacet());
352             }
353             if (facetType == RawValueSpanProvider.class)
354             {
355                 return facetType.cast(new RawValueSpanProviderFacet());
356             }
357         }
358 
359         if (facetType == _Private_ByteTransferReader.class)
360         {
361             // This is a rather sketchy use of Facets, since the availability
362             // of the facet depends upon the current state of this subject,
363             // and that can change over time.
364 
365             // TODO amzn/ion-java/issues/16 Our {@link #transferCurrentValue} doesn't handle
366             //  field names and annotations.
367 
368             // Ensure there's a contiguous buffer we can copy.
369             if (_input instanceof UnifiedInputStreamX.FromByteArray
370                 && getTypeAnnotationSymbols().length == 0
371                 && ! isInStruct())
372             {
373                 return facetType.cast(new ByteTransferReaderFacet());
374             }
375         }
376 
377         return super.asFacet(facetType);
378     }

423     {
424         public void transferCurrentValue(_Private_ByteTransferSink sink)
425             throws IOException
426         {
427             // Ensure there's a contiguous buffer we can copy.
428             // TODO Copy from a stream should also be possible.
429             if (! (_input instanceof UnifiedInputStreamX.FromByteArray))

432             }

436             // TODO when implementing that, be careful to handle the case where

438             //   Meaning: the user has set it and then called writeValue().
