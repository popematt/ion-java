36  /**
37   * Reader implementation that reads the token stream and validates
38   * the Ion grammar.  This does not care about system values.  It
39   * does not materialize values or convert them.  It does mark values
40   * in the UnifiedInputStream if they might be field names or annotations
41   * since it does populate these properties directly. Otherwise it
42   * accepts the TextRawToken's assessment of the type of the next
43   * token, which is based on as few characters as possible, typically
44   * 1 but generally less than 5.
45   *
46   * This is called by the {@link IonReaderTextSystemX}, which in turn is most
47   * often called by the {@link IonReaderTextUserX}.  One of these two (system
48   * reader or user reader) should be invoked by the user for reading text Ion
49   * data.  This class is not intended for general use.
50   *
51   * This reader scan skip values and in doing so it does not
52   * materialize the contents and it does not validate the contents.
53   * TODO amzn/ion-java/issues/7 We may want to make validation on skip optional.
54   *
55   * This manages the value buffer (_v ValueVariant) and the lob
56   * content (_lob_*) which is cached in some cases.  It's main
57   * job however is recognizing the correct order of the input
58   * tokens.  This is done in parse_to_next_value (called by hasNext).
59   *
60   * The current state is represented by an int (whose value should
61   * be one of the values of the STATE_* constants).  The legal
62   * transitions are stored in TransitionActions and TransitionActions2.
63   * The first (TransitionActions) is a two dimensional array whose
64   * dimensions are state and input token. The value stored is an
65   * int that represents the action to be taken (ACTION_*).  The
66   * second copy of this data (TransitionActions2) is a one dimensional
67   * array built from the first and manually dereferenced in the
68   * parse_to_next_value method.  This turns out to be a significant
69   * performance gain (<sigh>).  Logically these are the same.
70   *
71   */
72  abstract class IonReaderTextRawX
73      implements IonTextReader
74  {
75      public abstract BigInteger bigIntegerValue();

166         actions[STATE_BEFORE_ANNOTATION_DATAGRAM][IonTokenConstsX.TOKEN_TIMESTAMP]          = ACTION_LOAD_SCALAR;
167         actions[STATE_BEFORE_ANNOTATION_DATAGRAM][IonTokenConstsX.TOKEN_STRING_DOUBLE_QUOTE]= ACTION_LOAD_SCALAR;
168         actions[STATE_BEFORE_ANNOTATION_DATAGRAM][IonTokenConstsX.TOKEN_STRING_TRIPLE_QUOTE]= ACTION_LOAD_SCALAR;
169         actions[STATE_BEFORE_ANNOTATION_DATAGRAM][IonTokenConstsX.TOKEN_SYMBOL_IDENTIFIER]  = ACTION_LOAD_ANNOTATION;
170         actions[STATE_BEFORE_ANNOTATION_DATAGRAM][IonTokenConstsX.TOKEN_SYMBOL_QUOTED]      = ACTION_LOAD_ANNOTATION;
171         actions[STATE_BEFORE_ANNOTATION_DATAGRAM][IonTokenConstsX.TOKEN_OPEN_PAREN]         = ACTION_START_SEXP;
172         actions[STATE_BEFORE_ANNOTATION_DATAGRAM][IonTokenConstsX.TOKEN_OPEN_BRACE]         = ACTION_START_STRUCT;
173         actions[STATE_BEFORE_ANNOTATION_DATAGRAM][IonTokenConstsX.TOKEN_OPEN_SQUARE]        = ACTION_START_LIST;
174         actions[STATE_BEFORE_ANNOTATION_DATAGRAM][IonTokenConstsX.TOKEN_OPEN_DOUBLE_BRACE]  = ACTION_START_LOB;
175 
176         // both before_annotation and after_annotation are essentially the same as
177         // BOF (after_annotation can't accept EOF as valid however)
178         for (int ii=0; ii<IonTokenConstsX.TOKEN_MAX+1; ii++) {
179             actions[STATE_BEFORE_ANNOTATION_CONTAINED][ii] = actions[STATE_BEFORE_ANNOTATION_DATAGRAM][ii];
180             actions[STATE_BEFORE_ANNOTATION_SEXP][ii]      = actions[STATE_BEFORE_ANNOTATION_DATAGRAM][ii];

185         actions[STATE_BEFORE_ANNOTATION_CONTAINED][IonTokenConstsX.TOKEN_EOF]            = 0;
186         actions[STATE_BEFORE_ANNOTATION_CONTAINED][IonTokenConstsX.TOKEN_CLOSE_PAREN]    = ACTION_FINISH_CONTAINER;
187         actions[STATE_BEFORE_ANNOTATION_CONTAINED][IonTokenConstsX.TOKEN_CLOSE_BRACE]    = 0;

202         actions[STATE_BEFORE_VALUE_CONTENT_SEXP][IonTokenConstsX.TOKEN_SYMBOL_IDENTIFIER]= ACTION_LOAD_SCALAR;
203         actions[STATE_BEFORE_VALUE_CONTENT_SEXP][IonTokenConstsX.TOKEN_SYMBOL_QUOTED]    = ACTION_LOAD_SCALAR;
204         actions[STATE_BEFORE_VALUE_CONTENT_SEXP][IonTokenConstsX.TOKEN_SYMBOL_OPERATOR]  = ACTION_LOAD_SCALAR;
205 
206         actions[STATE_BEFORE_FIELD_NAME][IonTokenConstsX.TOKEN_EOF]                      = 0;
207         actions[STATE_BEFORE_FIELD_NAME][IonTokenConstsX.TOKEN_SYMBOL_IDENTIFIER]        = ACTION_LOAD_FIELD_NAME;
208         actions[STATE_BEFORE_FIELD_NAME][IonTokenConstsX.TOKEN_SYMBOL_QUOTED]            = ACTION_LOAD_FIELD_NAME;
209         actions[STATE_BEFORE_FIELD_NAME][IonTokenConstsX.TOKEN_STRING_DOUBLE_QUOTE]      = ACTION_LOAD_FIELD_NAME;
210         actions[STATE_BEFORE_FIELD_NAME][IonTokenConstsX.TOKEN_STRING_TRIPLE_QUOTE]      = ACTION_LOAD_FIELD_NAME;

212         actions[STATE_BEFORE_FIELD_NAME][IonTokenConstsX.TOKEN_CLOSE_BRACE]              = ACTION_FINISH_CONTAINER;
213         actions[STATE_BEFORE_FIELD_NAME][IonTokenConstsX.TOKEN_CLOSE_SQUARE]             = ACTION_FINISH_CONTAINER;
214 
215          // after a value we'll either see a separator (like ',')
216          // or a containers closing token. If we're not in a container
217          // (i.e. we're at the top level) then this isn't the state we
218          // should be in.  We'll be in STATE_BEFORE_ANNOTATION_DATAGRAM
219          actions[STATE_AFTER_VALUE_CONTENTS][IonTokenConstsX.TOKEN_COMMA]                = ACTION_EAT_COMMA;
220          actions[STATE_AFTER_VALUE_CONTENTS][IonTokenConstsX.TOKEN_CLOSE_PAREN]          = ACTION_FINISH_CONTAINER;
221          actions[STATE_AFTER_VALUE_CONTENTS][IonTokenConstsX.TOKEN_CLOSE_BRACE]          = ACTION_FINISH_CONTAINER;
222          actions[STATE_AFTER_VALUE_CONTENTS][IonTokenConstsX.TOKEN_CLOSE_SQUARE]         = ACTION_FINISH_CONTAINER;
223 
224          // the three "in_<lob>" value states have to be handled
225          // specially, they can only scan forward to the end of
226          // the content on next, or read content for the user otherwise
227          actions[STATE_IN_CLOB_DOUBLE_QUOTED_CONTENT][IonTokenConstsX.TOKEN_CLOSE_BRACE] = ACTION_FINISH_LOB;
228          actions[STATE_IN_CLOB_TRIPLE_QUOTED_CONTENT][IonTokenConstsX.TOKEN_CLOSE_BRACE] = ACTION_FINISH_LOB;
229          actions[STATE_IN_BLOB_CONTENT][IonTokenConstsX.TOKEN_CLOSE_BRACE]               = ACTION_FINISH_LOB;
230 
231          // the eof action exists because finishing an unread value can place the scanner just before
232          // the input stream eof and set the current state to eof - in which case we just need to return eof
233          for (int ii=0; ii<IonTokenConstsX.TOKEN_MAX+1; ii++) {
234              actions[STATE_EOF][ii] =  ACTION_EOF;
235          }
236 
237          return actions;
238     }

240     static final int[] TransitionActions2 = makeTransition2ActionArray();
241     static int[] makeTransition2ActionArray() {
242         int   s, s_count = STATE_MAX + 1;
243         int   t, t_count = IonTokenConstsX.TOKEN_MAX + 1;
244         int[] a = new int[s_count * t_count];
245         for (s = 0; s < s_count; s++) {
246             for (t=0; t < t_count; t++) {
247                 int ii = s * IonTokenConstsX.TOKEN_count + t;
248                 a[ii] = TransitionActions[s][t];
249             }
250         }
251         return a;
252     }

431         // empty text is checked by caller
432         int oldlen = _annotations.length;
433         if (_annotation_count >= oldlen) {

446     /**
447      * this looks forward to see if there is an upcoming value
448      * if there is it returns true.  It may have to clean up
449      * any value that's partially complete (for example a
450      * collection whose annotation has been read and loaded
451      * but the user has chosen not to step into the collection).
452      * @return true if more data remains, false on eof
453      */
454     public boolean hasNext()
455     {
456         boolean has_next = has_next_raw_value();
457         return has_next;
458     }
459     protected final boolean has_next_raw_value() {
460         if (!_has_next_called && !_eof) {
461             try {
462                 finish_value(null);
463                 clear_value();
464                 parse_to_next_value();
465             }
466             catch (IOException e) {
467                 throw new IonException(e);
468             }
469             _has_next_called = true;
470         }
471         return (_eof != true);
472     }

474     /**
475      * returns the type of the next value in the stream.
476      * it calls hasNext to assure that the value has been properly
477      * started, since hasNext prepares a value as a side effect of
478      * determining whether or not a value is pending.
479      * A NoSuchElementException is thrown if there are not values remaining.
480      * Once called if there is a value available it's contents can
481      * be accessed through the other public API's (such as getLong()).
482      * @return type of the next value, or null if there is none.
483      */
484     public IonType next()
485     {
486         if (!hasNext()) {
487             return null;
488         }
489         if (_value_type == null && _scanner.isUnfinishedToken()) {
490             try {
491                 token_contents_load(_scanner.getToken());
492             }
493             catch (IOException e) {
494                 throw new IonException(e);
495             }
496         }
497         _has_next_called = false;
498         return _value_type;
499     }
500     private final void finish_and_save_value() throws IOException
501     {
502         if (!_current_value_save_point_loaded) {
503             _scanner.save_point_start(_current_value_save_point);
504             finish_value(_current_value_save_point);
505             _current_value_save_point_loaded = true;
506         }

508     private final void finish_value(SavePoint sp) throws IOException
509     {
510         if (_scanner.isUnfinishedToken()) {
511             if (sp != null && _value_type != null) {
512                 switch (_value_type) {
513                 case STRUCT:
514                 case SEXP:
515                 case LIST:
516                     sp = null;
517                     break;
518                 default:
519                     break;
520                 }
521             }
522             _scanner.finish_token(sp);
523 
524             int new_state = get_state_after_value();
525             set_state(new_state);
526         }
527         _has_next_called = false;
528     }

736     private final SymbolToken parseSymbolToken(String context,
737                                                   StringBuilder sb,
738                                                   int t)
739         throws IOException
740     {
741         String text;
742         int sid;
743 
744         if (t == IonTokenConstsX.TOKEN_SYMBOL_IDENTIFIER) {
745             int kw = IonTokenConstsX.keyword(sb, 0, sb.length());
746             switch (kw) {
747                 case IonTokenConstsX.KEYWORD_FALSE:
748                 case IonTokenConstsX.KEYWORD_TRUE:
749                 case IonTokenConstsX.KEYWORD_NULL:
750                 case IonTokenConstsX.KEYWORD_NAN:
751                     // keywords are not ok unless they're quoted
752                     String reason =
753                     "Cannot use unquoted keyword " +
754                         sb.toString() + " as " + context;
755                     parse_error(reason);
756                 case IonTokenConstsX.KEYWORD_sid:
757                     text = null;
758                     sid = IonTokenConstsX.decodeSid(sb);
759                     break;
760                 default:
761                     text = sb.toString();
762                     sid = UNKNOWN_SYMBOL_ID;
763                     break;
764             }
765         }
766         else {
767             text = sb.toString();
768             sid = UNKNOWN_SYMBOL_ID;
769         }
770 
771         return new SymbolTokenImpl(text, sid);
772     }

775     protected final void parse_to_next_value() throws IOException
776     {
777         int t;
778         int action, temp_state;
779         boolean trailing_whitespace = false;  // TODO: there's a better way to do this
780         StringBuilder sb;
781 
782         // FIXME: check depth and type before doing anything further
783         //        if we're on a collection and at the correct depth
784         //        we need to skip over the contents of the collection
785         //        before doing any more parsing
786 
787         // we'll need a token to get started here
788         // we'll also remember where we were when we started if the
789         // user later wants to get a span over this value.  In the
790         // case where we just before a comma, after the comma we'll
791         // reset this offset since for the span the comma isn't part
792         // of the span when it's hoisted
793         _value_start_offset = _scanner.getStartingOffset();
794         _value_start_line   = _scanner.getLineNumber();
795         _value_start_column = _scanner.getLineOffset();
796 
797         t = _scanner.nextToken();
798 
799         for (;;) {
800             int idx = get_state_int() * IonTokenConstsX.TOKEN_count + t;
801             action = TransitionActions2[idx];
802             // this used to be (but the 2d array is 9072ms vs 8786ms
803             // timing, 3% of total file parse time!):
804             // action = TransitionActions[get_state_int()][t];
805             switch (action) {
806             case ACTION_NOT_DEFINED:
807                 {
808                     // TODO why would we get here?
809                     boolean span_eof = false;
810 
811                     if (_nesting_parent != null) {
812                         switch (_nesting_parent) {
813                             case LIST:
814                                 if (t == IonTokenConstsX.TOKEN_CLOSE_SQUARE) {
815                                     span_eof = true;
816                                 }
817                                 break;
818                             case SEXP:
819                                 if (t == IonTokenConstsX.TOKEN_CLOSE_PAREN){
820                                     span_eof = true;
821                                 }
822                                 break;
823                             case STRUCT:
824                                 if (t == IonTokenConstsX.TOKEN_CLOSE_BRACE) {
825                                     span_eof = true;
826                                 }
827                                 break;
828                             default:
829                                 break;
830                         }
831                     }
832                     if (span_eof != true) {
833                         String message = "invalid syntax [state:"
834                                        + get_state_name()
835                                        + " on token:"
836                                        +IonTokenConstsX.getTokenName(t)
837                                        +"]";
838                         parse_error(message);
839                     }
840                     set_state(STATE_EOF);
841                     _eof = true;
842                     return;
843                 }
844             case ACTION_EOF:
845                 set_state(STATE_EOF);
846                 _eof = true;
847                 return;
848             case ACTION_LOAD_FIELD_NAME:
849             {
850                 if (!is_in_struct_internal()) {
851                     throw new IllegalStateException("field names have to be in structs");
852                 }
853                 //finish_value(_current_value_save_point);
854                 finish_and_save_value();
855 
856                 sb = token_contents_load(t);
857 
858                 SymbolToken sym = parseSymbolToken("a field name", sb, t);
859                 set_fieldname(sym);
860                 clear_current_value_buffer();
861 
862                 t = _scanner.nextToken();
863                 if (t != IonTokenConstsX.TOKEN_COLON) {
864                     String message = "field name must be followed by a colon, not a "
865                                    + IonTokenConstsX.getTokenName(t);
866                     parse_error(message);
867                 }
868                 _scanner.tokenIsFinished();
869                 set_state(STATE_BEFORE_ANNOTATION_CONTAINED);
870                 t = _scanner.nextToken();
871                 break;
872             }
873             case ACTION_LOAD_ANNOTATION:
874             {
875                 sb = token_contents_load(t);
876 
877                 trailing_whitespace = _scanner.skip_whitespace();
878                 if (!_scanner.skipDoubleColon()) {
879                     // unnecessary: set_current_value(sp);
880                     // this will "loop around" to ACTION_LOAD_SCALAR
881                     // since this is necessarily a symbol of one
882                     // sort of another
883                     temp_state = get_state_after_annotation();
884                     set_state(temp_state);
885                     break;
886                 }
887 
888                 // We have an annotation!
889                 SymbolToken sym = parseSymbolToken("an annotation", sb, t);
890                 append_annotation(sym);
891                 clear_current_value_buffer();
892 
893                 // Consumed the annotation, move on.
894                 // note: that peekDoubleColon() consumed the two colons
895                 // so nextToken won't see them
896                 t = _scanner.nextToken();
897                 switch(t) {
898                 case IonTokenConstsX.TOKEN_SYMBOL_IDENTIFIER:
899                 case IonTokenConstsX.TOKEN_SYMBOL_QUOTED:
900                     // This may be another annotation, so stay in this state
901                     // and come around the horn again to check it out.
902                     break;
903                 default:
904                     // we leave the error handling to the transition
905                     temp_state = get_state_after_annotation();
906                     set_state(temp_state);
907                     break;
908                 }
909                 break;
910             }
911             case ACTION_START_STRUCT:
912                 _value_type = IonType.STRUCT;
913                 temp_state = STATE_BEFORE_FIELD_NAME;
914                 set_state(temp_state);
915                 return;
916             case ACTION_START_LIST:
917                 _value_type = IonType.LIST;
918                 temp_state = STATE_BEFORE_ANNOTATION_CONTAINED;
919                 set_state(temp_state);
920                 return;
921             case ACTION_START_SEXP:
922                 _value_type = IonType.SEXP;
923                 temp_state = STATE_BEFORE_ANNOTATION_SEXP;
924                 set_state(temp_state);
925                 return;
926             case ACTION_START_LOB:
927                 switch (_scanner.peekLobStartPunctuation()) {
928                 case IonTokenConstsX.TOKEN_STRING_DOUBLE_QUOTE:
929                     set_state(STATE_IN_CLOB_DOUBLE_QUOTED_CONTENT);
930                     _lob_token = IonTokenConstsX.TOKEN_STRING_DOUBLE_QUOTE;
931                     _value_type = IonType.CLOB;
932                     break;
933                 case IonTokenConstsX.TOKEN_STRING_TRIPLE_QUOTE:
934                     set_state(STATE_IN_CLOB_TRIPLE_QUOTED_CONTENT);
935                     _lob_token = IonTokenConstsX.TOKEN_STRING_TRIPLE_QUOTE;
936                     _value_type = IonType.CLOB;
937                     break;
938                 default:
939                     set_state(STATE_IN_BLOB_CONTENT);
940                     _lob_token = IonTokenConstsX.TOKEN_OPEN_DOUBLE_BRACE;
941                     _value_type = IonType.BLOB;
942                     break;
943                 }
944                 return;
945             case ACTION_LOAD_SCALAR:
946                 if (t == IonTokenConstsX.TOKEN_SYMBOL_IDENTIFIER) {
947                     sb = token_contents_load(t);
948                     _value_keyword = IonTokenConstsX.keyword(sb, 0, sb.length());
949                     switch (_value_keyword) {
950                     case IonTokenConstsX.KEYWORD_NULL:
951                     {
952                         int kwt = trailing_whitespace ? IonTokenConstsX.KEYWORD_none : _scanner.peekNullTypeSymbol();
953                         switch (kwt) {
954                         case IonTokenConstsX.KEYWORD_NULL:      _null_type = IonType.NULL;       break;
955                         case IonTokenConstsX.KEYWORD_BOOL:      _null_type = IonType.BOOL;       break;
956                         case IonTokenConstsX.KEYWORD_INT:       _null_type = IonType.INT;        break;
957                         case IonTokenConstsX.KEYWORD_FLOAT:     _null_type = IonType.FLOAT;      break;
958                         case IonTokenConstsX.KEYWORD_DECIMAL:   _null_type = IonType.DECIMAL;    break;
959                         case IonTokenConstsX.KEYWORD_TIMESTAMP: _null_type = IonType.TIMESTAMP;  break;
960                         case IonTokenConstsX.KEYWORD_SYMBOL:    _null_type = IonType.SYMBOL;     break;
961                         case IonTokenConstsX.KEYWORD_STRING:    _null_type = IonType.STRING;     break;
962                         case IonTokenConstsX.KEYWORD_BLOB:      _null_type = IonType.BLOB;       break;
963                         case IonTokenConstsX.KEYWORD_CLOB:      _null_type = IonType.CLOB;       break;
964                         case IonTokenConstsX.KEYWORD_LIST:      _null_type = IonType.LIST;       break;
965                         case IonTokenConstsX.KEYWORD_SEXP:      _null_type = IonType.SEXP;       break;
966                         case IonTokenConstsX.KEYWORD_STRUCT:    _null_type = IonType.STRUCT;     break;
967                         case IonTokenConstsX.KEYWORD_none:      _null_type = IonType.NULL;       break; // this happens when there isn't a '.' otherwise peek throws the error or returns none
968                         default: parse_error("invalid keyword id ("+kwt+") encountered while parsing a null");
969                         }
970                         // at this point we've consumed a dot '.' and it's preceding whitespace
971                         // clear_value();
972                         current_value_is_null(_null_type);
973                         // set to null_type in above call: _value_type = IonType.NULL;
974                         break;
975                     }
976                     case IonTokenConstsX.KEYWORD_TRUE:
977                         _value_type = IonType.BOOL;
978                         current_value_is_bool(true);
979                         break;
980                     case IonTokenConstsX.KEYWORD_FALSE:
981                         _value_type = IonType.BOOL;
982                         current_value_is_bool(false);
983                         break;
984                     case IonTokenConstsX.KEYWORD_NAN:
985                         _value_type = IonType.FLOAT;
986                         clear_current_value_buffer();
987                         _v.setValue(Double.NaN);
988                         _v.setAuthoritativeType(AS_TYPE.double_value);
989                         break;
990                     case IonTokenConstsX.KEYWORD_sid:
991                     {
992                         int sid = IonTokenConstsX.decodeSid(sb);
993                         _v.setValue(sid);
994                         _v.setAuthoritativeType(AS_TYPE.int_value);
995                     }
996                     default:
997                         // We don't care about any other 'keywords'
998                         _value_type = IonType.SYMBOL;
999                         break;
1000                     }
1001                 }
1002                 else if (t == IonTokenConstsX.TOKEN_DOT) {
1003                     _value_type = IonType.SYMBOL;
1004                     clear_current_value_buffer();
1005                     _v.setValue(".");
1006                     _v.setAuthoritativeType(AS_TYPE.string_value);
1007                 }
1008                 else {
1009                     // if it's not a symbol we just look at the token type
1010                     _value_type = IonTokenConstsX.ion_type_of_scalar(t);
1011                 }
1012                 int state_after_scalar = get_state_after_value();
1013                 set_state(state_after_scalar);
1014                 return;
1015             case ACTION_PLUS_INF:
1016                 _value_type = IonType.FLOAT;
1017                 clear_current_value_buffer();
1018                 _v.setValue(Double.POSITIVE_INFINITY);
1019                 _v.setAuthoritativeType(AS_TYPE.double_value);
1020                 state_after_scalar = get_state_after_value();
1021                 set_state(state_after_scalar);
1022                 return;
1023             case ACTION_MINUS_INF:
1024                 _value_type = IonType.FLOAT;
1025                 clear_current_value_buffer();
1026                 _v.setValue(Double.NEGATIVE_INFINITY);
1027                 _v.setAuthoritativeType(AS_TYPE.double_value);
1028                 state_after_scalar = get_state_after_value();
1029                 set_state(state_after_scalar);
1030                 return;
1031             case ACTION_EAT_COMMA:
1032                 if (_container_prohibits_commas) {
1033                     parse_error("commas aren't used to separate values in "+getContainerType().toString());
1034                 }
1035                 int new_state = STATE_BEFORE_ANNOTATION_CONTAINED;
1036                 if (_container_is_struct) {
1037                     new_state = STATE_BEFORE_FIELD_NAME;
1038                 }
1039                 set_state(new_state);
1040                 _scanner.tokenIsFinished();
1041                 // when we eat a comma we need to reset the current
1042                 // value start used to define a span, since the comma
1043                 // isn't part of the span when it's hoisted
1044                 _value_start_offset = _scanner.getStartingOffset();
1045                 t = _scanner.nextToken();
1046                 break;
1047             case ACTION_FINISH_CONTAINER:
1048                 new_state = get_state_after_container(t);
1049                 set_state(new_state);
1050                 _eof = true;
1051                 return;
1052             case ACTION_FINISH_LOB:
1053                 state_after_scalar = get_state_after_value();
1054                 set_state(state_after_scalar);
1055                 return;
1056             case ACTION_FINISH_DATAGRAM:
1057                 if (getDepth() != 0) {
1058                     parse_error("state failure end of datagram encounterd with a non-container stack");
1059                 }
1060                 set_state(STATE_EOF);
1061                 _eof = true;
1062                 return;
1063             default: parse_error("unexpected token encountered: "+IonTokenConstsX.getTokenName(t));
1064             }
1065         }
1066     }

1068     protected final StringBuilder token_contents_load(int token_type) throws IOException
1069     {
1070         StringBuilder sb = _current_value_buffer;
1071         boolean       clob_chars_only;
1072         int           c;
1073 
1074         if (_current_value_buffer_loaded) {
1075             return sb;
1076         }
1077         else if (_current_value_save_point_loaded) {
1078             assert(!_scanner.isUnfinishedToken() && !_current_value_save_point.isClear());
1079             // _scanner.load_save_point_contents( _current_value_save_point, sb);
1080 
1081             _scanner.save_point_activate(_current_value_save_point);
1082             switch (token_type) {
1083             default:
1084                 _scanner.load_raw_characters(sb);
1085                 break;
1086             case IonTokenConstsX.TOKEN_SYMBOL_IDENTIFIER:
1087                 _scanner.load_symbol_identifier(sb);
1088                 _value_type = IonType.SYMBOL;
1089                 break;
1090             case IonTokenConstsX.TOKEN_SYMBOL_OPERATOR:
1091                 _scanner.load_symbol_operator(sb);
1092                 _value_type = IonType.SYMBOL;
1093                 break;
1094             case IonTokenConstsX.TOKEN_SYMBOL_QUOTED:
1095                 clob_chars_only = (IonType.CLOB == _value_type);
1096                 _scanner.load_single_quoted_string(sb, clob_chars_only);
1097                 _value_type = IonType.SYMBOL;
1098                 break;
1099             case IonTokenConstsX.TOKEN_STRING_DOUBLE_QUOTE:
1100                 clob_chars_only = (IonType.CLOB == _value_type);
1101                 _scanner.load_double_quoted_string(sb, clob_chars_only);
1102                 _value_type = IonType.STRING;
1103                 break;
1104             case IonTokenConstsX.TOKEN_STRING_TRIPLE_QUOTE:
1105                 clob_chars_only = (IonType.CLOB == _value_type);
1106                 _scanner.load_triple_quoted_string(sb, clob_chars_only);
1107                 _value_type = IonType.STRING;
1108                 break;
1109             }
1110             _scanner.save_point_deactivate(_current_value_save_point);
1111             _current_value_buffer_loaded = true;
1112         }
1113         else {
1114             _scanner.save_point_start(_current_value_save_point);
1115             switch (token_type) {
1116             case IonTokenConstsX.TOKEN_UNKNOWN_NUMERIC:
1117             case IonTokenConstsX.TOKEN_INT:
1118             case IonTokenConstsX.TOKEN_BINARY:
1119             case IonTokenConstsX.TOKEN_HEX:
1120             case IonTokenConstsX.TOKEN_FLOAT:
1121             case IonTokenConstsX.TOKEN_DECIMAL:
1122             case IonTokenConstsX.TOKEN_TIMESTAMP:
1123                 _value_type = _scanner.load_number(sb);
1124                 break;
1125             case IonTokenConstsX.TOKEN_SYMBOL_IDENTIFIER:
1126                 _scanner.load_symbol_identifier(sb);
1127                 _value_type = IonType.SYMBOL;
1128                 break;
1129             case IonTokenConstsX.TOKEN_SYMBOL_OPERATOR:
1130                 _scanner.load_symbol_operator(sb);
1131                 _value_type = IonType.SYMBOL;
1132                 break;
1133             case IonTokenConstsX.TOKEN_SYMBOL_QUOTED:
1134                 clob_chars_only = (IonType.CLOB == _value_type);
1135                 c = _scanner.load_single_quoted_string(sb, clob_chars_only);

1137                     //String message = "EOF encountered before closing single quote";
1138                     //parse_error(message);
1139                     _scanner.unexpected_eof();
1140                 }
1141                 _value_type = IonType.SYMBOL;
1142                 break;
1143             case IonTokenConstsX.TOKEN_STRING_DOUBLE_QUOTE:
1144                 clob_chars_only = (IonType.CLOB == _value_type);
1145                 c = _scanner.load_double_quoted_string(sb, clob_chars_only);
1146                 if (c == UnifiedInputStreamX.EOF) {
1147                     // String message = "EOF encountered before closing single quote";
1148                     // parse_error(message);
1149                     _scanner.unexpected_eof();
1150                 }
1151                 _value_type = IonType.STRING;
1152                 break;
1153             case IonTokenConstsX.TOKEN_STRING_TRIPLE_QUOTE:
1154                 clob_chars_only = (IonType.CLOB == _value_type);
1155                 c = _scanner.load_triple_quoted_string(sb, clob_chars_only);
1156                 if (c == UnifiedInputStreamX.EOF) {
1157                     //String message = "EOF encountered before closing single quote";
1158                     //parse_error(message);
1159                     _scanner.unexpected_eof();
1160                 }
1161                 _value_type = IonType.STRING;
1162                 break;
1163             default:
1164                 String message = "unexpected token "
1165                                + IonTokenConstsX.getTokenName(token_type)
1166                                + " encountered";
1167                 throw new IonException(message);
1168             }
1169             _current_value_save_point.markEnd();
1170             _current_value_save_point_loaded = true;
1171             _current_value_buffer_loaded = true;
1172             tokenValueIsFinished();
1173         }

1324     public void stepIn()
1325     {
1326         if (_value_type == null || _eof) {
1327             throw new IllegalStateException();
1328         }
1329         switch (_value_type) {
1330         case STRUCT:
1331         case LIST:
1332         case SEXP:
1333             break;
1334         default:
1335             throw new IllegalStateException("Unexpected value type: " + _value_type);
1336         }
1337 
1338         int new_state = get_state_at_container_start(_value_type);
1339         set_state(new_state);
1340 
1341         push_container_state(_value_type);
1342 
1343         _scanner.tokenIsFinished();
1344         try {
1345             finish_value(null);
1346         }
1347         catch (IOException e) {
1348             throw new IonException(e);
1349         }
1350         if (_v.isNull()) {
1351             _eof = true;
1352             _has_next_called = true;  // there are no contents in a null container
1353         }
1354 
1355         _value_type = null;
1356 
1357         if (_debug) System.out.println("stepInto() new depth: "+getDepth());
1358     }
1359     public void stepOut()
1360     {
1361         if (getDepth() < 1) {
1362             throw new IllegalStateException(IonMessages.CANNOT_STEP_OUT);
1363         }
1364         try {
1365             finish_value(null);
1366             switch (getContainerType()) {
1367             case STRUCT:
1368                 if (!_eof) _scanner.skip_over_struct();
1369                 break;
1370             case LIST:
1371                 if (!_eof) _scanner.skip_over_list();
1372                 break;
1373             case SEXP:
1374                 if (!_eof) _scanner.skip_over_sexp();
1375                 break;
1376             case DATAGRAM:
1377                 break;
1378             default:
1379                 throw new IllegalStateException("Unexpected value type: " + _value_type);
1380             }
1381         }
1382         catch (IOException e) {
1383             throw new IonException(e);
1384         }
1385         pop_container_state();
1386         _scanner.tokenIsFinished();
1387         try {
1388             finish_value(null);
1389         }
1390         catch (IOException e) {
1391             throw new IonException(e);
1392         }
1393 
1394         clear_value();
1395 
1396         if (_debug) System.out.println("stepOUT() new depth: "+getDepth());
1397     }
