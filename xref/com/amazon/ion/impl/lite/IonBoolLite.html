16  package com.amazon.ion.impl.lite;
17  
18  import com.amazon.ion.IonBool;
19  import com.amazon.ion.IonType;
20  import com.amazon.ion.IonWriter;
21  import com.amazon.ion.NullValueException;
22  import com.amazon.ion.ValueVisitor;
23  import java.io.IOException;
24  
25  
26  final class IonBoolLite
27      extends IonValueLite
28      implements IonBool
29  {
30      private static final int HASH_SIGNATURE =
31          IonType.BOOL.toString().hashCode();
32  
33      /**
34       * Optimizes out a function call for a const result
35       */
36      protected static final int TRUE_HASH
37              = HASH_SIGNATURE ^ (16777619 * Boolean.TRUE.hashCode());
38  
39      /**
40       * Optimizes out a function call for a const result
41       */
42      protected static final int FALSE_HASH
43              = HASH_SIGNATURE ^ (16777619 * Boolean.FALSE.hashCode());
44  
45      /**
46       * Constructs a null bool value.
47       */
48      IonBoolLite(ContainerlessContext context, boolean isNull)
49      {
50          super(context, isNull);
51      }
52  
53      IonBoolLiteref="../../../../../com/amazon/ion/impl/lite/IonBoolLite.html#IonBoolLite">IonBoolLite(IonBoolLite existing, IonContext context)
54      {
55          super(existing, context);
56      }
57  
58      @Override
59      IonBoolLite clone(IonContext context)
60      {
61          return new IonBoolLite(this, context);
62      }
63  
64      @Override
65      public IonBoolLite clone()
66      {
67          return clone(ContainerlessContext.wrap(getSystem()));
68      }
69  
70      @Override
71      public IonType getType()
72      {
73          return IonType.BOOL;
74      }
75  
76      @Override
77      int hashCode(SymbolTableProvider symbolTableProvider)
78      {
79          int result = HASH_SIGNATURE;
80  
81          if (!isNullValue())
82          {
83              result = booleanValue() ? TRUE_HASH : FALSE_HASH;
84          }
85  
86          return hashTypeAnnotations(result, symbolTableProvider);
87      }
88  
89      public boolean booleanValue()
90          throws NullValueException
91      {
92          validateThisNotNull();
93          return this._isBoolTrue();
94      }
95  
96      public void setValue(boolean b)
97      {
98          // the called setValue will check if this is locked
99          setValue(Boolean.valueOf(b));
100     }
101 
102     public void setValue(Boolean b)
103     {
104         checkForLock();
105         if (b == null) {
106             _isBoolTrue(false);
107             _isNullValue(true);
108         }
109         else {
110             _isBoolTrue(b.booleanValue());
111             _isNullValue(false);
112         }
113     }
114 
115     @Override
116     final void writeBodyTo(IonWriter writer, SymbolTableProvider symbolTableProvider)
117         throws IOException
118     {
119         if (isNullValue())
120         {
121             writer.writeNull(IonType.BOOL);
122         }
123         else
124         {
125             writer.writeBool(_isBoolTrue());
126         }
127     }
128 
129     @Override
130     public void accept(ValueVisitor visitor)
131         throws Exception
132     {
133         visitor.visit(this);
134     }
135 }
