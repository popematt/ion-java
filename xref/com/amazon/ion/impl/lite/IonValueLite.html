91      /**
92       * May be used by subclasses to reuse _flag bits for purposes specific
93       * to that subclass. It is important that only flag bits not currently
94       * used by that subclass are chosen; otherwise important data may be
95       * overwritten. NOTE: only the lower 8 bits may be used, because the
96       * upper 24 are reserved for the element ID.
97       * @param metadata the metadata to set.
98       * @param mask the location at which to set the metadata. Must be within
99       *             the lower 8 bits.
100      * @param shift the number of bits to left-shift the metadata so that
101      *              it starts at the index of the mask's LSB.
102      */
103     protected final void _setMetadata(int metadata, int mask, int shift) {
104         assert(mask <= ELEMENT_MASK); // don't overwrite the element ID
105         _flags &= ~mask;
106         _flags |= ((metadata << shift) & mask);

789     protected int hashTypeAnnotations(final int original, SymbolTableProvider symbolTableProvider)
790     {
791         final SymbolToken[] tokens = getTypeAnnotationSymbols(symbolTableProvider);
792         if (tokens.length == 0)
793         {
794             return original;
795         }
796 
797         final int sidHashSalt   = 127;      // prime to salt sid of annotation
798         final int textHashSalt  = 31;       // prime to salt text of annotation
799         final int prime = 8191;
800         int result = original ^ TYPE_ANNOTATION_HASH_SIGNATURE;
801 
802         result = prime * original + tokens.length;
803 
804         for (final SymbolToken token : tokens)
805         {
806             String text = token.getText();
807 
808             int tokenHashCode = text == null
809                 ? token.getSid()  * sidHashSalt
810                 : text.hashCode() * textHashSalt;
811 
812             // mixing to account for small text and sid deltas
813             tokenHashCode ^= (tokenHashCode << 19) ^ (tokenHashCode >> 13);
814 
815             result = prime * result + tokenHashCode;
816 
817             // mixing at each step to make the hash code order-dependent
818             result ^= (result << 25) ^ (result >> 7);
819         }
820 
821         return result;
822     }
