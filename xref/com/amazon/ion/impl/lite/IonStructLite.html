89      protected void build_field_map()
90      {
91          int size = (_children == null) ? 0 : _children.length;
92  
93          _field_map = new HashMap<String, Integer>(size);
94          _field_map_duplicate_count = 0;
95  
96          int count = get_child_count();
97          for (int ii=0; ii<count; ii++) {
98              IonValueLite v = get_child(ii);
99              SymbolToken fieldNameSymbol = v.getFieldNameSymbol();
100             String name = fieldNameSymbol.getText();
101             if (_field_map.get(name) != null) {
102                 _field_map_duplicate_count++;
103             }
104             _field_map.put(name, ii); // this causes the map to have the largest index value stored
105         }
106         return;
107     }
108     private void add_field(String fieldName, int newFieldIdx)
109     {
110         Integer idx = _field_map.get(fieldName);
111         if (idx != null) {
112             _field_map_duplicate_count++;
113             if (idx.intValue() > newFieldIdx) {
114                 newFieldIdx = idx.intValue();
115             }
116         }
117         _field_map.put(fieldName, newFieldIdx);
118     }
119     private void remove_field(String fieldName, int lowest_idx, int copies)
120     {
121         if (_field_map == null) {
122             return;
123         }
124 
125         Integer field_idx = _field_map.get(fieldName);
126         assert(field_idx != null);
127         _field_map.remove(fieldName);
128         _field_map_duplicate_count -= (copies - 1);

131     private void remove_field_from_field_map(String fieldName, int idx)
132     {
133         Integer field_idx = _field_map.get(fieldName);
134         assert(field_idx != null);
135 
136         if (field_idx.intValue() != idx) {
137             // if the map has a different index, this must
138             // be a duplicate, and this copy isn't in the map
139             assert(_field_map_duplicate_count > 0);
140             _field_map_duplicate_count--;
141         }
142         else if (_field_map_duplicate_count > 0) {
143             // if we have any duplicates we have to check
144             // every time since we don't track which field
145             // is duplicated - so any dup can be expensive
146             int ii = find_last_duplicate(fieldName, idx);
147 
148             if (ii == -1) {
149                 // this is the last copy of this key
150                 _field_map.remove(fieldName);
151             }
152             else {
153                 // replaces this fields (the one being
154                 // removed) array idx in the map with
155                 // the preceding duplicates index
156                 _field_map.put(fieldName, ii);
157                 _field_map_duplicate_count--;
158             }
159         }
160         else {
161             // since there are not dup's we can just update
162             // the map by removing this fieldname
163             _field_map.remove(fieldName);
164         }
165     }

167     private void patch_map_elements_helper(int removed_idx)
168     {
169         if (_field_map == null) {
170             return;
171         }
172 
173         if (removed_idx >= get_child_count()) {
174             // if this was the at the end of the list
175             // there's nothing to change
176             return;
177         }
178 
179         for (int ii=removed_idx; ii<get_child_count(); ii++) {
180             IonValueLite value = get_child(ii);
181             String  field_name = value.getFieldName();
182             Integer map_idx = _field_map.get(field_name);
183             if (map_idx.intValue() != ii) {
184                 // if this is a field that to the right of
185                 // the removed (in process of removing) value
186                 // we need to patch the index value
187                 _field_map.put(field_name, ii);
188             }
189         }
190     }

236     private int find_last_duplicate(String fieldName, int existing_idx)
237     {
238         for (int ii=existing_idx; ii>0; ) {
239             ii--;
240             IonValueLite field = get_child(ii);
241             if (fieldName.equals(field.getFieldName())) {
242                 return ii;
243             }
244         }
245         assert(there_is_only_one(fieldName, existing_idx));
246         return -1;

285     /**
286      * Implements {@link Object#hashCode()} consistent with equals.
287      * This is insensitive to order of fields.
288      * <p>
289      * This method must follow the contract of {@link Object#equals(Object)},
290      * which is located at {@link Equivalence#ionEquals(IonValue, IonValue)}.
291      *
292      * @return  An int, consistent with the contracts for
293      *          {@link Object#hashCode()} and {@link Object#equals(Object)}.
294      */
295     @Override
296     int hashCode(SymbolTableProvider symbolTableProvider)
297     {
298         final int nameHashSalt  = 16777619; // prime to salt name of each Field
299         final int valueHashSalt = 8191;     // prime to salt value of each Field
300         final int sidHashSalt   = 127;      // prime to salt sid of fieldname
301         final int textHashSalt  = 31;       // prime to salt text of fieldname
302 
303         int result = HASH_SIGNATURE;
304 
305         if (!isNullValue())  {
306             for (IonValue v : this)  {
307                 IonValueLite/../../../com/amazon/ion/impl/lite/IonValueLite.html#IonValueLite">IonValueLite vlite = (IonValueLite) v;
308                 // If fieldname's text is unknown, use its sid instead
309                 SymbolToken token = vlite.getFieldNameSymbol(symbolTableProvider);
310                 String text = token.getText();
311 
312                 int nameHashCode = text == null
313                     ? token.getSid()  * sidHashSalt
314                     : text.hashCode() * textHashSalt;
315 
316                 // mixing to account for small text and sid deltas
317                 nameHashCode ^= (nameHashCode << 17) ^ (nameHashCode >> 15);
318 
319                 int fieldHashCode = HASH_SIGNATURE;
320                 fieldHashCode = valueHashSalt * fieldHashCode + vlite.hashCode(symbolTableProvider);
321                 fieldHashCode = nameHashSalt  * fieldHashCode + nameHashCode;
322 
323                 // another mix step for each Field of the struct
324                 fieldHashCode ^= (fieldHashCode << 19) ^ (fieldHashCode >> 13);
325 
326                 // Additive hash is used to ensure insensitivity to order of
327                 // fields, and will not lose data on value hash codes
328                 result += fieldHashCode;
329             }
330         }
331 
332         return hashTypeAnnotations(result, symbolTableProvider);
333     }

345     private IonStruct doClone(boolean keep, String... fieldNames)
346     {
347         IonStruct clone;
348         if (isNullValue())
349         {
350             clone = getSystem().newNullStruct();
351         }
352         else
353         {
354             Set<String> fields =
355                 new HashSet<String>(Arrays.asList(fieldNames));
356             if (keep && fields.contains(null))
357             {
358                 throw new NullPointerException("Can't retain an unknown field name");
359             }
360 
361             clone = getSystem().newEmptyStruct();
362             for (IonValue value : this)
363             {
364                 SymbolToken fieldNameSymbol = value.getFieldNameSymbol();
365                 String fieldName = fieldNameSymbol.getText();
366                 if (fields.contains(fieldName) == keep)
367                 {
368                     // This ensures that we don't copy an unknown field name.
369                     fieldName = value.getFieldName();
370                     clone.add(fieldName, value.clone());
371                 }
372             }
373         }
374 
375         clone.setTypeAnnotationSymbols(getTypeAnnotationSymbols());
376 
377         return clone;
378     }

400     public IonValue get(String fieldName)
401     {
402         int field_idx = find_field_helper(fieldName);
403         IonValue field;
404 
405         if (field_idx < 0) {
406             if(hasNullFieldName) throw new UnknownSymbolException("Unable to determine whether the field exists because the struct contains field names with unknown text.");
407             field = null;
408         } else {
409             field = get_child(field_idx);
410         }
411 
412         return field;

414     private int find_field_helper(String fieldName)
415     {
416         validateFieldName(fieldName);
417 
418         if (isNullValue()) {
419             // nothing to see here, move along
420         }
421         else if (_field_map != null) {
422             Integer idx = _field_map.get(fieldName);
423             if (idx != null) {
424                 return idx.intValue();
425             }
426         }
427         else {
428             int ii, size = get_child_count();
429             for (ii=0; ii<size; ii++) {
430                 IonValue field = get_child(ii);
431                 if (fieldName.equals(field.getFieldName())) {
432                     return ii;
433                 }
434             }
435         }
436         return -1;
437     }

553     /**
554      * put is "make this value the one and only value
555      * associated with this fieldName".  The side effect
556      * is that if there were multiple fields with this
557      * name when put is complete there will only be the
558      * one value in the collection.
559      */
560     public void put(String fieldName, IonValue value)
561     {
562         checkForLock();
563 
564         validateFieldName(fieldName);
565         if (value != null) validateNewChild(value);
566 
567         int lowestRemovedIndex = get_child_count();
568         boolean any_removed = false;
569 
570         // first we remove the any existing fields
571         // associated with fieldName (which may be none)
572         if (_field_map != null && _field_map_duplicate_count == 0)
573         {
574             // we have a map and no duplicates so the index
575             // (aka map) is all we need to find the only
576             // value associated with fieldName, if there is one
577             Integer idx = _field_map.get(fieldName);
578             if (idx != null) {
579                 lowestRemovedIndex = idx.intValue();
580                 remove_field_from_field_map(fieldName, lowestRemovedIndex);
581                 remove_child(lowestRemovedIndex);
582                 any_removed = true;
583             }
584         }
585         else {
586             // either we don't have a map (index) or there
587             // are duplicates in both cases we have to
588             // scan the child list directly.
589             // Walk backwards to minimize array movement
590             // as we remove fields as we encounter them.
591             int copies_removed = 0;
592             for (int ii = get_child_count(); ii > 0; )
593             {
594                 ii--;
595                 IonValueLite child = get_child(ii);
596                 if (fieldName.equals(child.getFieldNameSymbol().getText()))
597                 {
598                     // done by remove_child: child.detachFromContainer();
599                     remove_child(ii);
600                     lowestRemovedIndex = ii;
601                     copies_removed++;
602                     any_removed = true;
603                 }
604             }
605             if (any_removed) {
606                 remove_field(fieldName, lowestRemovedIndex, copies_removed);
607             }
608         }
609         if (any_removed) {
610             patch_map_elements_helper(lowestRemovedIndex);
611             patch_elements_helper(lowestRemovedIndex);
612         }
613 
614         // once we've removed any existing copy we now add,
615         // this (delete + add == put) turns out be be the
616         // right choice since:
617         //   1 - because of possible duplicates we can't
618         //       guarantee the idx is stable
619         //   2 - we have to maintain the hash and that
620         //       really means we end up with the delete
621         //       anyway
622         // strictly speaking this approach, while simpler,
623         // is more expensive when we don't have a has and
624         // the value already exists, and it's not at the
625         // end of the field list anyway.
626         if (value != null) {
627             add(fieldName, value);
628         }
629     }

763         return removedAny;
764     }

766     /**
767      *
768      * @param field must not be null.  It is not required to have a field name.
769      * @param fields must not be null, and must not contain and nulls.
770      * @return true if {@code field.getFieldName()} is in {@code fields}.
771      */
772     private static boolean isListedField(IonValue field, String[] fields)
773     {
774         String fieldName = field.getFieldName();
775         for (String key : fields)
776         {
777             if (key.equals(fieldName)) return true;
778         }
779         return false;
780     }

785      * methods that have that precondition.
786      *
787      * @throws NullPointerException if the <code>fieldName</code>
788      * is <code>null</code>.

792         if (fieldName == null)
793         {
794             throw new NullPointerException("fieldName is null");
795         }

798     @Override
799     final void writeBodyTo(IonWriter writer, SymbolTableProvider symbolTableProvider)
800         throws IOException
801     {
802         if (isNullValue())
803         {
804             writer.writeNull(IonType.STRUCT);
805         }
806         else
807         {
808             writer.stepIn(IonType.STRUCT);
809             writeChildren(writer, this, symbolTableProvider);
810             writer.stepOut();
811         }
812     }
