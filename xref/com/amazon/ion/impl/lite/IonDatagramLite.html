44  /**
45   *  The datagram generally behaves as an IonSexp. A list with space
46   *  separated values.
47   *
48   *  As a user datagram it only shows user values as current members
49   *  of the list.
50   *
51   *  As a system datagram it also includes system values (adding Ion
52   *  version marker symbols and local symbol tables to the list).
53   *
54   *  Most API's operate on the user view.  The exceptions are getBytes
55   *  (and related), and the system* API's.  These synthesize a system
56   *  view over the user values.
57   *
58   *  When system values are added they translate into setsymboltableat
59   *  operations.  Which sets a pending symbol table at the current
60   *  user position (as specified by the add pos).  When the next add
61   *  occurs if it is at the specified position the pending symbol
62   *  table is applied to the value if the value doesn't already have
63   *  a local symbol table defined.
64   *
65   *  in any event any membership update invalidates the pending symbol
66   *  table.
67   *
68   *  In general on add if there is no pending symbol table the preceding
69   *  value local symbol table is applied as the local symbol table to
70   *  the new value, if it needs one.
71   *
72   *  The system iterator inserts system values to create the minimum
73   *  additional values to represent a correct Ion sequence by injecting
74   *  IVM's when the symbol table transitions to system, and local
75   *  symbol tables when they first occur.
76   *
77   */

303     @Override
304     public int hashCode() {
305         int prime  = 8191;
306         int result = HASH_SIGNATURE;
307 
308         if (!isNullValue()) {
309             // As we are a datagram then the children need to resolve their own symbol tables -
310             // so we use the 'top level' #hashCode() which will force each child to resolve it's
311             // own symbol table.
312             for (IonValue v : this) {
313                 result = prime * result + v.hashCode();
314                 // mixing at each step to make the hash code order-dependent
315                 result ^= (result << 29) ^ (result >> 3);
316             }
317         }
318         return result;
319     }

353     @Override
354     public IonContext getContextForIndex(IonValue element, int index)
355     {
356         if (index == this._pending_symbol_table_idx) {
357             SymbolTable symbols = _pending_symbol_table;
358             _pending_symbol_table = null;
359             _pending_symbol_table_idx = -1;
360             return TopLevelContext.wrap(symbols, this);
361         }
362         // the preceding elements symbol table is our next
363         IonValueLite preceding = (index > 0) ? get_child(index - 1) : null;
364         if (preceding != null && preceding._context != this) {
365             return preceding._context;
366         }
367         // otherwise element will just default to the system
368         // symbol table
369         return TopLevelContext.wrap(null, this);

405     /*
406      * NOTE: IonDatagramLite overrides the main writeTo mechanism prescribed in:
407      * IonValueLite#writeTo which works by eagerly resolving the SymbolTable as the IonDatagram does
408      * not have a symbol table (throws UnsupportedOperationException) and so needs to write each
409      * child out independently where the child will act as it's own top-level value (inc symbol table)
410      */
411     @Override
412     public final void writeTo(IonWriter writer)
413     {
414         if (writer.getSymbolTable().isSystemTable()) {
415             // If the writer was configured with a non-default symbol table, writing an IVM here will
416             // reset that symbol table. If the datagram contains any symbols with unknown text that
417             // refer to slots in shared symbol table imports declared by the discarded table, an
418             // error will be raised unnecessarily. To avoid that, only write an IVM when the writer's
419             // symbol table is already the system symbol table.
420             // TODO evaluate whether an IVM should ever be written here. amzn/ion-java#200
421             try {
422                 writer.writeSymbol(SystemSymbols.ION_1_0);
423             } catch (IOException ioe) {
424                 throw new IonException(ioe);
425             }
426         }
427         for (IonValue iv : this) {
428             iv.writeTo(writer);
429         }
430     }

565     /**
566      * Encapsulates an iterator and implements a custom remove method
567      *
568      *  this is tied to the _child array of the IonSequenceImpl
569      *  through the _children and _child_count members which this
570      *  iterator directly uses.
571      *
572      *  This is a specialization for returning a system view of the
573      *  current datagram.  It does this by synthesizing system values
574      *  (Ion Version Markers & local symbol tables) as symbol tables
575      *  change between user values.
576      *
577      *  It accumulates a count of the system values it encounters
578      *  the need to inject them in the iteration stream.
579      *
580      *  It may need more than 1 system value between user values
581      *  if the system is reset or the local symbol table itself
582      *  has local symbols (this is rare but in principal could
583      *  be an arbitrarily long sequence).
584      *
585      *  The current position is between the value to be returned
586      *  by next and the value that would be returned by previous.
587      *  This is calculated in next_index_helper() (or previous_).
588      *  It is represented by a "struct" with a position in the
589      *  child array, an optional position in the local stack
590      *  of system values, and a flag indicating which list should
591      *  be used to fetch the actual value.
592      *
593      *  The local synthetic system values are held in the array
594      *  __local_system_value.  These are the values that should
595      *  preceed the value at _child[__user_content_pos].
596      *
597      *  TODO with the updated next and previous logic, particularly
598      *  the force_position_sync logic and lastMoveWasPrevious flag
599      *  we could implement add and set correctly.
600      *
601      *  NOTE this closely resembles the iterator defined in IonSequenceLite,
602      *  so changes here are likely to be needed in IonSequenceLite as well.
603      */
604     protected final class SystemContentIterator
605         implements ListIterator<IonValue>
606     {
607         private final boolean                __readOnly;
608         private       IonValueLite           __current;
609         private       SystemIteratorPosition __pos;
610 
611         private       SystemIteratorPosition __temp_pos;
612 
613         public SystemContentIterator(boolean readOnly)
614         {
615             if (_isLocked() && !readOnly) {
616                 throw new IllegalStateException("you can't open an updatable iterator on a read only value");
617             }
618             __readOnly = readOnly;
619 
620             __temp_pos = new SystemIteratorPosition(this); // we flip back and forth between two positions to avoid allocating these on every value (or more)
621             __pos = new SystemIteratorPosition(this);
622             __pos.load_initial_position();
623         }
624 
625         private IonSystem getSystem()
626         {
627             return IonDatagramLite.this.getSystem();
628         }
629 
630         protected IonValueLite set_position(SystemIteratorPosition newPos)
631         {
632             // swap our active position with our temp position
633             __temp_pos = __pos;
634             __pos = newPos;
635 
636             // load out current value from the position, and return it
637             __current = __pos.load_position();
638             return __current;
639         }
640 
641         private void force_position_sync()
642         {
643             int user_index = __pos.__user_index;
644             if (user_index < 0 || user_index >= _child_count) {
645                 return;
646             }
647             IonValueLite user_value = __pos.__current_user_value;
648             if (user_value == null || user_value == _children[user_index]) {
649                 return;
650             }
651             if (__readOnly) {
652                 throw new IonException("read only sequence was changed");
653             }
654             __pos.force_position_sync_helper();
655         }

678         public final int nextIndex()
679         {
680             SystemIteratorPosition pos = next_index_helper();
681             if (pos == null) {
682                 // we can do this because we hold the __pos
683                 // even when we run off the end, it will be

686             }

688             return idx;

691         private final SystemIteratorPosition next_index_helper()
692         {
693             SystemIteratorPosition next = null;
694             force_position_sync();
695 
696             if (__pos.has_next() == false) {
697                 return null;
698             }
699 
700             // at this point we will have a next so prep out position now
701             next = __temp_pos;
702             assert(next != null && next != __pos);
703             next.copyFrom(__pos);
704 
705             // see if there's a system value waiting for use
706             next.__local_index++;
707             if (next.__local_index < next.__local_value_count) {
708                 return next;
709             }
710 
711             // if there's not system value there must be another user value
712             // so we shouldn't get here since has_next() should have failed
713             assert(next.__user_index <= get_child_count());
714 
715             // if we were on a system value then we're just stepping onto the
716             // since has_next() already declared we do have a waiting value
717             next.__user_index++;
718             next.load_updated_position();
719             // we step onto the first local value
720             next.__local_index = 0;
721             return next;
722         }

750         private final SystemIteratorPosition previous_index_helper()
751         {
752             SystemIteratorPosition prev = null;
753 
754             force_position_sync();
755             if (__pos.has_prev() == false) {
756                 return null;
757             }
758 
759             // at this point we will have a prev so prep out position now
760             prev = __temp_pos;
761             assert(prev != null && prev != __pos);
762             prev.copyFrom(__pos);
763 
764             prev.__local_index--;
765             if (prev.__local_index >= 0) {
766                 return prev;
767             }
768 
769             // if there's not system value there must be another user value
770             // we would have bailed with has_prev returned false above otherwise
771             assert(prev.__user_index > 0);
772 
773             // if this is the 2nd prev then we really have to back up
774             prev.__index_adjustment -= prev.__local_value_count;
775             prev.__user_index--;
