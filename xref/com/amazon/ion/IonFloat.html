16  package com.amazon.ion;
17  import java.math.BigDecimal;
18  
19  
20  /**
21   * An Ion <code>float</code> value.
22   * <p>
23   * <b>WARNING:</b> This interface should not be implemented or extended by
24   * code outside of this library.
25   *
26   * <h2>Precision Problems</h2>
27   *
28   * Use of binary floating-point numbers is prone to countless problems.
29   * The vast majority of applications should use {@code decimal} values instead.
30   * Please read the
31   * <a href="http://speleotrove.com/decimal/decifaq1.html">Decimal Arithmetic
32   * FAQ</a> for horror stories.
33   * <p>
34   * If you have any doubt whatsoever on whether you should use {@code float} or
35   * {@code decimal}, then you should use {@code decimal}.
36   *
37   * @see IonDecimal
38   */
39  public interface IonFloat
40      extends IonNumber
41  {
42      /**
43       * Gets the value of this Ion <code>float</code> as a Java
44       * <code>float</code> value.
45       *
46       * @return the float value.
47       * @throws NullValueException if <code>this.isNullValue()</code>.
48       */
49      public float floatValue()
50          throws NullValueException;
51  
52      /**
53       * Gets the value of this Ion <code>float</code> as a Java
54       * <code>double</code> value.
55       *
56       * @return the double value.
57       * @throws NullValueException if <code>this.isNullValue()</code>.
58       */
59      public double doubleValue()
60          throws NullValueException;
61  
62  
63      /**
64       * Gets the value of this Ion {@code float} as a Java {@link BigDecimal}.
65       * This follows the behavior of {@link BigDecimal#valueOf(double)}. It's
66       * recommended to call {@link IonFloat#isNumericValue()} before calling
67       * this method.
68       *
69       * If you need negative zeros, use {@link #doubleValue()}.
70       *
71       * @return the {@link BigDecimal} value, or {@code null} if
72       * {@code this.isNullValue()}.
73       *
74       * @throws NumberFormatException if this value is {@code nan},
75       * {@code +inf}, or {@code -inf}, because {@link BigDecimal} cannot
76       * represent those values.
77       */
78      @Override
79      public BigDecimal bigDecimalValue()
80          throws NullValueException;
81  
82      /**
83       * Sets the value of this element.
84       */
85      public void setValue(float value);
86  
87      /**
88       * Sets the value of this element.
89       */
90      public void setValue(double value);
91  
92      /**
93       * Sets the value of this element.
94       * Since Ion {@code float}s are essentially Java {@code double}s,
95       * this performs a narrowing conversion as described by
96       * {@link BigDecimal#doubleValue()}.
97       *
98       * @param value the new value of this float;
99       * may be <code>null</code> to make this <code>null.float</code>.
100      */
101     public void setValue(BigDecimal value);
102 
103     /**
104      * Determines whether this value is numeric. Returns true if this value
105      * is none of {@code null}, {@code nan}, {@code +inf}, and {@code -inf},
106      * and false if it is any of them.
107      *
108      * @return a checked condition whether this value is numeric.
109      */
110     public boolean isNumericValue();
111 
112     public IonFloat clone()
113         throws UnknownSymbolException;
114 }
