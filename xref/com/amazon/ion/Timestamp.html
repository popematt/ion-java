30  /**
31   * An immutable representation of a point in time. Ion defines a simple
32   * representation of time based on Coordinated Universal Time (UTC).
33   * In practice the use of time could be more accurately described as
34   * UTC-SLS (UTC Smoothed Leap Seconds) as there is no representation for the
35   * leap second discontinuities that UTC has added.
36   * <p>
37   * Timestamps preserve precision, meaning the fields that are included, and the
38   * significant digits of any fractional second.  Only common break
39   * points in the values are supported.  Any unspecified fields are handled
40   * as the start of the new year/month/day.
41   *
42   *
43   * <h3>Equality and Comparison</h3>
44   *
45   * As with {@link IonValue} classes, the {@link #equals equals} methods on this class
46   * perform a strict equivalence that observes the precision and local offset
47   * of each timestamp.
48   * This means that it's possible to have two {@link Timestamp} instances that
49   * represent the same point in time but are not {@code equals}.
50   * <p>
51   * On the other hand, the {@link #compareTo} methods perform point in time
52   * comparison, ignoring precision and local offset.
53   * Thus the <em>natural comparison method</em> of this class is <em>not
54   * consistent with equals</em>. See the documentation of {@link Comparable} for
55   * further discussion.
56   * <p>
57   * To illustrate this distinction, consider the following timestamps. None are
58   * {@link #equals} to each other, but any pair will return a zero result from
59   * {@link #compareTo}.
60   * <ul>
61   *   <li>{@code 2009T}</li>
62   *   <li>{@code 2009-01T}</li>
63   *   <li>{@code 2009-01-01T}</li>
64   *   <li>{@code 2009-01-01T00:00Z}</li>
65   *   <li>{@code 2009-01-01T00:00:00Z}</li>
66   *   <li>{@code 2009-01-01T00:00:00.0Z}</li>
67   *   <li>{@code 2009-01-01T00:00:00.00Z}</li>
68   *   <li>{@code 2009-01-01T00:00:00.000Z} <em>etc.</em></li>
69   * </ul>
70   *
71   * @see #equals(Timestamp)
72   * @see #compareTo(Timestamp)
73   */

164             switch (isIncluded)

219     private static byte last_day_in_month(int year, int month) {
220         boolean is_leap;
221         if ((year % 4) == 0) {
222             // divisible by 4 (lower 2 bits are zero) - may be a leap year
223             if ((year % 100) == 0) {
224                 // and divisible by 100 - not a leap year
225                 if ((year % 400) == 0) {
226                     // but divisible by 400 - then it is a leap year
227                     is_leap = true;
228                 }
229                 else {
230                     is_leap = false;
231                 }
232             }
233             else {
234                 is_leap = true;
235             }
236         }
237         else {
238             is_leap = false;
239         }
240         return (byte) (is_leap ? LEAP_DAYS_IN_MONTH[month] : NORMAL_DAYS_IN_MONTH[month]);
241     }

243     /**
244      * Applies the local time zone offset from UTC to the applicable time field
245      * values. Depending on the local time zone offset, adjustments
246      * (i.e. rollover) will be made to the Year, Day, Hour, Minute time field
247      * values.
248      *
249      * @param offset the local offset, in minutes from UTC.
250      */
251     private void apply_offset(int offset)
252     {
253         if (offset == 0) return;
254         if (offset < -24*60 || offset > 24*60) {
255             throw new IllegalArgumentException("bad offset " + offset);
256         }
257         // To convert _to_ UTC you must SUBTRACT the local offset
258         offset = -offset;
259         byte hour_offset = (byte) (offset / 60);
260         byte min_offset = (byte) (offset - (hour_offset * 60));
261 
262         if (offset < 0) {
263             _minute += min_offset;  // lower the minute value by adding a negative offset
264             _hour += hour_offset;
265             if (_minute < 0) {
266                 _minute += (byte) 60;
267                 _hour -= 1;
268             }
269             if (_hour >= 0) return;  // hour is 0-23
270             _hour += (byte) 24;
271             _day -= (byte) 1;
272             if (_day >= 1) return;  // day is 1-31
273             // we can't do this until we've figured out the month and year: _day += last_day_in_month(_year, _month);
274             _month -= 1;
275             if (_month >= 1) {
276                 _day += last_day_in_month(_year, _month);  // now we know (when the year doesn't change
277                 assert(_day == last_day_in_month(_year, _month));
278                 return;  // 1-12
279             }
280             _month += (byte) 12;
281             _year -= (short) 1;
282             if (_year < 1) throw new IllegalArgumentException("year is less than 1");
283             _day += last_day_in_month(_year, _month);  // and now we know, even if the year did change
284             assert(_day == last_day_in_month(_year, _month));
285         }
286         else {
287             _minute += min_offset;  // lower the minute value by adding a negative offset
288             _hour += hour_offset;
289             if (_minute > 59) {
290                 _minute -= (byte) 60;
291                 _hour += (byte) 1;
292             }
293             if (_hour < 24) return;  // hour is 0-23
294             _hour -= (byte) 24;
295             _day += (byte) 1;
296             if (_day <= last_day_in_month(_year, _month)) return;  // day is 1-31
297             // we can't do this until we figure out the final month and year: _day -= last_day_in_month(_year, _month);
298             _day = 1; // this is always the case
299             _month += (byte) 1;
300             if (_month <= 12) {
301                 return;  // 1-12
302             }
303             _month -= (byte) 12;
304             _year += (short) 1;
305             if (_year > 9999) throw new IllegalArgumentException("year exceeds 9999");
306         }
307     }

323     /**
324      * This method uses deprecated methods from {@link java.util.Date}
325      * instead of {@link Calendar} so that this code can be used (more easily)
326      * on the mobile Java platform (which has Date but does not have Calendar).
327      */
328     @SuppressWarnings("deprecation")
329     private void set_fields_from_millis(long millis)
330     {
331         if(millis < MINIMUM_TIMESTAMP_IN_MILLIS){
332             throw new IllegalArgumentException("year is less than 1");
333         }
334 
335         Date date = new Date(millis);
336 
337         // The Date getters return values in the Date's time zone (i.e. the system time zone).
338         // The components need to be converted to UTC before being validated for exact ranges, because the offset
339         // conversion can affect which values are considered valid. Simply verify that the values can fit in the
340         // destination type here. If they do not, they would be out of range no matter the offset.
341         _minute = requireByte(date.getMinutes(), "Minute");
342         _second = requireByte(date.getSeconds(), "Second");
343         _hour = requireByte(date.getHours(), "Hour");
344         _day = requireByte(date.getDate(), "Day");
345         _month = requireByte(date.getMonth() + 1, "Month");
346 
347         // Date does not correctly handle year values that represent year 0 or earlier in the system time zone through
348         // getYear(). This case is detected and forced to zero.
349         int offset = -date.getTimezoneOffset();
350         if(offset < 0 && MINIMUM_TIMESTAMP_IN_MILLIS - offset > millis) {
351             _year = 0;
352         } else {
353             _year = requireShort(date.getYear() + 1900, "Year");
354         }
355 
356         // Now apply the offset to convert the components to UTC.
357         apply_offset(offset);
358 
359         // Now that all components are in UTC, they may be validated for exact ranges.
360         this._year    = checkAndCastYear(_year);
361         this._month   = checkAndCastMonth(_month);
362         this._day     = checkAndCastDay(_day, _year, _month);
363         this._hour    = checkAndCastHour(_hour);
364         this._minute  = checkAndCastMinute(_minute);
365         this._second  = checkAndCastSecond(_second);
366     }

368     /**
369      * Copies data from a {@link Calendar} into this timestamp.
370      * Must only be called during construction due to timestamp immutabliity.
371      *
372      * @param cal must have at least one field set.
373      *
374      * @throws IllegalArgumentException if the calendar has no fields set.
375      */
376     private void set_fields_from_calendar(Calendar cal,
377                                           Precision precision,
378                                           boolean setLocalOffset)
379     {
380         _precision = precision;
381         _offset = UNKNOWN_OFFSET;
382         boolean dayPrecision = false;
383         boolean calendarHasMilliseconds = cal.isSet(Calendar.MILLISECOND);
384 
385         switch (this._precision) {
386             case FRACTION:
387             case SECOND:
388                 this._second = checkAndCastSecond(cal.get(Calendar.SECOND));
389                 if (calendarHasMilliseconds) {
390                     BigDecimal millis = BigDecimal.valueOf(cal.get(Calendar.MILLISECOND));
391                     this._fraction = millis.movePointLeft(3); // convert to fraction
392                     checkFraction(precision, this._fraction);
393                 }
394             case MINUTE:
395             {
396                 this._hour   = checkAndCastHour(cal.get(Calendar.HOUR_OF_DAY));
397                 this._minute = checkAndCastMinute(cal.get(Calendar.MINUTE));
398 
399                 // If this test is made before calling get(), it will return
400                 // false even when Calendar.setTimeZone() was called.
401                 if (setLocalOffset && cal.isSet(Calendar.ZONE_OFFSET))
402                 {
403                     int offset = cal.get(Calendar.ZONE_OFFSET);
404                     if (cal.isSet(Calendar.DST_OFFSET)) {
405                         offset += cal.get(Calendar.DST_OFFSET);
406                     }
407 
408                     // convert ms to minutes
409                     _offset = offset / (1000*60);
410                 }
411             }
412             case DAY:
413                 dayPrecision = true;
414             case MONTH:
415                 // Calendar months are 0 based, Timestamp months are 1 based
416                 this._month  = checkAndCastMonth((cal.get(Calendar.MONTH) + 1));
417             case YEAR:
418                 int year;
419                 if(cal.get(Calendar.ERA) == GregorianCalendar.AD) {
420                     year = cal.get(Calendar.YEAR);
421                 }
422                 else {
423                     year = -cal.get(Calendar.YEAR);
424                 }
425 
426                 this._year = checkAndCastYear(year);
427         }
428 
429         if (dayPrecision)
430         {
431             this._day = checkAndCastDay(cal.get(Calendar.DAY_OF_MONTH), _year, _month);
432         }
433 
434         if (_offset != UNKNOWN_OFFSET) {
435             // Transform our members from local time to Zulu
436             this.apply_offset(_offset);
437         }
438     }

547     /**
548      * Creates a new Timestamp from the individual time components. The
549      * individual time components are expected to be in UTC,
550      * with the local offset from UTC (i.e. {@code offset}) <em>already
551      * applied</em> to the time components.
552      * <p>
553      * Any time component that is more precise
554      * than the precision parameter {@code p} will be <em>excluded</em> from the
555      * calculation of the resulting Timestamp's point in time.
556      *
557      * @param frac must be >= 0 and < 1
558      *
559      * @param offset
560      *          the local offset from UTC, measured in minutes;
561      *          may be {@code null} to represent an unknown local offset
562      *
563      * @see #createFromUtcFields(Precision, int, int, int, int, int, int, BigDecimal, Integer)
564      */
565     private Timestamp(Precision p, int zyear, int zmonth, int zday,
566                       int zhour, int zminute, int zsecond, BigDecimal frac,
567                       Integer offset, boolean shouldApplyOffset)
568     {
569         boolean dayPrecision = false;
570 
571         switch (p) {
572         default:
573             throw new IllegalArgumentException("invalid Precision passed to constructor");
574         case FRACTION:
575         case SECOND:
576             if (frac == null || frac.equals(BigDecimal.ZERO))
577             {
578                 _fraction = null;
579             }
580             else
581             {
582                 _fraction = frac.abs();
583             }
584             _second = checkAndCastSecond(zsecond);
585         case MINUTE:
586             _minute = checkAndCastMinute(zminute);
587             _hour   = checkAndCastHour(zhour);
588             _offset = offset;      // offset must be null for years/months/days
589         case DAY:
590              dayPrecision = true;
591         case MONTH:
592             _month  = checkAndCastMonth(zmonth);
593         case YEAR:
594             _year   = checkAndCastYear(zyear);
595         }
596 
597         if (dayPrecision)
598         {
599             _day    = checkAndCastDay(zday, zyear, zmonth);
600         }
601 
602         _precision = checkFraction(p, _fraction);
603 
604         if (shouldApplyOffset && offset != null) {
605             apply_offset(offset);
606         }
607     }

609     /**
610      * Creates a new Timestamp from the individual time components. The
611      * individual time components are expected to be in UTC,
612      * with the local offset from UTC (i.e. {@code offset}) <em>already
613      * applied</em> to the time components.
614      * As such, if the given {@code offset} is non-null or zero, the resulting
615      * Timestamp will have time values that <em>DO NOT</em> match the time
616      * parameters. This method also has a behavior of precision "narrowing",
617      * detailed in the sub-section below.
618      *
619      * <p>
620      * For example, the following method calls will return Timestamps with
621      * values (in its local time) respectively:
622      *<pre>
623      * createFromUtcFields(Precision.SECOND, 2012, 2, 3, 4, 5, 6, 0.007, <b>null</b>)    will return 2012-02-03T04:05:06.007-00:00 (match)
624      * createFromUtcFields(Precision.SECOND, 2012, 2, 3, 4, 5, 6, 0.007, <b>0</b>)       will return 2012-02-03T04:05:06.007+00:00 (match)
625      * createFromUtcFields(Precision.SECOND, 2012, 2, 3, 4, 5, 6, 0.007, <b>480</b>)     will return 2012-02-03T<b>12</b>:05:06.007<b>+08:00</b> (do not match)
626      * createFromUtcFields(Precision.SECOND, 2012, 2, 3, 4, 5, 6, 0.007, <b>-480</b>)    will return 2012-02-<b>02</b>T<b>20</b>:05:06.007<b>-08:00</b> (do not match)
627      * createFromUtcFields(Precision.SECOND, 2012, 2, 3, 4, 5, 6, 0.007, <b>720</b>)     will return 2012-02-03T<b>16</b>:05:06.007<b>+12:00</b> (do not match)
628      * createFromUtcFields(Precision.SECOND, 2012, 2, 3, 4, 5, 6, 0.007, <b>-720</b>)    will return 2012-02-<b>02</b>T<b>16</b>:05:06.007<b>-12:00</b> (do not match)
629      *</pre>
630      * Note: All of these resulting Timestamps have the similar value (in UTC) 2012-02-03T04:05:06.007Z.
631      *
632      * <h3>Precision "Narrowing"</h3>
633      *
634      * <p>
635      * Any time component that is more precise
636      * than the precision parameter {@code p} will be <em>excluded</em> from the
637      * calculation of the resulting Timestamp's point in time.
638      * <p>
639      * For example, the following method calls will return Timestamps with
640      * values respectively:
641      *<pre>
642      * createFromUtcFields(<b>Precision.YEAR</b>    , 2012, 2, 3, 4, 5, 6, 0.007, 0)    will return 2012T
643      * createFromUtcFields(<b>Precision.MONTH</b>   , 2012, 2, 3, 4, 5, 6, 0.007, 0)    will return 2012-02T
644      * createFromUtcFields(<b>Precision.DAY</b>     , 2012, 2, 3, 4, 5, 6, 0.007, 0)    will return 2012-02-03T
645      * createFromUtcFields(<b>Precision.MINUTE</b>  , 2012, 2, 3, 4, 5, 6, 0.007, 0)    will return 2012-02-03T04:05Z
646      * createFromUtcFields(<b>Precision.SECOND</b>  , 2012, 2, 3, 4, 5, 6,  null, 0)    will return 2012-02-03T04:05:06Z
647      * createFromUtcFields(<b>Precision.SECOND</b>  , 2012, 2, 3, 4, 5, 6, 0.007, 0)    will return 2012-02-03T04:05:06.007Z
648      *</pre>
649      *
650      * @param p the desired timestamp precision. The result may have a
651      * different precision if the input data isn't precise enough.
652      *
653      * @param offset
654      *          the local offset from UTC, measured in minutes;
655      *          may be {@code null} to represent an unknown local offset.
656      */

731     private Timestamp(BigDecimal millis, Precision precision, Integer localOffset)
732     {
733         // check bounds to avoid hanging when calling longValue() on decimals with large positive exponents,
734         // e.g. 1e10000000
735         if(millis.compareTo(MINIMUM_TIMESTAMP_IN_MILLIS_DECIMAL) < 0 ||
736                 MAXIMUM_ALLOWED_TIMESTAMP_IN_MILLIS_DECIMAL.compareTo(millis) <= 0) {
737             throwTimestampOutOfRangeError(millis);
738         }
739         // quick handle integral zero
740         long ms = isIntegralZero(millis) ? 0 : millis.longValue();
741         set_fields_from_millis(ms);
742 
743         switch (precision)
744         {
745             case YEAR:
746                 _month  = 1;
747             case MONTH:
748                 _day    = 1;
749             case DAY:
750                 _hour   = 0;
751                 _minute = 0;
752             case MINUTE:
753                 _second = 0;
754             case SECOND:
755             case FRACTION:
756         }
757 
758         _offset = localOffset;
759         // The given BigDecimal may contain greater than milliseconds precision, which is the maximum precision that
760         // a Calendar can handle. Set the _fraction here so that extra precision (if any) is not lost.
761         // However, don't set the fraction if the given BigDecimal does not have precision at least to the tenth of
762         // a second.
763         if ((precision.includes(Precision.SECOND)) && millis.scale() > -3) {
764             BigDecimal secs = millis.movePointLeft(3);
765             BigDecimal secsDown = fastRoundZeroFloor(secs);
766             _fraction = secs.subtract(secsDown);
767         } else {
768             _fraction = null;
769         }
770         _precision = checkFraction(precision, _fraction);
771     }

839     private BigDecimal fastRoundZeroFloor(final BigDecimal decimal) {
840         BigDecimal fastValue = decimal.signum() < 0 ? BigDecimal.ONE.negate() : BigDecimal.ZERO;
841 
842         return isIntegralZero(decimal) ? fastValue : decimal.setScale(0, RoundingMode.FLOOR);

845     private boolean isIntegralZero(final BigDecimal decimal) {
846         // zero || no low-order bits || < 1.0
847         return  decimal.signum() == 0
848             || decimal.scale() < -63
849             || (decimal.precision() - decimal.scale() <= 0);

907     static final String NULL_TIMESTAMP_IMAGE = "null.timestamp";
908     static final int    LEN_OF_NULL_IMAGE    = NULL_TIMESTAMP_IMAGE.length();
909     static final int    END_OF_YEAR          =  4;  // 1234T
910     static final int    END_OF_MONTH         =  7;  // 1234-67T
911     static final int    END_OF_DAY           = 10;  // 1234-67-90T
912     static final int    END_OF_MINUTES       = 16;
913     static final int    END_OF_SECONDS       = 19;

916     /**
917      * Returns a new Timestamp that represents the point in time, precision
918      * and local offset defined in Ion format by the {@link CharSequence}.
919      *
920      * @param ionFormattedTimestamp
921      *          a sequence of characters that is the Ion representation of a
922      *          Timestamp
923      *
924      * @throws IllegalArgumentException
925      *          if the {@code CharSequence} is an invalid Ion representation
926      *          of a Timestamp;
927      *          or if the {@code CharSequence} has excess characters which
928      *          are not one of the following valid thirteen numeric-stop
929      *          characters (escaped accordingly for readability):
930      *          <code>{}[](),\"\'\ \t\n\r}</code>
931      *
932      * @return
933      *          {@code null} if the {@code CharSequence} is "null.timestamp"
934      *
935      * @see <a href="http://amzn.github.io/ion-docs/docs/spec.html#timestamp">Ion Timestamp Page</a>
936      * @see <a href="http://www.w3.org/TR/NOTE-datetime">W3C Note on Date and Time Formats</a>
937      */
938     public static Timestamp valueOf(CharSequence ionFormattedTimestamp)
939     {
940         final CharSequence in = ionFormattedTimestamp;
941         int pos;
942 
943         final int length = in.length();
944         if (length == 0)
945         {
946             throw fail(in);
947         }
948 
949         // check for 'null.timestamp'
950         if (in.charAt(0) == 'n') {
951             if (length >= LEN_OF_NULL_IMAGE
952                 && NULL_TIMESTAMP_IMAGE.contentEquals(in.subSequence(0, LEN_OF_NULL_IMAGE)))
953             {
954                 if (length > LEN_OF_NULL_IMAGE) {
955                     if (!isValidFollowChar(in.charAt(LEN_OF_NULL_IMAGE))) {
956                         throw fail(in);
957                     }
958                 }
959                 return null;
960             }
961             throw fail(in);
962         }
963 
964         int year  = 1;
965         int month = 1;
966         int day   = 1;
967         int hour  = 0;
968         int minute = 0;
969         int seconds = 0;
970         BigDecimal fraction = null;
971         Precision precision;
972 
973         // fake label to turn goto's into a break so Java is happy :) enjoy
974         do {
975             // otherwise we expect yyyy-mm-ddThh:mm:ss.ssss+hh:mm
976             if (length < END_OF_YEAR + 1) {  // +1 for the "T"
977                 throw fail(in, "year is too short (must be at least yyyyT)");
978             }
979             pos = END_OF_YEAR;
980             precision = Precision.YEAR;
981             year  = read_digits(in, 0, 4, -1, "year");
982 
983             char c = in.charAt(END_OF_YEAR);
984             if (c == 'T') break;
985             if (c != '-') {
986                 throw fail(in,
987                            "expected \"-\" between year and month, found "
988                                + printCodePointAsString(c));
989             }
990             if (length < END_OF_MONTH + 1) {  // +1 for the "T"
991                 throw fail(in, "month is too short (must be yyyy-mmT)");
992             }
993             pos = END_OF_MONTH;
994             precision = Precision.MONTH;
995             month = read_digits(in, END_OF_YEAR + 1, 2, -1,  "month");
996 
997             c = in.charAt(END_OF_MONTH);
998             if (c == 'T') break;
999             if (c != '-') {
1000                 throw fail(in,
1001                            "expected \"-\" between month and day, found "
1002                                + printCodePointAsString(c));
1003             }
1004             if (length < END_OF_DAY) {
1005                 throw fail(in, "too short for yyyy-mm-dd");
1006             }
1007             pos = END_OF_DAY;
1008             precision = Precision.DAY;
1009             day   = read_digits(in, END_OF_MONTH + 1, 2, -1, "day");
1010             if (length == END_OF_DAY) break;
1011             c = in.charAt(END_OF_DAY);
1012             if (c != 'T') {
1013                 throw fail(in,
1014                            "expected \"T\" after day, found "
1015                                + printCodePointAsString(c));
1016             }
1017             if (length == END_OF_DAY + 1) break;
1018 
1019             // now lets see if we have a time value
1020             if (length < END_OF_MINUTES) {
1021                 throw fail(in, "too short for yyyy-mm-ddThh:mm");
1022             }
1023             hour   = read_digits(in, 11, 2, ':', "hour");
1024             minute = read_digits(in, 14, 2, -1, "minutes");
1025             pos = END_OF_MINUTES;
1026             precision = Precision.MINUTE;
1027 
1028             // we may have seconds
1029             if (length <= END_OF_MINUTES || in.charAt(END_OF_MINUTES) != ':')
1030             {
1031                 break;
1032             }
1033             if (length < END_OF_SECONDS) {
1034                 throw fail(in, "too short for yyyy-mm-ddThh:mm:ss");
1035             }
1036             seconds = read_digits(in, 17, 2, -1, "seconds");
1037             pos = END_OF_SECONDS;
1038             precision = Precision.SECOND;
1039 
1040             if (length <= END_OF_SECONDS || in.charAt(END_OF_SECONDS) != '.')
1041             {
1042                 break;
1043             }
1044             pos = END_OF_SECONDS + 1;
1045             while (length > pos && Character.isDigit(in.charAt(pos))) {
1046                 pos++;
1047             }
1048             if (pos <= END_OF_SECONDS + 1) {
1049                 throw fail(in,
1050                            "must have at least one digit after decimal point");
1051             }
1052             fraction = new BigDecimal(in.subSequence(19, pos).toString());
1053         } while (false);
1054 
1055         Integer offset;
1056 
1057         // now see if they included a timezone offset
1058         char timezone_start = pos < length ? in.charAt(pos) : '\n';
1059         if (timezone_start == 'Z') {
1060             offset = 0;
1061             pos++;
1062         }
1063         else if (timezone_start == '+' || timezone_start == '-')
1064         {
1065             if (length < pos + 5) {
1066                 throw fail(in, "local offset too short");
1067             }
1068             // +/- hh:mm
1069             pos++;
1070             int tzdHours = read_digits(in, pos, 2, ':', "local offset hours");
1071             if (tzdHours < 0 || tzdHours > 23) {
1072                 throw fail(in,
1073                            "local offset hours must be between 0 and 23 inclusive");
1074             }
1075             pos += 3;
1076 
1077             int tzdMinutes = read_digits(in, pos, 2, -1, "local offset minutes");
1078             if (tzdMinutes > 59) {
1079                 throw fail(in,
1080                            "local offset minutes must be between 0 and 59 inclusive");
1081             }
1082             pos += 2;
1083 
1084             int temp = tzdHours * 60 + tzdMinutes;
1085             if (timezone_start == '-') {
1086                 temp = -temp;
1087             }
1088             if (temp == 0 && timezone_start == '-') {
1089                 // int doesn't do negative zero very elegantly
1090                 offset = null;
1091             }
1092             else {
1093                 offset = temp;
1094             }
1095         }
1096         else {
1097             switch (precision) {
1098                 case YEAR:
1099                 case MONTH:
1100                 case DAY:
1101                     break;
1102                 default:
1103                     throw fail(in, "missing local offset");
1104             }
1105             offset = null;
1106         }
1107         if (length > (pos + 1) && !isValidFollowChar(in.charAt(pos + 1)))
1108         {
1109             throw fail(in, "invalid excess characters");
1110         }
1111 
1112         Timestamp ts =
1113             new Timestamp(precision, year, month, day,
1114                           hour, minute, seconds, fraction, offset, APPLY_OFFSET_YES);
1115         return ts;
1116     }

1118     private static int read_digits(CharSequence in, int start, int length,
1119                                    int terminator, String field)
1120     {
1121         int ii, value = 0;
1122         int end = start + length;
1123 
1124         if (in.length() < end) {
1125             throw fail(in,
1126                        field + " requires " + length + " digits");
1127         }
1128 
1129         for (ii=start; ii<end; ii++) {
1130             char c = in.charAt(ii);
1131             if (!Character.isDigit(c)) {
1132                 // FIXME this will give incorrect message if c is a surrogate
1133                 throw fail(in,
1134                            field + " has non-digit character "
1135                                + printCodePointAsString(c));
1136             }
1137             value *= 10;
1138             value += c - '0';
1139         }
1140 
1141         // Check the terminator if requested.
1142         if (terminator != -1) {
1143             if (ii >= in.length() || in.charAt(ii) != terminator) {
1144                 throw fail(in,
1145                            field + " should end with "
1146                                + printCodePointAsString(terminator));
1147             }
1148         }
1149         // Otherwise make sure we don't have too many digits.
1150         else if (ii < in.length() && Character.isDigit(in.charAt(ii))) {
1151             throw fail(in,
1152                        field + " requires " + length + " digits but has more");
1153         }
1154 
1155         return value;
1156     }

2591         // is found to be a common input data set.
2592 
2593         final int prime = 8191;
2594         int result = HASH_SIGNATURE;
2595 
2596         result = prime * result + (_fraction != null
2597             ? _fraction.hashCode()
2598             : 0);
2599 
2600         result ^= (result << 19) ^ (result >> 13);
2601 
2602         result = prime * result + this._year;
2603         result = prime * result + this._month;
2604         result = prime * result + this._day;
2605         result = prime * result + this._hour;
2606         result = prime * result + this._minute;
2607         result = prime * result + this._second;
2608 
2609         result ^= (result << 19) ^ (result >> 13);
2610 
2611         Precision precision = this._precision == Precision.FRACTION ? Precision.SECOND : this._precision;
2612         result = prime * result + precision.toString().hashCode();
2613 
2614         result ^= (result << 19) ^ (result >> 13);
2615 
2616         result = prime * result + (_offset == null ? 0 : _offset.hashCode());
2617 
2618         result ^= (result << 19) ^ (result >> 13);
2619 
2620         return result;
2621     }

2625     /**
2626      * Performs a comparison of the two points in time represented by two
2627      * Timestamps.
2628      * If the point in time represented by this Timestamp precedes that of
2629      * {@code t}, then {@code -1} is returned.
2630      * If {@code t} precedes this Timestamp then {@code 1} is returned.
2631      * If the Timestamps represent the same point in time, then
2632      * {@code 0} is returned.
2633      * Note that a {@code 0} result does not imply that the two Timestamps are
2634      * {@link #equals}, as the local offset or precision of the two Timestamps
2635      * may be different.
2636      *
2637      * <p>
2638      * This method is provided in preference to individual methods for each of
2639      * the six boolean comparison operators (&lt;, ==, &gt;, &gt;=, !=, &lt;=).
2640      * The suggested idiom for performing these comparisons is:
2641      * {@code (x.compareTo(y)}<em>&lt;op&gt;</em>{@code 0)},
2642      * where <em>&lt;op&gt;</em> is one of the six comparison operators.
2643      *
2644      * <p>
2645      * For example, the pairs below will return a {@code 0} result:
2646      * <ul>
2647      *   <li>{@code 2009T}</li>
2648      *   <li>{@code 2009-01T}</li>
2649      *   <li>{@code 2009-01-01T}</li>
2650      *   <li>{@code 2009-01-01T00:00Z}</li>
2651      *   <li>{@code 2009-01-01T00:00:00Z}</li>
2652      *   <li>{@code 2009-01-01T00:00:00.0Z}</li>
2653      *   <li>{@code 2009-01-01T00:00:00.00Z}</li>
2654      *
2655      *   <li>{@code 2008-12-31T16:00-08:00}</li>
2656      *   <li>{@code 2008-12-31T12:00-12:00}</li>
2657      *   <li>{@code 2009-01-01T12:00+12:00}</li>
2658      * </ul>
2659      *
2660      * <p>
2661      * Use the {@link #equals(Timestamp)} method to compare the point
2662      * in time, <em>including</em> precision and local offset.
2663      *
2664      * @param t
2665      *          the other {@code Timestamp} to compare this {@code Timestamp} to
2666      *
2667      * @return
2668      *          -1, 0, or 1 if this {@code Timestamp}
2669      *          is less than, equal to, or greater than {@code t} respectively
2670      *
2671      * @throws NullPointerException if {@code t} is null.
2672      *
2673      * @see #equals(Timestamp)
2674      */
2675     public int compareTo(Timestamp t)
2676     {
2677         // Test at millisecond precision first.
2678         long this_millis = this.getMillis();
2679         long arg_millis = t.getMillis();
2680         if (this_millis != arg_millis) {
2681             return (this_millis < arg_millis) ? -1 : 1;
2682         }
2683 
2684         // Values are equivalent at millisecond precision, so compare fraction
2685 
2686         BigDecimal this_fraction =
2687             ((this._fraction == null) ? BigDecimal.ZERO : this._fraction);
2688         BigDecimal arg_fraction =
2689             (( t._fraction == null) ? BigDecimal.ZERO :  t._fraction);
2690         return this_fraction.compareTo(arg_fraction);
2691     }

2713     /**
2714      * Compares this {@link Timestamp} to another {@link Timestamp} object.
2715      * The result is {@code true} if and only if the parameter
2716      * represents the same point in time and has
2717      * the same precision and local offset as this object.
2718      * <p>
2719      * These pairs are {@link #equals} to each other, as they
2720      * represent the same points in time, precision and local offset:
2721      *
2722      * <ul>
2723      *   <li>{@code 2001-01-01T11:22+00:00} (minute precision, in UTC)</li>
2724      *   <li>{@code 2001-01-01T11:22Z} (minute precision, in UTC)</li>
2725      * </ul>
2726      *
2727      * <p>
2728      * On the other hand, none of these pairs are {@link #equals} to each other,
2729      * they represent the same points in time, but with different precisions
2730      * and/or local offsets:
2731      *
2732      * <ul>
2733      *   <li>{@code 2001T} (year precision, unknown local offset)</li>
2734      *   <li>{@code 2001-01T} (month precision, unknown local offset)</li>
2735      *   <li>{@code 2001-01-01T} (day precision, unknown local offset)</li>
2736      *
2737      *   <li>{@code 2001-01-01T00:00-00:00} (second precision, unknown local offset)</li>
2738      *   <li>{@code 2001-01-01T00:00+00:00} (second precision, in UTC)</li>
2739      *
2740      *   <li>{@code 2001-01-01T00:00.000-00:00} (millisecond precision, unknown local offset)</li>
2741      *   <li>{@code 2001-01-01T00:00.000+00:00} (millisecond precision, in UTC)</li>
2742      * </ul>
2743      *
2744      * <p>
2745      * Use the {@link #compareTo(Timestamp)} method to compare only the point
2746      * in time, <em>ignoring</em> precision and local offset.
2747      *
2748      * @see #compareTo(Timestamp)
2749      */
2750     public boolean equals(Timestamp t)
2751     {
2752         if (this == t) return true;
2753         if (t == null) return false;
2754 
2755         // FRACTION precision is equivalent to SECOND precision.
2756         Precision thisPrecision = this._precision.includes(Precision.SECOND) ? Precision.SECOND : this._precision;
2757         Precision thatPrecision = t._precision.includes(Precision.SECOND) ? Precision.SECOND : t._precision;
2758 
2759         // if the precisions are not the same the values are not
2760         // precision doesn't matter WRT equality
2761         if (thisPrecision != thatPrecision) return false;
2762 
2763         // if the local offset are not the same the values are not
2764         if (this._offset == null) {
2765             if (t._offset != null)  return false;
2766         }
2767         else {
2768             if (t._offset == null) return false;
2769         }
2770 
2771         // so now we check the actual time value
2772         if (this._year   != t._year)    return false;
2773         if (this._month  != t._month)   return false;
2774         if (this._day    != t._day)     return false;
2775         if (this._hour   != t._hour)    return false;
2776         if (this._minute != t._minute)  return false;
2777         if (this._second != t._second)  return false;
2778 
2779         // and if we have a local offset, check the value here
2780         if (this._offset != null) {
2781             if (this._offset.intValue() != t._offset.intValue()) return false;
2782         }
2783 
2784         // we only look at the fraction if we know that it's actually there
2785         if ((this._fraction != null && t._fraction == null)
2786                 || (this._fraction == null && t._fraction != null)) {
2787             // one of the fractions are null
2788             return false;
2789         }
2790         if (this._fraction == null && t._fraction == null) {
2791             // both are null
2792             return true;
2793         }
2794         return this._fraction.equals(t._fraction);
2795     }
