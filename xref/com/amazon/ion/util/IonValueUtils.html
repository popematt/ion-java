16  package com.amazon.ion.util;
17  
18  import com.amazon.ion.IonValue;
19  
20  /**
21   * Utility methods for working with {@link IonValue}s.
22   */
23  public class IonValueUtils
24  {
25      /**
26       * Determines whether a value is Java null, or any Ion null.
27       *
28       * @param value may be null.
29       *
30       * @return {@code (value == null || value.isNullValue())}
31       */
32      public static final boolean anyNull(IonValue value)
33      {
34          return (value == null || value.isNullValue());
35      }
36  }
