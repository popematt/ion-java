16  package com.amazon.ion.util;
17  
18  import static com.amazon.ion.facet.Facets.asFacet;
19  
20  import com.amazon.ion.Span;
21  import com.amazon.ion.SpanProvider;
22  
23  /**
24   * Utility methods for working with {@link Span}s.
25   *
26   */
27  public final class Spans
28  {
29      /**
30       * Attempts to get a {@link Span} from the given object, if it
31       * supports {@link SpanProvider#currentSpan()}.
32       *
33       * @param spanProvider may be null.
34       * @return null if there's not a current span.
35       */
36      public static Span currentSpan(Object spanProvider)
37      {
38          SpanProvider sp = asFacet(SpanProvider.class, spanProvider);
39          Span span = (sp == null ? null : sp.currentSpan());
40          return span;
41      }
42  
43  
44      /**
45       * Attempts to get a {@link Span} facet from the given object, if it
46       * supports {@link SpanProvider#currentSpan()}.
47       *
48       * @param spanProvider may be null.
49       * @return null if there's not a current span supporting
50       *  {@code spanFacetType}.
51       */
52      public static <T> T currentSpan(Class<T> spanFacetType,
53                                      Object spanProvider)
54      {
55          Span span = currentSpan(spanProvider);
56          T spanFacet = asFacet(spanFacetType, span);
57          return spanFacet;
58      }
59  }
