42  /**
43   * Provides equivalence comparisons between two {@link IonValue}s, following
44   * the contract of {@link IonValue#equals(Object)}.
45   *
46   * <p>
47   * Basic usage of this class is as follows:
48   *
49   *<pre>
50   *    IonValue v1 = ...;
51   *    IonValue v2 = ...;
52   *    com.amazon.ion.util.Equivalence.ionEquals( v1, v2 );
53   *</pre>
54   *
55   * More likely, a static import would make using this class easier.
56   *
57   *<pre>
58   *    import static com.amazon.ion.util.Equivalence.ionEquals;
59   *    ...
60   *    boolean equivalent = ionEquals( v1, v2 );
61   *</pre>
62   *
63   * <p>
64   * Additional options are available by configuring an <code>Equivalence</code>
65   * instance using {@link Equivalence.Builder}. For example:
66   *
67   * <pre>
68   *     com.amazon.ion.util.Equivalence equivalence =
69   *         new com.amazon.ion.util.Equivalence.Builder()
70   *             .withEpsilon(1e-6)
71   *             .build();
72   *     IonValue v1 = ...;
73   *     IonValue v2 = ...;
74   *     equivalence.ionValueEquals( v1, v2 );
75   * </pre>
76   *
77   * <p>
78   * <h3>Ion Equivalence</h3>
79   * In order to make Ion a useful model to describe data, we must first define
80   * the notion of equivalence for all values representable in Ion. Equivalence
81   * with respect to Ion values means that if two Ion values, X and Y, are
82   * equivalent, they represent the same data and can be substituted for the other
83   * without loss of information.
84   *
85   * This relationship is:
86   * <ul>
87   *   <li>
88   *     symmetric: X is equivalent to Y if and only if Y is equivalent to X.
89   *   </li>
90   *   <li>
91   *     transitive: if X is equivalent to Y and Y is equivalent to Z, then X is
92   *     equivalent to Z.
93   *   </li>
94   *   <li>
95   *     reflexive: X is equivalent to X.
96   *   </li>
97   * </ul>
98   *
99   * <h4>Ordered Sequence Equivalence</h4>
100  * When an ordered sequence (i.e. tuple) of elements is specified in this
101  * document, equivalence over such an ordered sequence is defined as follows.
102  *
103  * A tuple, A = (a1, a2, ..., an), is equivalent to another tuple, B = (b1, b2,
104  * ..., bm) if and only if the cardinality (number of elements) of A equals the
105  * cardinality of B (i.e. n == m) and ai is equivalent to bi for i = 1 ... n.
106  *
107  * <h4>Un-Ordered Sequence Equivalence</h4>
108  * When an un-ordered sequence (i.e. bag or multi-set) is specified in this
109  * document, equivalence over such a sequence is defined as follows.
110  *
111  * A bag, A = {a1, a2, ..., an} is equivalent to B = {b1, b2, ..., bm} if and
112  * only if the cardinality of A is equal to the cardinality of B and for each
113  * element, x, in A there exists a distinct element, y, in B for which x is
114  * equivalent to y.
115  *
116  * <h4>Values</h4>
117  * Any arbitrary, atomic value in the Ion Language can be denoted as the tuple,
118  * (A, V), where A is an ordered list of annotations, and V is an Ion Primitive
119  * Data or Ion Complex Data value. The list of annotations, A is an tuple of Ion
120  * Symbols (a specific type of Ion Primitive).
121  *
122  * <p>
123  * <h3>Terminology</h3>
124  * Within this class, <i>strict</i> equivalence refers to Ion data model
125  * equivalence as defined above and by the
126  * <a href="http://amzn.github.io/ion-docs/docs/spec.html">Ion
127  * Specification</a>. <i>Structural</i> or <i>non-strict</i> equivalence
128  * follows the same rules as strict equivalence, except that
129  * <ul>
130  *   <li>Annotations are not considered, and</li>
131  *   <li>Timestamps that represent the same instant in time are always
132  *   considered equivalent.</li>
133  * </ul>
134  */
135 public final class Equivalence {

219     private static int compareAnnotations(SymbolToken[] ann1,
220                                           SymbolToken[] ann2)
221     {
222         int len = ann1.length;
223         int result = len - ann2.length;
224 
225         if (result == 0) {
226             for (int i=0; (result == 0) && (i < len); i++) {
227                 result = compareSymbolTokens(ann1[i], ann2[i]);
228             }
229         }
230 
231         return result;
232     }

234 
235     private static int compareSymbolTokens(SymbolToken tok1,
236                                            SymbolToken tok2)
237     {
238         String text1 = tok1.getText();
239         String text2 = tok2.getText();
240         if (text1 == null || text2 == null) {
241             if (text1 != null) return  1;
242             if (text2 != null) return -1;
243 
244             int sid1 = tok1.getSid();
245             int sid2 = tok2.getSid();
246             if (sid1 < sid2) return -1;
247             if (sid1 > sid2) return  1;
248             return 0;
249         }
250 
251         return text1.compareTo(text2);
252     }

255     /**
256      * Converts an IonStruct to a multi-set for use in IonStruct equality
257      * checks. This method returns the multi-set as a {@code Map<Field, Field>}.
258      * <p>
259      * A multi-set supports order-independent equality, and may have duplicate
260      * elements.
261      * <p>
262      * Take special note that {@link Set} is missing a {@code get()} API,
263      * and cannot contain duplicate elements, hence we cannot use it.
264      */
265     private static final Map<Field, Field>
266         convertToMultiSet(final IonStruct struct, final Configuration configuration) {
267 
268         final Map<Field, Field> structMultiSet =
269             new HashMap<Field, Field>();
270 
271         for (final IonValue val : struct) {
272             final Field item = new Field(val, configuration);
273             Field curr = structMultiSet.put(item, item);
274             // curr will be non-null if the multi-set already contains the
275             // name/value pair
276             if (curr != null) {
277                 // Set the 'occurrences' of the Field that is contained in
278                 // the multi-set (i.e. item) to that of the previously mapped
279                 // Field (i.e. curr)
280                 item.occurrences = curr.occurrences;
281             }
282             // At this point, item will be an existing
283             // name/value pair in the multi-set - increment its occurrence
284             item.occurrences++;
285         }
286 
287         return structMultiSet;
288     }

291     private static int compareStructs(final IonStruct s1,
292                                       final IonStruct s2,
293                                       final Configuration configuration)
294     {
295         int result = s1.size() - s2.size();
296         if (result == 0) {
297             // We convert IonStruct s1 to a multi-set (which is a
298             // Map<Field, Field>). Refer to convertToMultiSet()'s
299             // documentation for more info
300             final Map<Field, Field> s1MultiSet
301                     = convertToMultiSet(s1, configuration);
302 
303             // Iterates through each name/value pair in IonStruct s2 and
304             // determine if it also occurs in s1MultiSet.
305             // During each iteration:
306             //          If it does, remove an occurrence from s1MultiSet
307             //          If it doesn't, the two IonStructs aren't equal
308             for (IonValue val : s2) {
309                 Field field = new Field(val, configuration);
310 
311                 // Find an occurrence of the name/value pair in s1MultiSet
312                 Field mappedValue = s1MultiSet.get(field);
313 
314                 if (mappedValue == null || mappedValue.occurrences == 0) {
315                     // No match in occurrences, the IonStructs aren't equal
316                     return -1;
317                 }
318 
319                 // Remove an occurrence by decrementing count instead of
320                 // explicitly calling Map.remove(), as Map.remove() is a slower
321                 // operation
322                 mappedValue.occurrences--;
323             }
324         }
325         return result;
326     }

329     private static int compareSequences(final IonSequence s1,
330                                         final IonSequence s2,
331                                         final Configuration configuration)
332     {
333         int result = s1.size() - s2.size();
334         if (result == 0) {
335             Iterator<IonValue> iter1 = s1.iterator();
336             Iterator<IonValue> iter2 = s2.iterator();
337             while (iter1.hasNext()) {
338                 result = ionCompareToImpl(iter1.next(),
339                                           iter2.next(),
340                                           configuration);
341                 if (result != 0) {
342                     break;
343                 }
344             }
345         }
346         return result;
347     }

350     /** Compare LOB content by stream--assuming non-null. */
351     private static int compareLobContents(final IonLobIonLob">IonLob lob1, final IonLob lob2)
352     {
353         int in1 = lob1.byteSize();
354         int in2 = lob2.byteSize();
355         int result = (in1- in2);
356 
357         if (result == 0) {
358             final InputStream stream1 = lob1.newInputStream();
359             final InputStream stream2 = lob2.newInputStream();
360 
361             // too bad Java doesn't do RAII with better syntax...
362             try {
363               try {
364                 try {
365                   while (result == 0) {
366                       in1 = stream1.read();
367                       in2 = stream2.read();
368                       if (in1 == -1 || in2 == -1) {
369                         if (in1 != -1) result = 1;
370                         if (in2 != -1) result = -1;
371                         break;
372                       }
373                       result = (in1 - in2);
374                   }
375                 } finally {
376                   stream1.close();
377                 }
378               } finally {
379                 stream2.close();
380               }
381             } catch (final IOException e) {
382                 // XXX hopefully won't happen with LOB streams--if it does rethrow unchecked.
383                 //     this would violate Object.equals() would it not?
384                 throw new IonException(e);
385             }
386         }
387 
388         return result;
389     }

392     /**
393      * Class that denotes a name/value pair in Structs.
394      * <p>
395      * Structs are unordered collections of name/value pairs. The names are
396      * symbol tokens, and the values are unrestricted. Each name/value pair is
397      * called a <em>field</em>.
398      * <p>
399      * The responsibilities of this class is to expose a
400      * {@code Map<Field, Field>} as a multi-set. A Field instance holds the name
401      * and value of a Struct field, and counts the number of times that
402      * name/value pair occurs within the multi-set.
403      * <p>
404      * For example, an IonStruct:
405      *<pre>
406      *  {
407      *    a : 123,
408      *    a : 123
409      *  }
410      *</pre>
411      * will be converted into a multi-set {@code Map<Field, Field>} with
412      * a single {@code Field} -> {@code Field} with {@code occurrences} of 2.
413      * <p>
414      * Refer to
415      * {@link Equivalence#convertToMultiSet(IonStruct, Configuration)} and
416      * {@link Field#equals(Object)} for more info.
417      * <p>
418      * NOTE: This class should only be instantiated for the sole purpose of
419      * using it as either a <em>key</em> or <em>value</em> in a {@link Map}.
420      */
421     static class Field {
422         private final String    name; // aka field name
423         private final IonValue  value;
424         private final Configuration configuration;
425 
426         /**
427          * Number of times that this specific field (with the same name
428          * and value) occurs within the struct.
429          */
430         private int occurrences;
431 
432         Field(final IonValue value, final Configuration configuration)
433         {
434             SymbolToken tok = value.getFieldNameSymbol();
435             String name = tok.getText();
436             if (name == null) {
437                 // TODO amzn/ion-java/issues/23 Problematic with unknown field names.
438                 name = UNKNOWN_SYMBOL_TEXT_PREFIX + tok.getSid();
439             }
440             this.name = name;
441             this.value = value;
442             this.configuration = configuration;
443 
444             // Occurrences of this name/value pair is 0 initially
445             this.occurrences = 0;
446         }
447 
448         @Override
449         public int hashCode() {
450             return name.hashCode();
451             // TODO amzn/ion-java/issues/58 : implement hash code such that it respects
452             // 'strict'. The prevously attempted fix is commented out below but
453             // is not sufficient because value.hasCode will always include
454             // type annotations in the hash computation. Type annotations
455             // should not be part of the hash computation if strict=true.
456 //            result = (31 * result) + value.hashCode();
457 //            return result;
458         }
459 
460         /**
461          * This method is implicitly called by {@link Map#get(Object)} to
462          * obtain a value to which {@code other} is mapped.
463          */
464         @Override
465         public boolean equals(final Object other) {
466             // We can assume other is always a Field and strict
467             // is the same - internal usage dictates it.
468             final Field sOther = (Field) other;
469 
470             return name.equals(sOther.name)
471                 && ionEqualsImpl(value, ((Field) other).value, configuration);
472         }
473     }

477                                          final Configuration configuration)
478     {
479         return (ionCompareToImpl(v1, v2, configuration) == 0);
480     }

482     private static int ionCompareToImpl(final IonValue v1,
483                                         final IonValue v2,
484                                         final Configuration configuration)
485     {
486         int result = 0;
487 
488         if (v1 == null || v2 == null) {
489             if (v1 != null) result = 1;
490             if (v2 != null) result = -1;
491             // otherwise v1 == v2 == null and result == 0
492             return result;
493         }
494 
495         // check type
496         IonType ty1 = v1.getType();
497         IonType ty2 = v2.getType();
498         result = ty1.compareTo(ty2);
499 
500         if (result == 0) {
501             boolean bo1 = v1.isNullValue();
502             boolean bo2 = v2.isNullValue();
503 
504             if (bo1 || bo2) {
505                  // null combination case--we already know they are
506                  // the same type
507                 if (!bo1) result = 1;
508                 if (!bo2) result = -1;
509                 // othersize they're equal (and null values)
510             }
511             else {
512                 // value compare only if both are not null
513                 switch (ty1)
514                 {
515                 case NULL:
516                     // never visited, precondition is that both are not null
517                     break;
518                 case BOOL:
519                     if (((IonBool) v1).booleanValue()) {
520                         result = ((IonBool) v2).booleanValue() ? 0 : 1;
521                     }
522                     else {
523                         result = ((IonBool) v2).booleanValue() ? -1 : 0;
524                     }
525                     break;
526                 case INT:
527                     result = ((IonInt) v1).bigIntegerValue().compareTo(
528                              ((IonInt) v2).bigIntegerValue());
529                     break;
530                 case FLOAT:
531                     double double1 = ((IonFloat) v1).doubleValue();
532                     double double2 = ((IonFloat) v2).doubleValue();
533                     if (configuration.epsilon != null
534                             && (double1 == double2 || Math.abs(double1 - double2) <= configuration.epsilon)) {
535                         result = 0;
536                     } else {
537                         result = Double.compare(double1, double2);
538                     }
539                     break;
540                 case DECIMAL:
541                     assert !PUBLIC_COMPARISON_API; // TODO amzn/ion-java/issues/26
542                     result = Decimal.equals(((IonDecimal) v1).decimalValue(),
543                                             ((IonDecimal) v2).decimalValue())
544                                             ? 0 : 1;
545                     break;
546                 case TIMESTAMP:
547                     if (configuration.isStrict) {
548                         assert !PUBLIC_COMPARISON_API; // TODO amzn/ion-java/issues/26
549                         result = (((IonTimestamp) v1).timestampValue().equals(
550                                   ((IonTimestamp) v2).timestampValue())
551                                   ? 0 : 1);
552                     }
553                     else {
554                         // This is kind of lying here, the 'strict' boolean
555                         // (if false) denotes ONLY that annotations are not
556                         // check for equality. But what this is doing here is
557                         // that it is also ignoring IonTimesamps' precision and
558                         // local offset.
559                         result = ((IonTimestamp) v1).timestampValue().compareTo(
560                                  ((IonTimestamp) v2).timestampValue());
561                     }
562                     break;
563                 case STRING:
564                     result = (((IonText) v1).stringValue()).compareTo(
565                               ((IonText) v2).stringValue());
566                     break;
567                 case SYMBOL:
568                     result = compareSymbolTokens(((IonSymbol) v1).symbolValue(),
569                                                  ((IonSymbol) v2).symbolValue());
570                     break;
571                 case BLOB:
572                 case CLOB:
573                     result = compareLobContents((IonLobef="../../../../com/amazon/ion/IonLob.html#IonLob">IonLob) v1, (IonLob) v2);
574                     break;
575                 case STRUCT:
576                     assert !PUBLIC_COMPARISON_API; // TODO amzn/ion-java/issues/26
577                     result = compareStructs((IonStruct) v1,
578                                             (IonStruct) v2,
579                                             configuration);
580                     break;
581                 case LIST:
582                 case SEXP:
583                 case DATAGRAM:
584                     result = compareSequences((IonSequence) v1,
585                                               (IonSequence) v2,
586                                               configuration);
587                     break;
588                 }
589             }
590         }
591 
592         // if the values are otherwise equal, but the caller wants strict
593         // comparison, then we check the annotations
594         if ((result == 0) && configuration.isStrict) {
595             // check tuple equality over the annotations
596             // (which are symbol tokens)
597             result = compareAnnotations(v1.getTypeAnnotationSymbols(),
598                                         v2.getTypeAnnotationSymbols());
599         }
600 
601         return result;
602     }
