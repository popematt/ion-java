35  /**
36   * A base class for extending Ion {@link ValueVisitor}s.
37   * All <code>visit</code> methods are implemented to call
38   * {@link #defaultVisit(IonValue)}.
39   */
40  public abstract class AbstractValueVisitor
41      implements ValueVisitor
42  {
43  
44      /**
45       * Default visitation behavior, called by all <code>visit</code> methods
46       * in {@link AbstractValueVisitor}.  Subclasses should override this unless
47       * they override all <code>visit</code> methods.
48       * <p>
49       * This implementation always throws {@link UnsupportedOperationException}.
50       *
51       * @param value the value to visit.
52       * @throws UnsupportedOperationException always thrown unless subclass
53       * overrides this implementation.
54       * @throws Exception subclasses can throw this; it will be propagated by
55       * the other <code>visit</code> methods.
56       */
57      protected void defaultVisit(IonValue value) throws Exception
58      {
59          throw new UnsupportedOperationException();
60      }
61  
62      public void visit(IonBlob value) throws Exception
63      {
64          defaultVisit(value);
65      }
66  
67      public void visit(IonBool value) throws Exception
68      {
69          defaultVisit(value);
70      }
71  
72      public void visit(IonClob value) throws Exception
73      {
74          defaultVisit(value);
75      }
76  
77      public void visit(IonDatagram value) throws Exception
78      {
79          defaultVisit(value);
80      }
81  
82      public void visit(IonDecimal value) throws Exception
83      {
84          defaultVisit(value);
85      }
86  
87      public void visit(IonFloat value) throws Exception
88      {
89          defaultVisit(value);
90      }
91  
92      public void visit(IonInt value) throws Exception
93      {
94          defaultVisit(value);
95      }
96  
97      public void visit(IonList value) throws Exception
98      {
99          defaultVisit(value);
100     }
101 
102     public void visit(IonNull value) throws Exception
103     {
104         defaultVisit(value);
105     }
106 
107     public void visit(IonSexp value) throws Exception
108     {
109         defaultVisit(value);
110     }
111 
112     public void visit(IonString value) throws Exception
113     {
114         defaultVisit(value);
115     }
116 
117     public void visit(IonStruct value) throws Exception
118     {
119         defaultVisit(value);
120     }
121 
122     public void visit(IonSymbol value) throws Exception
123     {
124         defaultVisit(value);
125     }
126 
127     public void visit(IonTimestamp value) throws Exception
128     {
129         defaultVisit(value);
130     }
131 }
