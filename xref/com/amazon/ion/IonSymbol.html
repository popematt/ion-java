16  package com.amazon.ion;
17  
18  /**
19   * An Ion <code>symbol</code> value.
20   * <p>
21   * <b>WARNING:</b> This interface should not be implemented or extended by
22   * code outside of this library.
23   */
24  public interface IonSymbol
25      extends IonText
26  {
27      /**
28       * Gets the text content of this symbol.
29       *
30       * @return the text of the symbol, or <code>null</code> if this is
31       * <code>null.symbol</code>.
32       *
33       * @throws UnknownSymbolException if this symbol has unknown text.
34       */
35      public String stringValue()
36          throws UnknownSymbolException;
37  
38  
39      /**
40       * Gets the integer symbol id used in the binary encoding of this symbol.
41       *
42       * @return an integer greater than zero, if this value has an associated
43       * symbol table.  Otherwise, return {@link SymbolTable#UNKNOWN_SYMBOL_ID}.
44       *
45       * @throws NullValueException if this is <code>null.symbol</code>.
46       *
47       * @deprecated Use {@link #symbolValue()} instead.
48       */
49      @Deprecated
50      public int getSymbolId()
51          throws NullValueException;
52  
53  
54      /**
55       * Returns this value as a symbol token (text + ID).
56       *
57       * @return null if {@link #isNullValue()}
58       *
59       */
60      public SymbolToken symbolValue();
61  
62  
63      /**
64       * Changes the value of this element.
65       *
66       * @param value the new value of this symbol;
67       * may be <code>null</code> to make this <code>null.symbol</code>.
68       *
69       */
70      public void setValue(String value);
71  
72      public IonSymbol clone()
73          throws UnknownSymbolException;
74  }
