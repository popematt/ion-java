16  package com.amazon.ion.system;
17  
18  import com.amazon.ion.IonCatalog;
19  import com.amazon.ion.IonSystem;
20  
21  /**
22   * The factory for creating {@link IonSystem}s.
23   * Most applications will only have one or two system instances;
24   * see {@link IonSystem} for important constraints.
25   * <p>
26   * Most long-lived applications will want to provide a custom
27   * {@link IonCatalog} implementation rather than using the default
28   * {@link SimpleCatalog}.
29   *
30   * @deprecated Use the more configurable {@link IonSystemBuilder} instead.
31   */
32  @Deprecated
33  public final class SystemFactory
34  {
35      /**
36       * Constructs a new system instance with a default configuration.
37       * <p>
38       * The catalog used by the new instance will be a {@link SimpleCatalog}
39       * with no initial entries, so please be aware of the limitations of that
40       * class.
41       *
42       * @return a new {@link IonSystem} instance; not null.
43       *
44       * @deprecated Use {@link IonSystemBuilder IonSystemBuilder.standard().build()} instead.
45       */
46      @Deprecated
47      public static IonSystem newSystem()
48      {
49          return IonSystemBuilder.standard().build();
50      }
51  
52      /**
53       * Constructs a new system instance with the given catalog.
54       *
55       * @param catalog the catalog to use in the new system.
56       *   If null, a new {@link SimpleCatalog} will be used.
57       * @return a new {@link IonSystem} instance; not null.
58       *
59       * @deprecated Use {@link IonSystemBuilder IonSystemBuilder.standard().withCatalog(catalog).build()} instead.
60       */
61      @Deprecated
62      public static IonSystem newSystem(IonCatalog catalog)
63      {
64          return IonSystemBuilder.standard().withCatalog(catalog).build();
65      }
66  }
