16  package com.amazon.ion;
17  
18  import java.util.Collection;
19  import java.util.Iterator;
20  
21  /**
22   * Common functionality of Ion <code>struct</code>, <code>list</code>, and
23   * <code>sexp</code> types.
24   * <p>
25   * <b>WARNING:</b> This interface should not be implemented or extended by
26   * code outside of this library.
27   */
28  public interface IonContainer
29      extends IonValue, Iterable<IonValue>
30  {
31  
32      /**
33       * Returns the number of elements in this container.
34       *
35       * @return the number of elements, zero if {@code this.isNullValue()}.
36       */
37      public int size();
38  
39      /**
40       * Creates an iterator providing the (direct) elements of this container.
41       * If this is an {@linkplain #isNullValue() Ion null value}, then this
42       * method returns an empty iterator.
43       * <p>
44       * Note that iteration over a {@link IonStruct} has unspecified ordering,
45       * while iteration over an {@link IonSequence} ({@link IonList} or
46       * {@link IonSexp}) must return elements
47       * in order of their appearance in the Ion representation.
48       *
49       * @return a new iterator, not {@code null}.
50       */
51      public Iterator<IonValue> iterator();
52  
53  
54      /**
55       * Removes the given element from this container.
56       * If this is an {@linkplain #isNullValue() Ion null value}, then this
57       * method returns {@code false}.
58       * <p>
59       * Note that, unlike {@link Collection#remove(Object)}, this method uses
60       * object identity, not {@link Object#equals(Object)}, to find the element.
61       * That is, the given instance is removed, not other similarly "equal"
62       * instances.
63       *
64       * @param element the element to be removed from this container,
65       * if present.
66       * @return <code>true</code> if this container contained the specified
67       * element.
68       * @throws NullPointerException if the <code>element</code>
69       * is <code>null</code>.
70       */
71      public boolean remove(IonValue element);
72  
73  
74      /**
75       * Checks if this container is empty.
76       *
77       * @return <code>true</code> if this container has no contents
78       * @throws NullValueException if this container is an Ion
79       * <code>null</code> value
80       */
81      public boolean isEmpty()
82          throws NullValueException;
83  
84      /**
85       * Clears the contents of this container (if any) and set it to empty.
86       * If this container is an Ion <code>null</code> value, set it to
87       * empty.
88       * <p>
89       * Use {@link #makeNull} to make this container null rather than empty.
90       * <p>
91       * <b>WARNING:</b> This method is not support for instances of
92       * {@link IonDatagram}.
93       */
94      public void clear();
95  
96      /**
97       * Sets the contents of this container to an Ion <code>null</code>
98       * value.
99       * <p>
100      * Use {@link #clear} to make this container empty rather than null.
101      */
102     public void makeNull();
103 
104 
105     public IonContainer clone()
106         throws UnknownSymbolException;
107 }
