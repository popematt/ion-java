16  package com.amazon.ion;
17  
18  /**
19   * An error caused by a request for an Ion version that is not supported by
20   * this implementation.
21   */
22  public class UnsupportedIonVersionException
23      extends IonException
24  {
25      private static final long serialVersionUID = -1166749371823975664L;
26  
27      private final String _unsupportedIonVersionId;
28  
29      public UnsupportedIonVersionException(String unsupportedIonVersionId)
30      {
31          _unsupportedIonVersionId = unsupportedIonVersionId;
32      }
33  
34      public String getUnsuportedIonVersionId()
35      {
36          return _unsupportedIonVersionId;
37      }
38  
39      @Override
40      public String getMessage()
41      {
42          return "Unsupported Ion version " + _unsupportedIonVersionId;
43      }
44  }
