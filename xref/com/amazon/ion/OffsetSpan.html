16  package com.amazon.ion;
17  
18  import com.amazon.ion.facet.Faceted;
19  import com.amazon.ion.facet.Facets;
20  import java.io.InputStream;
21  
22  /**
23   * Exposes the positions of a {@link Span} in the form of zero-based offsets
24   * within the source.  The "unit of measure" the offsets count depends on the
25   * source type: for byte arrays or {@link InputStream}s, the offsets count
26   * octets, but for {@link String}s or {@link java.io.Reader}s the offsets count
27   * UTF-16 code units.
28   * <p>
29   * <b>WARNING:</b> This interface should not be implemented or extended by
30   * code outside of this library.
31   * <p>
32   * As with all spans, positions lie <em>between</em> values, and when the start
33   * and finish positions are equal, the span is said to be <em>empty</em>.
34   * <p>
35   * To get one of these from a {@link Span}, use
36   * {@link Faceted#asFacet(Class) asFacet}{@code (OffsetSpan.class)} or one of
37   * the helpers from {@link Facets}.
38   *
39   */
40  public interface OffsetSpan
41  {
42      /**
43       * Returns this span's start position as a zero-based offset within the
44       * source.
45       */
46      public long getStartOffset();
47  
48      /**
49       * Returns this span's finish position as a zero-based offset within the
50       * source.  In some cases, the finish position is implicit and this method
51       * returns {@code -1}.  This includes most text sources, since in
52       * general (notably for containers) the finish offset can't be determined
53       * without significant effort to parse to the end of the value.
54       */
55      public long getFinishOffset();
56  }
