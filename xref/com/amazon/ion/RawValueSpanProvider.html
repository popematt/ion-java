16  package com.amazon.ion;
17  
18  /**
19   * Provide the ability to retrieve {@link Span}s (abstract value positions)
20   * of raw Ion values, excluding type and length octets.
21   * <p>
22   * <b>WARNING:</b> This interface should not be implemented or extended by
23   * code outside of this library.
24   * <p>
25   * This functionality may be accessed as a facet of binary {@link IonReader}s.
26   *
27   * @deprecated This is a private API subject to change without notice.
28   */
29  @Deprecated
30  public interface RawValueSpanProvider
31  {
32  
33      /**
34       * Constructs a Span, which may be faceted as an {@link OffsetSpan}, that
35       * provides the start and end byte positions of the current value.
36       * <p>
37       * NOTE: for Ion {@code int} values, users should not use these byte positions to
38       * determine which primitive (if any) the value can fit into, because the
39       * sign of Ion {@code int} values is encoded into the type ID byte, which is
40       * not included in this span.
41       * <p>
42       * <b>WARNING:</b> Spans provided by this method are not compatible with the
43       * {@link SeekableReader} facet, because they lack the type ID and length
44       * bytes that are important when reconstructing the context on seek. For
45       * {@link SpanProvider#currentSpan()} should be used to retrieve seekable
46       * Spans.
47       * @return the constructed Span
48       */
49      public Span valueSpan();
50  
51      /**
52       * @return the byte[] that backs this span. This span's start and end positions
53       * may be used as indices into the returned buffer. NOTE: does NOT perform
54       * a copy of the buffer; care must be taken not to mutate and corrupt the
55       * data.
56       */
57      public byte[] buffer();
58  
59  }
