146     public static synchronized void loadSystemProperties()
147     {
148         if (! ourSystemPropertiesLoaded)
149         {
150             InputStream stream =
151                 IonTestCase.class.getResourceAsStream("/system.properties");
152 
153             if (stream != null)
154             {
155                 Properties props = new Properties();
156 
157                 try
158                 {
159                     props.load(stream);
160 
161                     Iterator entries = props.entrySet().iterator();
162                     while (entries.hasNext())
163                     {
164                         Map.Entry entry = (Map.Entry) entries.next();
165 
166                         String key = (String) entry.getKey();
167 
168                         // Command-line values override system.properties
169                         if (System.getProperty(key) == null)
170                         {
171                             System.setProperty(key, (String) entry.getValue());
172                         }
173                     }
174                 }
175                 catch (IOException e)
176                 {
177                     synchronized (System.out)
178                     {
179                         System.out.println("Caught exception while loading system.properties:");
180                         e.printStackTrace(System.out);
181                     }
182                 }
183             }
184 
185             ourSystemPropertiesLoaded = true;
186         }
187     }

190     public static String requireSystemProperty(String prop)
191     {
192         loadSystemProperties();
193 
194         String value = System.getProperty(prop);
195         if (value == null)
196         {
197             value = System.getenv(prop);
198         }
199         if (value == null) {
200             String message = "Missing required system property: " + prop;
201             throw new IllegalStateException(message);
202         }
203         return value;
204     }

335      * For example, <code>makeEscapedCharString('n')</code> returns
336      * (Java) <code>"\"\\n\""</code>, equivalent to Ion <code>"\n"</code>.

368     /**
369      * Put value into a datagram, get bytes, and reload to single value.
370      * Result should be equivalent.
371      */
372     public IonValue reload(IonValue value)
373     {
374         IonDatagram dg = system().newDatagram(value);
375         byte[] bytes = dg.getBytes();
376         checkBinaryHeader(bytes);
377 
378         try {
379             dg = loader().load(bytes);
380         } catch (final IonException e) {
381             final String hex = BinaryTest.bytesToHex(bytes);
382             throw new IonException("Bad bytes: " + hex, e);
383         }
384         assertEquals(1, dg.size());
385         return dg.get(0);
386     }

435     /**
436      * Parses text as a single Ion value.  If the text contains more than that,
437      * a failure is thrown.
438      *
439      * @param iterator must not be <code>null</code>.
440      * @return a single value, or <code>null</code> if the scanner has nothing
441      * on it.
442      */
443     public IonValue singleValue(Iterator<IonValue> iterator)
444     {
445         IonValue value = null;
446 
447         if (iterator.hasNext())
448         {
449             value = iterator.next();
450 
451             if (iterator.hasNext())
452             {
453                 IonValue part = iterator.next();
454                 fail("Found unexpected part: " + part);
455             }
456         }
457 
458         return value;
459     }

463     /**
464      * Parses text as a single Ion value.  If the text contains more than that,
465      * a failure is thrown.
466      *
467      * @param text must not be <code>null</code>.
468      * @return a single value, not <code>null</code>.
469      */
470     public IonValue oneValue(String text)
471     {
472         IonValue value = null;
473 
474         Iterator<IonValue> iterator = system().iterate(text);
475         if (iterator.hasNext())
476         {
477             value = iterator.next();
478 
479             if (iterator.hasNext())
480             {
481                 IonValue part = iterator.next();
482                 fail("Found unexpected part <" + part + "> in text: " + text);
483             }
484         }
485         else
486         {
487             fail("No data found in text: " + text);
488         }
489 
490         return value;
491     }

563     public static void checkBinaryHeader(byte[] datagram)
564     {
565         if (datagram.length != 0) // Allow empty "binary" stream
566         {
567             assertTrue("datagram is too small", datagram.length >= 4);
568 
569             assertEquals("datagram cookie byte 1", 0xE0, datagram[0] & 0xff );
570             assertEquals("datagram cookie byte 2", 0x01, datagram[1] & 0xff);
571             assertEquals("datagram cookie byte 3", 0x00, datagram[2] & 0xff);
572             assertEquals("datagram cookie byte 4", 0xEA, datagram[3] & 0xff);
573         }
574     }

854         assert !dupe || text != null;
855 
856         String msg = "text:" + text + " sid:" + sid;
857 
858         if (text != null)
859         {
860             if (sid != UNKNOWN_SYMBOL_ID)
861             {
862                 assertEquals(msg, text, symtab.findKnownSymbol(sid));
863             }
864 
865             // Can't do this stuff when we have a duplicate symbol.
866             if (! dupe)
867             {
868                 assertEquals(msg, sid, symtab.findSymbol(text));
869 
870                 SymbolToken symToken = symtab.intern(text);
871                 assertEquals(msg, sid, symToken.getSid());
872 
873                 symToken = symtab.find(text);
874                 assertEquals(msg, sid, symToken.getSid());
875             }
876         }
877         else // No text expected, must have sid
878         {
879             assertEquals(msg, text /* null */, symtab.findKnownSymbol(sid));
880         }
881     }

983     public static SymbolTable findImportedTable(SymbolTable localTable,
984                                                 String importName)
985     {
986         SymbolTable[] imports = localTable.getImportedTables();
987         if (imports == null) return null;
988 
989         for (int i = 0; i < imports.length; i++)
990         {
991             SymbolTable current = imports[i];
992             // FIXME why does this allow null?
993             if (current != null && importName.equals(current.getName()))
994             {
995                 return current;
996             }
997         }
998 
999         return null;
1000     }
