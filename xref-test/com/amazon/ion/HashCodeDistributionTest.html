39  /** Chi-square test for {@link IonValue#hashCode()} implementations.
40   * Using selected test data, determine Chi^2 value for various hash table
41   * sizes and determine whether it is within 95% confidence interval using a
42   * goodness of fit test. The Chi-square test tests for the fact that hashes
43   * generated from the hash functions are random, with a 95% confidence.
44   * <p>
45   * See Knuth vol. 2 sec 3.3.1 for background.
46   */
47  @Ignore
48  public class HashCodeDistributionTest
49      extends IonTestCase
50  {
51      /**
52       * This value is the initial capacity of the data set (i.e. number of
53       * buckets). This must be a power of 2.
54       * <p>
55       * We want to keep halving the number of buckets in the data set to test on
56       * different sizes of buckets. The halving of the buckets is performed at
57       * {@link #collapse(int[])}.
58       * <p>
59       * The rationale behind this is that we want to mimic the internal
60       * implementation of how {@link HashMap} and {@link ConcurrentHashMap}
61       * stores its hash table, which is an {@link Entry} array with size that is
62       * a power of 2.
63       */
64      private static final int DATA_SET_INITIAL_CAPACITY = 1 << 12;
65  
66      /**
67       * This is sync'ed with {@link HashMap}'s DEFAULT_INITIAL_CAPACITY
68       * constant, which is 16.
69       */
70      private static final int HASH_MAP_INITIAL_CAPACITY = 1 << 4;
71  
72      private static final String BULK_TEST_DATA_PATH = "items-A";
73  
74      protected Set<IonValue> loadBulkDataSet()
75          throws IOException
76      {
77          Set<IonValue> dataSet = new HashSet<IonValue>();
78          File directory = getBulkTestdataFile(BULK_TEST_DATA_PATH);
79  
80          File[] testFiles = directory.listFiles(new FilenameFilter()
81          {
82              public boolean accept(File dir, String name)
83              {
84                  return name.endsWith(".ion") || name.endsWith(".10n");
85              }
86          });
87  
88          IonSystem ionSystem = system();
89          for (File testFile : testFiles)
90          {
91              InputStream in =
92                  new BufferedInputStream(new FileInputStream(testFile));
93              try
94              {
95                  Iterator<IonValue> i = ionSystem.iterate(in);
96                  while (i.hasNext())
97                  {
98                      loadValue(i.next(), dataSet);
99                  }
100             }
101             finally
102             {
103                 in.close();
104             }
105         }
106 
107         return dataSet;
108     }

110     /**
111      * Loads the {@link IonValue} and its nested values into the {@code dataSet}.
112      *
113      * @param value
114      * @param dataSet
115      */
116     protected void loadValue(IonValue value, Set<IonValue> dataSet)
117     {
118         dataSet.add(value);
119         if (value instanceof IonContainer)
120         {
121             IonContainer container = (IonContainer) value;
122             for (IonValue v : container)
123             {
124                 loadValue(v, dataSet);
125             }
126         }
127     }
128 
129     /**
130      * Analyze key entries, {@code counts}, against the Chi^2 distribution as
131      * provided, while collapsing {@code counts} down to {@code limit} from the
132      * initial size.
133      *
134      * @param counts key entries int array
135      * @param totalCount total number of key entries
136      * @param limit limit for {@link #collapse(int[])}
137      */
138     protected void analyzeDistribution(int[] counts, int totalCount, int limit)
139     {
140         while (counts.length >= limit)
141         {
142             checkChiSquareValue(counts, totalCount);
143             counts = collapse(counts);
144         }
145     }
146 
147     /**
148      * Analyze counts against Chi^2 distribution at 95% confidence interval.
149      *
150      * @param counts key entries int array
151      */
152     protected void checkChiSquareValue(int[] counts, int totalCount)
153     {
154         int expectedTotalCount = 0;
155         for (int count : counts)
156         {
157             expectedTotalCount += count;
158         }
159 
160         assertEquals("Total counts is incorrect", totalCount, expectedTotalCount);
161 
162         if (expectedTotalCount == 0)
163         {
164             return; // nothing to test
165         }
166 
167         double expectedValue = (double) totalCount / (double) counts.length;
168         double chiSquared = 0.0;
169         for (int count : counts)
170         {
171             chiSquared += Math.pow((count - expectedValue), 2) / expectedValue;
172         }
173         int degreesOfFreedom = counts.length - 1;
174 
175         double threshold;
176         if (degreesOfFreedom < upperTailChiSquared95Table.length)
177         {
178             threshold = upperTailChiSquared95Table[degreesOfFreedom];
179         }
180         else
181         {
182             // Calculate critical chi-square value using formula for degrees of
183             // freedom greater than 100. Refer to Knuth vol. 2 sec 3.3.1
184             double x_p = 1.64;
185             threshold = degreesOfFreedom
186                         + Math.sqrt(2*degreesOfFreedom)*x_p
187                         + (2.0/3.0)*Math.pow(x_p, 2)
188                         - (2.0/3.0);
189         }
190 
191         if (chiSquared >= threshold)
192         {
193             fail(String.format("Chi-square value above 95 percent " +
194                                "threshold: %1$f, limit: %2$f, nu: %3$d",
195                                chiSquared, threshold, degreesOfFreedom));
196         }
197     }
198 
199     /**
200      * Collapse key entries int array and return a new int array.
201      *
202      * @param counts
203      * @return the new key entries int array
204      */
205     protected int[] collapse(int[] counts)
206     {
207         int[] new_counts = new int[counts.length >> 1]; // halve the number of buckets
208         for (int i = 0; i < new_counts.length; ++i)
209         {
210             new_counts[i] = counts[i] + counts[i + new_counts.length];
211         }
212         return new_counts;
213     }

215     /**
216      * Duplicate protected method {@link HashMap#hash(int)}.  Formula from
217      * decompiled class in JDK 1.6.0.
218      * @param raw_hash Raw hash value
219      * @return cooked hash value
220      */
221     protected int hashMapHash(int raw_hash)
222     {
223         int cooked_hash = raw_hash;
224         cooked_hash ^= (cooked_hash >>> 20) ^ (cooked_hash >>> 12);
225         return cooked_hash ^ (cooked_hash >>> 7) ^ (cooked_hash >>> 4);
226     }

228     /**
229      * Counts each occurrence of {@link IonValue#hashCode()} and perform
230      * Chi^2 analysis over it.
231      *
232      * @param dataSet
233      *          the data set containing IonValue test data
234      * @param type
235      *          filter over IonValue test data; all IonValues are used if
236      *          {@code null}
237      * @param adjusted
238      *          {@code true} if using {@link #hashMapHash(int)} as a
239      *          supplementary hash function
240      * @param limit
241      *          the limit to {@link #collapse(int[])}
242      */
243     protected void checkChiSquareIonValueHashCode(Set<IonValue> dataSet,
244                                                   IonType type,
245                                                   boolean adjusted,
246                                                   int limit)
247     {
248         int[] counts = new int[DATA_SET_INITIAL_CAPACITY]; // must be power of 2
249         int countsLength = counts.length;
250         int totalCount = 0;
251         for (IonValue v : dataSet)
252         {
253             if (type == null || v.getType().equals(type))
254             {
255                 int index = adjusted ?
256                             hashMapHash(v.hashCode()) % countsLength :
257                             v.hashCode() % countsLength;
258                 if (index < 0)
259                 {
260                     index += countsLength;
261                 }
262                 counts[index]++;
263                 totalCount++;
264             }
265         }
266         analyzeDistribution(counts, totalCount, limit);
267     }
