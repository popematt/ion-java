16  package com.amazon.ion.streaming;
17  
18  import static com.amazon.ion.TestUtils.GLOBAL_SKIP_LIST;
19  import static com.amazon.ion.TestUtils.GOOD_IONTESTS_FILES;
20  import static com.amazon.ion.TestUtils.testdataFiles;
21  import static com.amazon.ion.junit.IonAssert.assertIonEquals;
22  
23  import com.amazon.ion.IonDatagram;
24  import com.amazon.ion.IonReader;
25  import com.amazon.ion.IonTestCase;
26  import com.amazon.ion.IonValue;
27  import com.amazon.ion.junit.Injected.Inject;
28  import java.io.File;
29  import java.io.IOException;
30  import org.junit.Test;
31  
32  public class ReaderDomCopyTest
33  extends IonTestCase
34  {
35      @Inject("testFile")
36      public static final File[] FILES =
37          testdataFiles(GLOBAL_SKIP_LIST, GOOD_IONTESTS_FILES);
38  
39      private File myTestFile;
40  
41      public void setTestFile(File file)
42      {
43          myTestFile = file;
44      }
45  
46  
47  
48      @Test
49      public void test()
50      throws Exception
51      {
52          iterateIon(myTestFile);
53      }
54  
55      void iterateIon(File myTestFile)
56      throws IOException
57      {
58          IonDatagram dg = system().getLoader().load(myTestFile);
59  
60          IonReader reader = system().newReader(dg);
61          for (int i = 0; reader.next() != null; i++)
62          {
63              IonValue actual = system().newValue(reader);
64              assertIonEquals(dg.get(i), actual);
65          }
66      }
67  }
