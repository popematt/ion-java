30  public class ReaderCompare
31      extends Assert

33      public static void compare(IonReader it1, IonReader it2) {
34          while (hasNext(it1, it2)) {
35              IonType t1 = (READER_HASNEXT_REMOVED ? it1.getType() : it1.next());
36              IonType t2 = (READER_HASNEXT_REMOVED ? it2.getType() : it2.next());
37  
38              if ((t1 != t2) && (t1 == null || t2 == null || !t1.equals(t2))) {
39                  assertEquals("ion type", t1, t2);
40              }
41              if (t1 == null) break;
42  
43              if (it1.isInStruct()) {
44                  compareFieldNames(it1, it2);
45              }
46              compareAnnotations(it1, it2);
47  
48              boolean isNull = it1.isNullValue();
49              assertEquals(isNull, it2.isNullValue());
50  
51              switch (t1) {
52                  case NULL:
53                      assertTrue(it1.isNullValue());
54                      assertTrue(it2.isNullValue());
55                      break;
56                  case BOOL:
57                  case INT:
58                  case FLOAT:
59                  case DECIMAL:
60                  case TIMESTAMP:
61                  case STRING:
62                  case SYMBOL:
63                  case BLOB:
64                  case CLOB:
65                      compareScalars(t1, isNull, it1, it2);
66                      break;
67                  case STRUCT:
68                  case LIST:
69                  case SEXP:
70                      it1.stepIn();
71                      it2.stepIn();
72                      compare(it1, it2);
73                      it1.stepOut();
74                      it2.stepOut();
75                      break;
76  
77                  default:
78                      throw new IllegalStateException();
79              }
80          }
81          assertFalse(hasNext(it1, it2));
82      }

85      public static boolean hasNext(IonReader it1, IonReader it2)
86      {
87          boolean more;
88          if (READER_HASNEXT_REMOVED)
89          {
90              more = (it1.next() != null);
91              assertEquals("next results don't match", more, it2.next() != null);
92          }
93          else
94          {
95              more = it1.hasNext();
96              assertEquals("hasNext results don't match", more, it2.hasNext());
97  
98              // Check that result doesn't change
99              assertEquals(more, it1.hasNext());
100             assertEquals(more, it2.hasNext());
101         }
102 
103         if (!more) {
104             assertEquals(null, it1.next());
105             assertEquals(null, it2.next());
106         }
107 
108         return more;
109     }

111     public static void compareNonNullStrings(String what, String s1, String s2)
112     {
113         assertNotNull(what, s1);
114         assertNotNull(what, s2);
115         assertEquals(what, s1, s2);
116     }

118     public static void compareFieldNames(IonReader r1, IonReader r2) {
119         SymbolToken tok1 = r1.getFieldNameSymbol();
120         SymbolToken tok2 = r1.getFieldNameSymbol();
121         String fn = tok1.getText();
122         assertEquals(fn, tok2.getText());
123 
124         if (fn != null) {
125             String f1 = r1.getFieldName();
126             String f2 = r2.getFieldName();
127             compareNonNullStrings("field name", fn, f1);
128             compareNonNullStrings("field name", fn, f2);
129         }
130     }

132     public static void compareAnnotations(IonReader it1, IonReader it2) {
133         SymbolToken[] syms1 = it1.getTypeAnnotationSymbols();
134         SymbolToken[] syms2 = it2.getTypeAnnotationSymbols();
135 
136         IonAssert.assertSymbolEquals("annotation", syms1, syms2);
137     }

139     public static void compareScalars(IonType t, boolean isNull,
140                                       IonReader it1, IonReader it2) {
141         switch (t) {
142             case BOOL:
143                 if (!isNull) assertEquals(it1.booleanValue(), it2.booleanValue());
144                 break;
145             case INT:
146                 assertEquals(it1.bigIntegerValue(), it2.bigIntegerValue());
147                 break;
148             case FLOAT:
149             {
150                 if (! isNull)
151                 {
152                     double v1 = it1.doubleValue();
153                     double v2 = it2.doubleValue();
154                     assertEquals(v1, v2, 0);
155                     // The last param is a delta, and we want exact match.
156                 }
157                 break;
158             }
159             case DECIMAL:
160             {
161                 BigDecimal dec1 = it1.bigDecimalValue();
162                 BigDecimal dec2 = it2.bigDecimalValue();
163                 IonTestCase.assertPreciselyEquals(dec1, dec2);
164                 dec1 = it1.decimalValue();
165                 dec2 = it2.decimalValue();
166                 IonTestCase.assertPreciselyEquals(dec1, dec2);
167                 // TODO also test double, long, int etc.
168                 break;
169             }
170             case TIMESTAMP:
171             {
172                 Timestamp t1 = it1.timestampValue();
173                 Timestamp t2 = it2.timestampValue();
174                 assertEquals(t1, t2);
175                 break;
176             }
177             case STRING:
178             {
179                 String s1 = it1.stringValue();
180                 String s2 = it2.stringValue();
181                 assertEquals(s1, s2);
182                 break;
183             }
184             case SYMBOL:
185             {
186                 SymbolToken tok1 = it1.symbolValue();
187                 SymbolToken tok2 = it2.symbolValue();
188                 if (isNull)
189                 {
190                     assertNull(tok1);
191                     assertNull(tok2);
192                 }
193                 else if (tok1.getText() == null || tok2.getText() == null)
194                 {
195                     assertEquals("sids", tok1.getSid(), tok2.getSid());
196                 }
197                 else
198                 {
199                     String s1 = tok1.getText();
200                     String s2 = tok2.getText();
201                     assertEquals(s1, s2);
202                 }
203                 break;
204             }
205             case BLOB:
206             case CLOB:
207             {
208                 if (!isNull)
209                 {
210                     byte[] b1 = it1.newBytes();
211                     byte[] b2 = it2.newBytes();
212                     assert b1 != null && b2 != null;
213                     assert b1.length == b2.length;
214                     for (int ii=0; ii<b1.length; ii++) {
215                         byte v1 = b1[ii];
216                         byte v2 = b2[ii];
217                         assertEquals(v1, v2);
218                     }
219                 }
220                 break;
221             }
222             default:
223                 throw new IllegalStateException("iterated to a type that's not expected");
224         }
225     }
