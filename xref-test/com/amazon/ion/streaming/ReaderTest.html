358     private void skipThroughTopLevelContainer(String data)
359     {
360         read(data);
361 
362         in.next();
363         in.stepIn();
364         {
365             while (in.next() != null)
366             {
367                 // Just skip the children
368             }
369             expectEof();
370         }
371         in.stepOut();
372         expectTopEof();
373     }

376     private String[] LOB_DATA = {
377           "{{\"\"}}",
378           "{{\"clob\"}}",
379 
380           "{{''''''}}",
381           "{{''' '''}}",
382           "{{'''clob'''}}",
383           "{{'''c}ob'''}}",
384           "{{'''c}}b'''}}",
385           "{{'''c\\'''ob'''}}",
386 
387           "{{}}",
388           "{{Zm9v}}",
389     };
390 
391     private void testSkippingLob(String containerPrefix,
392                                  String containerSuffix)
393     {
394         for (String lob : LOB_DATA)
395         {
396             String data = containerPrefix + lob + containerSuffix;
397             skipThroughTopLevelContainer(data);
398 
399             data = containerPrefix + " " + lob + containerSuffix;
400             skipThroughTopLevelContainer(data);
401 
402             data = containerPrefix + " " + lob + " " + containerSuffix;
403             skipThroughTopLevelContainer(data);
404 
405             data = containerPrefix + lob + " " + containerSuffix;
406             skipThroughTopLevelContainer(data);
407         }
408     }

427     @Test
428     public void testReadLobUsingGetBytes() throws Exception
429     {
430         if (!myReaderMaker.sourceIsBinary()) {
431             // TODO text implements getBytes() differently: https://github.com/amzn/ion-java/issues/319
432             return;
433         }
434         String data = "{{\"abcdefghijklmnopqrstuvwxyz\"}}";
435         byte[] partialBuffer = new byte[10];
436         read(data);
437         assertEquals(IonType.CLOB, in.next());
438         int remaining = in.byteSize();
439         assertEquals(26, remaining);
440         StringBuilder value = new StringBuilder();
441         while (remaining > 0) {
442             int bytesRead = in.getBytes(partialBuffer, 0, partialBuffer.length);
443             remaining -= bytesRead;
444             value.append(new String(partialBuffer, 0, bytesRead, "UTF-8"));
445         }
446         assertEquals("abcdefghijklmnopqrstuvwxyz", value.toString());
447     }
