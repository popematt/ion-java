28      private enum IntRadix {
29          DECIMAL
30          {
31              @Override
32              String getString(BigInteger integer)
33              {
34                  return integer.toString();
35              }
36          },
37          HEX
38          {
39              @Override
40              String getString(BigInteger integer)
41              {
42                  return injectRadixPrefix("0x", integer.toString(16));
43              }
44          },
45          HEX_UPPER
46          {
47              @Override
48              String getString(BigInteger integer)
49              {
50                  return injectRadixPrefix("0x", integer.toString(16).toUpperCase());
51              }
52          },
53          BINARY
54          {
55              @Override
56              String getString(BigInteger integer)
57              {
58                  return injectRadixPrefix("0b", integer.toString(2));
59              }
60          };
61  
62          abstract String getString(BigInteger integer);
63  
64          private static String injectRadixPrefix(String radixPrefix, String value)
65          {
66              if(value.charAt(0) == '-')
67              {
68                  value = "-" + radixPrefix + value.replaceFirst("-", "");
69              }
70              else
71              {
72                  value = radixPrefix + value;
73              }
74              return value;
75          }
76      }

142     private void testGetIntegerSizeIntBoundary(int boundaryValue, long pastBoundary)
143     {
144         in.next();
145         // It's fine if IntegerSize recommends a larger-than-necessary type, just not a smaller one.
146         assertTrue(IntegerSize.INT == in.getIntegerSize() || IntegerSize.LONG == in.getIntegerSize());
147         assertEquals(boundaryValue, in.intValue());
148         // assert nothing changes until next()
149         assertTrue(IntegerSize.INT == in.getIntegerSize() || IntegerSize.LONG == in.getIntegerSize());
150         in.next();
151         assertEquals(IntegerSize.LONG, in.getIntegerSize());
152         assertEquals(pastBoundary, in.longValue());
153         assertEquals(IntegerSize.LONG, in.getIntegerSize());
154     }
155 
156     private void testGetIntegerSizeLongBoundary(long boundaryValue, BigInteger pastBoundary)
157     {
158         in.next();
159         // It's fine if IntegerSize recommends a larger-than-necessary type, just not a smaller one.
160         assertTrue(IntegerSize.LONG == in.getIntegerSize() || IntegerSize.BIG_INTEGER == in.getIntegerSize());
161         assertEquals(boundaryValue, in.longValue());
162         assertTrue(IntegerSize.LONG == in.getIntegerSize() || IntegerSize.BIG_INTEGER == in.getIntegerSize());
163         in.next();
164         assertEquals(IntegerSize.BIG_INTEGER, in.getIntegerSize());
165         assertEquals(pastBoundary, in.bigIntegerValue());
166         assertEquals(IntegerSize.BIG_INTEGER, in.getIntegerSize());
167     }
168 
169     private BigInteger loadBoundaries(long boundaryValue, IntRadix intRadix)
170     {
171         BigInteger boundary = BigInteger.valueOf(boundaryValue);
172         BigInteger pastBoundary = (boundaryValue < 0) ? boundary.subtract(BigInteger.ONE) : boundary.add(BigInteger.ONE);
173         String ionText = intRadix.getString(boundary) + " " + intRadix.getString(pastBoundary);
174         read(ionText);
175         return pastBoundary;
176     }
