28      public static byte[] hexToBytes(final String hex)
29      {
30          String[] hexChunks = hex.split("\\s+");
31          byte[] data = new byte[hexChunks.length];
32          for (int i = 0; i < hexChunks.length; i++)
33          {
34              final String hexChunk = hexChunks[i];
35              final int ordinal = Integer.parseInt(hexChunk, 16);
36              if ((ordinal & 0xFFFFFF00) != 0)
37              {
38                  throw new IllegalArgumentException("Bad Chunk: " + hexChunk);
39              }
40              // this is safe because we only have the low bytes set
41              data[i] = (byte) ordinal;
42          }
43          return data;
44      }
45  
46      public static String bytesToHex(final byte[] bytes)
47      {
48          StringBuilder builder = new StringBuilder();
49          for (int i = 0; i < bytes.length; i++)
50          {
51              byte octet = bytes[i];
52              int ordinal = octet & 0xFF;
53              builder.append(String.format("%02X", ordinal));
54              if ((i + 1) < bytes.length)
55              {
56                  builder.append(' ');
57              }
58          }
59          return builder.toString();
60      }

72      /**
73       * Loads a string literal as Ion bytes no cookie required.
74       * We encode the string literal as hex digits as it is convenient to read.
75       *
76       * @param   hex     A simple space delimited hex encoding.
77       */
78      private IonValue ion(final String hex)
79      {
80          byte[] data = hexToBytes(MAGIC_COOKIE + hex);
81          return system().singleValue(data);
82      }
83  
84      /** Converts a single value to bytes using an empty datagram */
85      private byte[] ionBytes(IonValue val)
86      {
87          IonDatagram dg = system().newDatagram(val);
88          return dg.getBytes();
89      }
90  
91      /** Dumps byte arrays to hex strings */
92      private static String dump(byte[]... packets)
93      {
94          StringBuilder builder = new StringBuilder();
95          for (int i = 0; i < packets.length; i++)
96          {
97              byte[] packet = packets[i];
98              builder.append('"');
99              builder.append(bytesToHex(packet));
100             builder.append('"');
101 
102             if ((i + 1) < packets.length)
103             {
104                 builder.append('\n');
105             }
106         }
107         return builder.toString();
108     }

110     private static void assertDoubleBits(IonValue val, long bits)
111     {
112         assertTrue(val instanceof IonFloat);
113         IonFloat fval = (IonFloat) val;
114         long test = Double.doubleToRawLongBits(fval.doubleValue());
115         assertTrue(String.format("%X\n%X", test, bits), test == bits);
116     }

200     {
