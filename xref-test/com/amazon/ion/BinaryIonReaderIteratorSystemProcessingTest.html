5   public class BinaryIonReaderIteratorSystemProcessingTest
6       extends IteratorSystemProcessingTestCase
7   {
8       private byte[] myBytes;
9   
10      @Override
11      protected void prepare(String text)
12      {
13          myMissingSymbolTokensHaveText = false;
14          myBytes = encode(text);
15      }
16  
17      @Override
18      protected Iterator<IonValue> iterate()
19      {
20          return system().iterate(getStreamingMode().newIonReader(system().getCatalog(), myBytes));
21      }
22  
23      @Override
24      protected Iterator<IonValue> systemIterate()
25      {
26          return system().systemIterate(system().newSystemReader(myBytes));
27      }
28  
29      @Override
30      protected int expectedLocalNullSlotSymbolId() {
31          // The spec allows for implementations to treat these malformed symbols as "null slots", and all "null slots"
32          // in local symbol tables as equivalent to symbol zero.
33          return getStreamingMode() == StreamingMode.NEW_STREAMING_INCREMENTAL ? 0 : 10;
34      }
