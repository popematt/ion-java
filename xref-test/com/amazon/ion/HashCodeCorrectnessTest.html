55      /**
56       * Verify that the relationship between equals() and hashCode() is
57       * honored. That is, if {@link Object#equals(Object)}, then their
58       * {@link Object#hashCode()}s return the same result.
59       *
60       * @param v1 IonValue 1
61       * @param v2 IonValue 2
62       */
63      private static void assertIonEqImpliesHashEq(final IonValue v1,
64                                                   final IonValue v2)
65      {
66          // We only care about equal IonValues
67          if (v1.equals(v2))
68          {
69              int hashCode1 = v1.hashCode();
70              int hashCode2 = v2.hashCode();
71  
72              if (hashCode1 != hashCode2)
73              {
74                  fail(String.format("Equal IonValues must have same "
75                      + "hash codes:\n"
76                      + " v1 %1$s (hash %2$x)\n"
77                      + " v2 %3$s (hash %4$x)",
78                      v1, hashCode1,
79                      v2, hashCode2));
80              }
81          }
82      }

84      /**
85       * Test method for {@link IonLob#hashCode()}
86       */
87      @Test
88      public void testIonLobHashCode()
89      {
90          // Some clobs
91          IonValue clob1 = oneValue("{{'''The quick brown fox '''\n"
92                                    + "'''jumped over the lazy dog.'''}}");
93          IonValue clob2 = oneValue("{{'''Jackdaws love my '''\n"
94                                    + "'''big sphinx of quartz.'''}}");
95          IonValue clob3 = oneValue("{{\"The quick brown fox "
96                               + "jumped over the lazy dog.\"}}");
97          IonValue clob4 = oneValue("{{\"Jackdaws love my "
98                               + "big sphinx of quartz.\"}}");
99          assertIonEqImpliesHashEq(clob1, clob1); // equal
100         assertIonEqImpliesHashEq(clob1, clob2); // not eq
101         assertIonEqImpliesHashEq(clob1, clob3); // equal
102         assertIonEqImpliesHashEq(clob1, clob4); // not eq
103         assertIonEqImpliesHashEq(clob2, clob2); // equal
104         assertIonEqImpliesHashEq(clob2, clob3); // not eq
105         assertIonEqImpliesHashEq(clob2, clob4); // equal
106         assertIonEqImpliesHashEq(clob3, clob3); // equal
107         assertIonEqImpliesHashEq(clob3, clob4); // not eq
108         assertIonEqImpliesHashEq(clob4, clob4); // equal
109         // Some blobs
110         IonValue blob1 = oneValue("{{ VGhlIHF1aWNrIGJyb3duIGZveCBqd"
111                                   + "W1wZWQgb3ZlciB0aGUgbGF6eSBkb2cu}}");
112         IonValue blob2 = oneValue("{{ SmFja2Rhd3MgbG92ZSBteSBiaWcgc"
113                                   + "3BoaW54IG9mIHF1YXJ0ei4=}}");
114         IonValue blob3 = oneValue("{{ VGhlIHF1aWNrIGJyb3duIGZveCBqd\n"
115                                   + " W1wZWQgb3ZlciB0aGUgbGF6eSBkb2cu}}");
116         IonValue blob4 = oneValue("{{ SmFja2Rhd3MgbG92ZSBteSBiaWcgc\n"
117                                   + " 3BoaW54IG9mIHF1YXJ0ei4=}}");
118         assertIonEqImpliesHashEq(blob1, blob1);
119         assertIonEqImpliesHashEq(blob1, blob2);
120         assertIonEqImpliesHashEq(blob1, blob3);
121         assertIonEqImpliesHashEq(blob1, blob4);
122         assertIonEqImpliesHashEq(blob2, blob2);
123         assertIonEqImpliesHashEq(blob2, blob3);
124         assertIonEqImpliesHashEq(blob2, blob4);
125         assertIonEqImpliesHashEq(blob3, blob3);
126         assertIonEqImpliesHashEq(blob3, blob4);
127         assertIonEqImpliesHashEq(blob4, blob4);
128         // And a mashup
129         assertIonEqImpliesHashEq(blob1, clob1);
130         assertIonEqImpliesHashEq(blob1, clob2);
131         assertIonEqImpliesHashEq(clob1, blob3);
132         assertIonEqImpliesHashEq(clob1, blob4);
133         assertIonEqImpliesHashEq(blob2, clob2);
134         assertIonEqImpliesHashEq(clob2, blob3);
135         assertIonEqImpliesHashEq(blob2, clob4);
136         assertIonEqImpliesHashEq(blob3, clob3);
137         assertIonEqImpliesHashEq(blob3, clob4);
138         assertIonEqImpliesHashEq(clob4, blob4);
139     }

145     @Test
146     public void testIonBoolHashCode()
147     {
148         IonValue[] bools = {oneValue("true"), oneValue("true"),
149                             oneValue("false"), oneValue("false")};
150         for (int i = 0; i < bools.length; ++i)  {
151             for (int j = i; j < bools.length; ++j)  {
152                 assertIonEqImpliesHashEq(bools[i], bools[j]);
153             }
154         }
155 
156         IonValue trueIonBool = oneValue("true");
157         IonValue falseIonBool = oneValue("false");
158 
159         assertTrue(trueIonBool.hashCode() != Boolean.TRUE.hashCode());
160         assertTrue(falseIonBool.hashCode() != Boolean.FALSE.hashCode());
161         assertTrue(trueIonBool.hashCode() != falseIonBool.hashCode());
162     }

164     /**
165      * Test method for {@link IonInt#hashCode()}.
166      */
167     @Test
168     public void testIonIntHashCode()
169     {
170         Set<IonInt> unique_values = new HashSet<IonInt>();
171         Set<Integer> unique_hashes = new HashSet<Integer>();
172         for (int i = -100 ; i < 100; i += 7)  {
173             IonInt vi1 = integer(i);
174             IonInt vi2 = integer(i);
175             unique_values.add(vi1);
176             unique_values.add(vi2);
177             unique_hashes.add(vi1.hashCode());
178             unique_hashes.add(vi2.hashCode());
179             for (int j = -100;  j < 100; j += 5) {
180                 IonInt vj1 = integer(j);
181                 IonInt vj2 = integer(j);
182                 unique_values.add(vj1);
183                 unique_values.add(vj2);
184                 unique_hashes.add(vj1.hashCode());
185                 unique_hashes.add(vj2.hashCode());
186                 assertIonEqImpliesHashEq(vi1, vi1);
187                 assertIonEqImpliesHashEq(vi1, vi2);
188                 assertIonEqImpliesHashEq(vi1, vj1);
189                 assertIonEqImpliesHashEq(vi1, vj2);
190                 assertIonEqImpliesHashEq(vi2, vi2);
191                 assertIonEqImpliesHashEq(vi2, vj1);
192                 assertIonEqImpliesHashEq(vi2, vj2);
193                 assertIonEqImpliesHashEq(vj1, vj1);
194                 assertIonEqImpliesHashEq(vj1, vj2);
195                 assertIonEqImpliesHashEq(vj2, vj2);
196             }
197         }
198         assertEquals("All distinct values have distinct hashes",
199                      unique_values.size(),
200                      unique_hashes.size());
201     }

498      * Checks that two unequal IonValue's hash codes are distinct integers.
499      * Note that it is NOT required that two unequal objects must return
500      * distinct hash code results. However, we are enforcing this here to
501      * improve the performance of hash tables.
502      * <p>
503      * Refer to Effective Java Ed2 (Joshua Bloch) Item 9.
504      */
505     private void assertIonNotEqImpliesHashNotEq(final String s1,
506                                                 final String s2)
507     {
508         final IonValue v1 = oneValue(s1);
509         final IonValue v2 = oneValue(s2);
510 
511         assertTrue("v1 should be not equal to v2", !v1.equals(v2));
512 
513         int hashCode1 = v1.hashCode();
514         int hashCode2 = v2.hashCode();
515 
516         if (hashCode1 == hashCode2)
517         {
518             fail(String.format("Unequal IonValues should have distinct "
519                 + "hash codes:\n"
520                 + " v1 %1$s (hash %2$x)\n"
521                 + " v2 %3$s (hash %4$x)",
522                 v1, hashCode1,
523                 v2, hashCode2));
524         }
525     }
