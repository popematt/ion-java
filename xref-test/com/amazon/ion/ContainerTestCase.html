53      public void checkNullContainer(IonContainer value)
54      {
55          assertTrue(value.isNullValue());
56          assertFalse(value.iterator().hasNext());
57          assertEquals(0, value.size());
58  
59          try
60          {
61              value.remove(null);
62              fail("Expected NullPointerException");
63          }
64          catch (NullPointerException e) { }
65  
66          // Remove from null container is no-op.
67          assertFalse(value.remove(system().newNull()));
68  
69          try
70          {
71              value.isEmpty();
72              fail("Expected NullValueException");
73          }
74          catch (NullValueException e) { }
75      }

172     /**
173      * Using an iterator, remove a value in the container and verify
174      * that the value no longer exists in the container.
175      * @param container the container to test
176      */
177     public void testIteratorRemove(IonContainer container)
178     {
179         Iterator<IonValue> it = container.iterator();
180         IonValue removedValue = it.next();
181         int prevSize = container.size();
182         it.remove();
183 
184         // make sure we removed something
185         assertEquals(container.size(), prevSize - 1);
186         assertNull(removedValue.getContainer());
187         assertNull(removedValue.getFieldName());
188         // TODO check that index is not set.
189 
190         // make sure the value we removed isn't there anymore
191         it = container.iterator();
192         while (it.hasNext())
193         {
194             IonValue value = it.next();
195             assertNotSame(removedValue, value);
196         }
197 
198         // Another call to remove should be no-op
199         assertFalse(container.remove(removedValue));
200 
201         // TODO concurrency tests.  it.remove should throw after (eg) container.makeNull().
202         // TODO test proper behavior of remove() when next() not called, or when remove() called twice.
203         // (should throw IllegalStateException)
204     }

225     {
226         IonValue n = system().newNull();
227         assertFalse(n.removeFromContainer());
228 
229         IonValue m = system().newInt(2);

234         assertEquals(null, n.getFieldName());

257 Iterator<IonValue> it = dg.systemIterator();

259 
260         IonContainer list = (IonContainer) dg.get(0);

262 
263         SymbolTable topSymtab = list.getSymbolTable();
264         checkLocalTable(topSymtab);
265         assertNotNull(topSymtab);
266 
267         {
268             IonValue child = list.iterator().next();
269             assertSame(topSymtab, child.getSymbolTable());
270

272             assertEquals("sym1", child.toString());
273             assertNotSame(topSymtab, child.getSymbolTable());
274         }
275         {
276             IonList child = (IonList) list.iterator().next();
277             assertSame(topSymtab, child.getSymbolTable());
278 
279             list.remove(child);
280             assertEquals("[sym2]", child.toString());
281             assertNotSame(topSymtab, child.getSymbolTable());
282             assertNotSame(topSymtab, child.get(0).getSymbolTable());
283         }
284         {
285             IonStruct child = (IonStruct) list.iterator().next();
286             assertSame(topSymtab, child.getSymbolTable());
287 
288             list.remove(child);
289             assertEquals("{f:sym3}", child.toString());
290             assertNotSame(topSymtab, child.getSymbolTable());
291             assertNotSame(topSymtab, child.get("f").getSymbolTable());
292         }
293         {
294             IonValue child = list.iterator().next();
295             assertSame(topSymtab, child.getSymbolTable());
296 
297             list.remove(child);
298             assertEquals("a::3", child.toString());
299             assertNotSame(topSymtab, child.getSymbolTable());

389     private void checkClones(IonContainer c, IonSymbol child)
390     {
391         IonContainer clone = c.clone();
392         checkClone(c, child, clone);
393 
394         clone = system().clone(c); // clone w/ same system
395         checkClone(c, child, clone);
396 
397         clone = newSystem(new SimpleCatalog()).clone(c); // clone w/ other system
398         checkClone(c, child, clone);
399     }
400 
401     private void checkClone(IonContainer c, IonSymbol child, IonContainer clone)
402     {
403         assertEquals(c, clone);
404         assertFalse("clone should not be read-only", clone.isReadOnly());
405 
406         IonValue childClone = clone.iterator().next();
407         assertEquals(child, childClone);
408         assertFalse("clone should not be read-only", childClone.isReadOnly());
409     }
