343     private static final int VALUE_BITS_PER_VARUINT_BYTE = 7;
344     private static final int LOWER_SEVEN_BITS_BITMASK = 0x7F;
345 
346     @Test
347     public void succeedsOnVarUIntMaxLong() throws Exception {
348         initializeStandardPipedBuffer();
349         // The IVM and the start of a variable-length NOP pad.
350         input.receive(bytes(0xE0, 0x01, 0x00, 0xEA, 0x0E));
351         lookahead.fillInput();
352         assertTrue(lookahead.moreDataRequired());
353         // The following writes the VarUInt representation of Long.MAX_VALUE one byte at a time.
354         // Note: Long.MAX_VALUE is represented in Long.SIZE - 1 bits because long is signed.
355         for (int i = Long.SIZE - 1; i > 0; i -= VALUE_BITS_PER_VARUINT_BYTE) {
356             int bitsToShift = i - VALUE_BITS_PER_VARUINT_BYTE;
357             int endMask = 0;
358             if (i <= VALUE_BITS_PER_VARUINT_BYTE) {
359                 endMask = 0x80;
360                 bitsToShift = 0;
361             }
362             input.receive((byte) (((Long.MAX_VALUE >>> bitsToShift) & LOWER_SEVEN_BITS_BITMASK) | endMask));
363             lookahead.fillInput();
364             assertTrue(lookahead.moreDataRequired());
365         }
366         // This is the first of the Long.MAX_VALUE bytes of NOP pad that the Ion reader would skip. Writing
367         // all of them is out of scope for this test, but testing that the first one succeeds verifies that
368         // the lookahead wrapper has exited its VarUInt parsing state.
369         input.receive(0);
370         lookahead.fillInput();
371         assertTrue(lookahead.moreDataRequired());
372     }

374     @Test
375     public void errorOnVarUIntTooLarge() throws Exception {
376         BigInteger longPlusOne = BigInteger.valueOf(Long.MAX_VALUE).add(BigInteger.ONE);
377         initializeStandardPipedBuffer();
378         // The IVM and the start of a variable-length NOP pad.
379         input.receive(bytes(0xE0, 0x01, 0x00, 0xEA, 0x0E));
380         lookahead.fillInput();
381         // The following writes the VarUInt representation of `longPlusOne` one byte at a time.
382         for (int i = longPlusOne.bitLength(); i > 0; i -= VALUE_BITS_PER_VARUINT_BYTE) {
383             assertTrue(lookahead.moreDataRequired());
384             if (i >= VALUE_BITS_PER_VARUINT_BYTE) {
385                 int bitsToShift = i - VALUE_BITS_PER_VARUINT_BYTE;
386                 input.receive(
387                     longPlusOne
388                         .shiftRight(bitsToShift)
389                         .and(BigInteger.valueOf(LOWER_SEVEN_BITS_BITMASK))
390                         .byteValue()
391                 );
392                 if (bitsToShift < VALUE_BITS_PER_VARUINT_BYTE) {
393                     // This is the second-to-last byte. A failure is expected because this is not the end
394                     // byte, so no matter what the next byte is, the value can't fit in a long.
395                     thrown.expect(IonException.class);
396                 }
397                 lookahead.fillInput();
398             } else {
399                 fail("Reached the last byte, which should not be possible.");
400             }
401         }
402     }

574     @Test
575     public void skipOversizedValues() throws Exception {
576         if (initialBufferSize == null) {
577             // This test tests buffers of limited size, while an initialBufferSize of null indicates that the size
578             // of the buffer is unlimited. Skip.
579             return;
580         }
581         // 14 values ranging in size from 1-14 bytes. There are 14 - x values larger than x bytes.
582         TestUtils.BinaryIonAppender appender = new TestUtils.BinaryIonAppender().append(
583             0x11, // boolean true
584             0x21, 0x00, // int 0 (overpadded)
585             0x22, 0x00, 0x01, // int 1 (overpadded)
586             0x33, 0x00, 0x00, 0x02, // int -2 (overpadded)
587             0x44, 0x00, 0x00, 0x00, 0x00, // float 0e0
588             0x55, 0x00, 0x00, 0x80, 0x00, 0x00, // decimal 0d0 (overpadded)
589             0x66, 0x80, 0x81, 0x81, 0x81, 0x80, 0x80, // timestamp 0001-01-01T00:00Z
590             0x7E, 0x00, 0x00, 0x00, 0x00, 0x82, 0x00, 0x00, // symbol 0 (overpadded length and value)
591             0x88, 'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', // string abcdefgh
592             0x99, 'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', // clob abcdefghi
593             0xAA, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08, 0x09, 0x0A, // blob
594             // list with ten int 0
595             0xBB, 0x20, 0x20, 0x20, 0x20, 0x20, 0x20, 0x20, 0x20, 0x20, 0x20, 0x20,
596             // sexp with ten decimal 0d0
597             0xCC, 0x50, 0x50, 0x50, 0x50, 0x50, 0x50, 0x50, 0x50, 0x50, 0x50, 0x50, 0x50,
598             // struct with system symbol field names and int 0 values (some overpadded)
599             0xDD, 0x84, 0x20, 0x85, 0x20, 0x86, 0x20, 0x87, 0x21, 0x00, 0x88, 0x22, 0x00, 0x00
600         );
601         byte[] data = appender.toByteArray();
602         IonDatagram expectedValues = SYSTEM.getLoader().load(data);;
603         for (int maxValueSize : Arrays.asList(5, 7, 8, 10, 13)) {
604             if (maxValueSize < initialBufferSize) {
605                 // This would violate the IonBufferConfiguration contract; no need to test here.
606                 continue;
607             }
608             ByteArrayInputStream input = new ByteArrayInputStream(data);
609             RecordingEventHandler eventHandler = new RecordingEventHandler();
610             IonBufferConfiguration.Builder builder = IonBufferConfiguration.Builder.standard()
611                 .withMaximumBufferSize(maxValueSize)
612                 .onOversizedValue(eventHandler)
613                 .onOversizedSymbolTable(eventHandler)
614                 .onData(eventHandler)
615                 .withInitialBufferSize(initialBufferSize);
616             IonReaderLookaheadBuffer lookahead = new IonReaderLookaheadBuffer(builder.build(), input);
617             IonReader reader = null;
618             try {
