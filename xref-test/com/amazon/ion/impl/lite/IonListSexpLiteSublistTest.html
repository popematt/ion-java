16  package com.amazon.ion.impl.lite;
17  
18  import static org.junit.Assert.assertEquals;
19  
20  import java.util.List;
21  import java.util.ListIterator;
22  import org.junit.Assert;
23  import org.junit.Test;
24  import org.junit.runner.RunWith;
25  import com.amazon.ion.IonInt;
26  import com.amazon.ion.IonSequence;
27  import com.amazon.ion.IonValue;
28  import com.amazon.ion.impl.lite.BaseIonSequenceLiteSublistTestCase;
29  import com.amazon.ion.junit.Injected;
30  import com.amazon.ion.junit.Injected.Inject;
31  
32  /**
33   * Sublist tests for list and sexp
34   */
35  @RunWith(Injected.class)
36  public class IonListSexpLiteSublistTest extends BaseIonSequenceLiteSublistTestCase {
37  
38      interface IonSequenceProvider {
39  
40          IonSequence newSequence();
41      }
42  
43      @Inject("provider")
44      public static final IonSequenceProvider[] providers = new IonSequenceProvider[]{
45          new IonSequenceProvider() {
46              public IonSequence newSequence() {
47                  return SYSTEM.newList(INTS);
48              }
49          },
50          new IonSequenceProvider() {
51              public IonSequence newSequence() {
52                  return SYSTEM.newSexp(INTS);
53              }
54          },
55      };
56  
57      private IonSequenceProvider provider;
58  
59      public void setProvider(final IonSequenceProvider provider) {
60          this.provider = provider;
61      }
62  
63      protected IonSequence newSequence() {
64          return provider.newSequence();
65      }
66  
67      @Test
68      public void sublistSet() {
69          final IonSequence sequence = newSequence();
70          final List<IonValue> sublist = sequence.subList(2, 5);
71  
72          final IonInt element = SYSTEM.newInt(99);
73          final IonValue previous = sublist.set(0, element);
74  
75          assertEquals(2, ((IonInt) previous).intValue());
76          assertEquals(element, sublist.get(0));
77          assertEquals(element, sequence.get(2));
78      }
79  
80      @Test
81      public void sublistSetOnParent() {
82          final IonSequence sequence = newSequence();
83          final List<IonValue> sublist = sequence.subList(2, 5);
84  
85          final IonInt element = SYSTEM.newInt(99);
86          final IonValue previous = sequence.set(2, element);
87  
88          assertEquals(2, ((IonInt) previous).intValue());
89          assertEquals(element, sublist.get(0));
90          assertEquals(element, sequence.get(2));
91      }
92  
93      @Test(expected = IndexOutOfBoundsException.class)
94      public void sublistSetOutOfRange() {
95          final IonSequence sequence = newSequence();
96          final List<IonValue> sublist = sequence.subList(2, 5);
97  
98          final IonInt element = SYSTEM.newInt(99);
99          sublist.set(4, element);
100     }
101 
102     @Test
103     public void sublistSublistChangingParentValues() {
104         IonSequence sequence = newSequence();
105 
106         final List<IonValue> sublist = sequence.subList(1, 5);    // 1,2,3,4
107         final List<IonValue> subSublist = sublist.subList(0, 2);  // 1,2
108 
109         final IonInt newValue = SYSTEM.newInt(100);
110         sequence.set(1, newValue);
111         assertEquals(newValue, sublist.get(0));
112         assertEquals(newValue, subSublist.get(0));
113     }
114 
115     @Test
116     public void sublistListIteratorWithSet() {
117         final IonSequence sequence = newSequence();
118         final List<IonValue> sublist = sequence.subList(2, 5);
119 
120         final ListIterator<IonValue> iterator = sublist.listIterator();
121 
122         iterator.next();
123 
124         final IonInt newSetValue = SYSTEM.newInt(100);
125         iterator.set(newSetValue);
126         assertEquals(newSetValue, sublist.get(0));
127     }
128 }
