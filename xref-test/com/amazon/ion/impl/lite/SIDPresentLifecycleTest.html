25  /**
26   * Tests to ensure that the current emerged retention behaviors for SID's across the IonValueLite DOM are correctly
27   * preserved with the symbol ID flag / state management. As of time of authoring the observed behavior for SID
28   * management is as follows:
29   *
30   * -------------------------------------------------------------------------------------------------------
31   * |                     | fieldName                 | annotations               | value (IonSymbolLite) |
32   * |                     | text + SID | SID only     | text + SID | SID only     | text + SID | SID only |
33   * -------------------------------------------------------------------------------------------------------
34   * | <init>              | N/A        | N/A          | N/A        | N/A          | clear      | retain   |
35   * | clone               | clear      | resolve/fail | clear      | retain       | clear      | clear (?)|
36   * | clearSymbolIDValues | clear      | retain       | clear      | clear [null] | clear      | retain   |
37   * | set                 | retain     | retain       | retain     | retain       | N/A        | N/A      |
38   * | get                 | ephemeral  | ephemeral    | ephemeral  | N/A          | ephemeral  | memoizes |
39   * -------------------------------------------------------------------------------------------------------
40   *
41   */
42  public class SIDPresentLifecycleTest extends IonTestCase {

44      @Test
45      public void testFieldsWithSID() {
46          IonStringLite value = (IonStringLite) system().newString("Test Value");
47          // a new value should create without SID presence as it has no annotations
48          assertFalse(value._isSymbolIdPresent());
49          value.setFieldNameSymbol(_Private_Utils.newSymbolToken("Symbol-With-Sid", 12));
50          // setting a field with a SID means that SID Present now must be true
51          assertTrue(value._isSymbolIdPresent());
52  
53          // test clone of the value results in no SID present
54          assertFalse(value.clone()._isSymbolIdPresent());
55  
56          // verify detaching from the container clears the SID present flag
57          value.detachFromContainer();
58          assertFalse(value._isSymbolIdPresent());
59  
60          //verify that detaching still retains symbols - as SID is still present
61          value.setFieldNameSymbol(_Private_Utils.newSymbolToken(13));
62          assertTrue(value._isSymbolIdPresent());
63          value.clearSymbolIDValues();
64          // NOTE: SID is still present as clearSymbolIDValues doesn't clear SID-only tokens
65          assertEquals(13, value.getFieldId());
66          assertTrue(value._isSymbolIdPresent());
67      }

69      @Test
70      public void testAnnotationsWithSIDs() {
71          IonStringLite value = (IonStringLite) system().newString("Test Value 2");
72          // a new value should create without any SID's as it has no annotations
73          assertFalse(value._isSymbolIdPresent());
74          value.setTypeAnnotationSymbols(
75                  _Private_Utils.newSymbolToken("Symbol-Without-SID", -1),
76                  _Private_Utils.newSymbolToken("Symbol-With-SID", 99));
77          // setting an annotation with a SID means that SID present now must be true
78          assertTrue(value._isSymbolIdPresent());
79  
80          // test clone of the value (given no *unresolvable* SID's) results in no SID's present
81          assertFalse(value.clone()._isSymbolIdPresent());
82  
83          // ensure that setting annotations that don't have SIDs doesn't change the SID present status (need to wait for
84          // an explicit action such as detatch from container or clone).
85          value.setTypeAnnotationSymbols(_Private_Utils.newSymbolToken("Symbol-Without-Sid", -1));
86          assertTrue(value._isSymbolIdPresent());
87  
88          // verify detaching from the container clears the SID Present flag
89          value.detachFromContainer();
90          assertFalse(value._isSymbolIdPresent());
91  
92          // verify that where SID context preserves through the clone due to SID context not being known (believed a bug)
93          // that the SID present flag is *retained*
94          value.setTypeAnnotationSymbols(_Private_Utils.newSymbolToken(101));
95          assertTrue(value.clone()._isSymbolIdPresent());
96      }

98      @Test
99      public void testPropagationAcrossGraph() {
100         IonStructLite struct = (IonStructLite) system().newEmptyStruct();
101         assertFalse(struct._isSymbolIdPresent());
102 
103         // 1. test addition of child by field WITHOUT SID causes no propagation
104         IonStringLite keyValue1 = (IonStringLite) system().newString("Foo");
105         struct.add(_Private_Utils.newSymbolToken("field_1", -1), keyValue1);
106         assertFalse(struct._isSymbolIdPresent());
107         assertFalse(keyValue1._isSymbolIdPresent());
108 
109         // 2. test addition of child by field WITH SID but also with field text causes no propagation
110         // (struct only takes field name if present and strips SID)
111         IonStringLite keyValue2 = (IonStringLite) system().newString("Bar");
112         struct.add(_Private_Utils.newSymbolToken("field_2", 87), keyValue2);
113         assertFalse(struct._isSymbolIdPresent());
114         assertFalse(keyValue1._isSymbolIdPresent());
115         assertFalse(keyValue2._isSymbolIdPresent());
116 
117         // 3. test addition of child by field with SID ONLY causes propagation due to retention of field SID
118         IonStringLite keyValue3 = (IonStringLite) system().newString("Car");
119         struct.add(_Private_Utils.newSymbolToken(76), keyValue3);
120         assertTrue(struct._isSymbolIdPresent());
121         assertFalse(keyValue1._isSymbolIdPresent());
122         assertFalse(keyValue2._isSymbolIdPresent());
123         assertTrue(keyValue3._isSymbolIdPresent());
124 
125         // 4. test explicitly annotating a field propagates
126         IonStructLite struct2 = (IonStructLite) system().newEmptyStruct();
127         IonStringLite keyValue4 = (IonStringLite) system().newString("But");
128         struct2.add("field_1", keyValue4);
129         keyValue4.setTypeAnnotationSymbols(_Private_Utils.newSymbolToken("Lah", 13));
130         assertTrue(struct2._isSymbolIdPresent());
131 
132         // 5. test clone propagates clearing of status. NOTE - clone with an unresolvable SID for a field ID fails
133         try {
134             struct.clone();
135             fail("clone was expected NOT to succeed due to behavior at time of writing where unresolvable SID's fail");
136         } catch (UnknownSymbolException use) {
137             // this is expected??! until someone fixed the TO DO in IonValueLite#getFieldName
138         }
139 
140         // remove the field without SID such that the clone can be enacted.
141         struct.remove_child(2);
142 
143         IonStructLite clonedStruct = struct.clone();
144         assertFalse(clonedStruct._isSymbolIdPresent());
145         for (IonValue value : clonedStruct) {
146             assertFalse(((IonValueLite) value)._isSymbolIdPresent());
147         }
148 
149         // 6. ensure clearing symbols propogates from root to impacted leaves
150         // add back in field ID (SID) only child before clearing to test SID retained
151         struct.add(_Private_Utils.newSymbolToken(76), keyValue3);
152         struct.clearSymbolIDValues();
153         assertTrue(struct._isSymbolIdPresent());
154         assertFalse(keyValue1._isSymbolIdPresent());
155         assertFalse(keyValue2._isSymbolIdPresent());
156         assertTrue(keyValue3._isSymbolIdPresent());
157 
158         // 7. ensure that nested SID-only annotation clone behavior propogates to cloned root.
159         IonStructLite outer = (IonStructLite) system().newEmptyStruct();
160         IonListLite middle = (IonListLite) system().newEmptyList();
161         IonIntLite inner = (IonIntLite) system().newInt(10);
162         middle.add(inner);
163         outer.put("foo", middle);
164         // now add a SID only annotation to inner
165         inner.setTypeAnnotationSymbols(_Private_Utils.newSymbolToken(99));
166         // verify that all components are signaling SID present
167         assertTrue(outer._isSymbolIdPresent());
168         assertTrue(middle._isSymbolIdPresent());
169         assertTrue(inner._isSymbolIdPresent());
170         // conduct a clone and verify all cloned components have retained SID
171         IonStructLite clonedOuter = outer.clone();
172         assertTrue(clonedOuter._isSymbolIdPresent());
173         IonListLite clonedMiddle = (IonListLite) outer.get("foo");
174         assertTrue(clonedMiddle._isSymbolIdPresent());
175         assertTrue(clonedMiddle.get_child(0)._isSymbolIdPresent());
176     }

178     @Test
179     public void testSymbolValueLifecyle() {
180         // test creation from SymbolToken
181         IonSymbolLite symbolLite = (IonSymbolLite) system().newSymbol(_Private_Utils.newSymbolToken("version", 5));
182         // SID isn't present as it is stripped if there is text present
183         assertFalse(symbolLite._isSymbolIdPresent());
184 
185         IonListLite container1 = (IonListLite) system().newEmptyList();
186         container1.add(symbolLite);
187         // test propagation of context when symbol context is added
188         assertEquals(5, symbolLite.getSymbolId());
189         // as the getSymbolId memoizes SID into the entity SID present must be set
190         assertTrue(symbolLite._isSymbolIdPresent());
191         // check memoization propagates to container
192         assertTrue(container1._isSymbolIdPresent());
193         container1.clearSymbolIDValues();
194         assertFalse(container1._isSymbolIdPresent());
195         assertFalse(symbolLite._isSymbolIdPresent());
196 
197         IonSymbolLite symbolSIDOnly = (IonSymbolLite) system().newSymbol(_Private_Utils.newSymbolToken(321));
198         // SID Present is preserved when only SID is present
199         assertTrue(symbolSIDOnly._isSymbolIdPresent());
200         IonListLite container2 = (IonListLite) system().newEmptyList();
201         container2.add(symbolSIDOnly);
202         assertTrue(container2._isSymbolIdPresent());
203         // clear symbol ID values won't clear the symbolSIDOnly SID value - therefore container2 must still remain
204         // marked as having a SID present
205         container2.clearSymbolIDValues();
206         assertTrue(container2._isSymbolIdPresent());
207         assertTrue(symbolSIDOnly._isSymbolIdPresent());
208     }

210     @Test
211     public void testSIDRetentionAcrossObjectHierarchy() {
212         // field ID's can be retained on Container - so ensure these are retained.
213         IonStructLite outer = (IonStructLite) system().newEmptyStruct();
214         IonListLite middle = (IonListLite) system().newEmptyList();
215         IonStringLite inner = (IonStringLite) system().newString("A");
216         outer.add(_Private_Utils.newSymbolToken(321), middle);
217         middle.add(inner);
218         inner.setTypeAnnotationSymbols(_Private_Utils.newSymbolToken("B", 123));
219         assertTrue(outer._isSymbolIdPresent());
220         assertTrue(middle._isSymbolIdPresent());
221         assertTrue(inner._isSymbolIdPresent());
222         outer.clearSymbolIDValues();
223         assertTrue(outer._isSymbolIdPresent());
224         assertTrue(middle._isSymbolIdPresent());
225         assertFalse(inner._isSymbolIdPresent());
226 
227         // field ID's can be retained on IonSymbolLite - so ensure these are retained as well.
228         IonStructLite container = (IonStructLite) system().newEmptyStruct();
229         IonSymbolLite symbol = (IonSymbolLite) system().newSymbol(_Private_Utils.newSymbolToken("Foo", 123));
230         container.add(_Private_Utils.newSymbolToken(321), symbol);
231         container.clearSymbolIDValues();
232         assertEquals(SymbolTable.UNKNOWN_SYMBOL_ID, symbol.getSymbolId());
233         assertEquals("Foo", symbol.stringValue());
234         assertEquals(321, symbol.getFieldId());
235         assertTrue(symbol._isSymbolIdPresent());
236 
237         assertTrue(container._isSymbolIdPresent());
238     }
