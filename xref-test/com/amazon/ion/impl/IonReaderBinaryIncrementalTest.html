111     /**
112      * Creates an incremental reader over the binary Ion data created by invoking the given RawWriterFunction.
113      * @param writerFunction the function used to generate the data.
114      * @return a new reader.
115      * @throws Exception if an exception is raised while writing the Ion data.
116      */
117     private IonReaderBinaryIncremental readerFor(RawWriterFunction writerFunction) throws Exception {
118         ByteArrayOutputStream out = new ByteArrayOutputStream();
119         _Private_IonRawWriter writer = writerBuilder.build(out)
120             .asFacet(_Private_IonManagedWriter.class)
121             .getRawWriter();
122         out.write(_Private_IonConstants.BINARY_VERSION_MARKER_1_0);
123         writerFunction.write(writer, out);
124         writer.close();
125         return new IonReaderBinaryIncremental(readerBuilder, new ByteArrayInputStream(out.toByteArray()));
126     }

3214      * Compares an iterator to the given list.
3215      * @param expected the list containing the elements to compare to.
3216      * @param actual the iterator to be compared.
3217      * @param isEqualityExpected true if the iterator is expected to contain the same elements as the list; otherwise,
3218      *                           false.
3219      */
3220     private static void compareIterator(List<String> expected, Iterator<String> actual, boolean isEqualityExpected) {
3221         int numberOfElements = 0;
3222         while (actual.hasNext()) {
3223             String actualValue = actual.next();
3224             if (numberOfElements >= expected.size()) {
3225                 assertEquals(isEqualityExpected, !actual.hasNext());
3226                 break;
3227             }
3228             String expectedValue = expected.get(numberOfElements++);
3229             if (isEqualityExpected) {
3230                 assertEquals(expectedValue, actualValue);
3231             } else {
3232                 assertNotEquals(expectedValue, actualValue);
3233             }
3234         }
3235         assertEquals(isEqualityExpected, numberOfElements == expected.size());
3236     }
