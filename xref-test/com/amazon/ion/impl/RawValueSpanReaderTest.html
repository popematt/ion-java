83      /*
84       * Returns a byte array that contains the value ONLY (i.e. no type code
85       * or length), as encoded by Ion.
86       */
87      private byte[] valueBytes(OffsetSpan valueSpan)
88      {
89          // In a real-world use case, this byte buffer may already be allocated.
90          int valueSize = (int)(valueSpan.getFinishOffset() - valueSpan.getStartOffset());
91          byte[] value = new byte[valueSize];
92          System.arraycopy(spanProvider.buffer(), (int)valueSpan.getStartOffset(), value, 0, value.length);
93          return value;
94      }

96      private static class SpanTester
97      {
98          final Object expected;
99          final byte[] expectedBytes;
100         final IonType expectedType;
101         final Span span;
102 
103         SpanTester(Object expected, IonType expectedType, byte[] expectedBytes, Span span)
104         {
105             this.expected = expected;
106             this.expectedBytes = expectedBytes;
107             this.expectedType = expectedType;
108             this.span = span;
109         }
110     }
111 
112     private SpanTester generateSpan(Object expected, IonType type)
113     {
114         // This span includes the TID and length bytes
115         Span span = seekableReader.currentSpan();
116         // This gets just the value bytes
117         byte[] expectedBytes = valueBytes((OffsetSpan)spanProvider.valueSpan());
118         return new SpanTester(expected, type, expectedBytes, span);
119     }
120 
121     private void generateSpans(List<SpanTester> output)
122     {
123         IonType type = null;
124         while ((type = reader.next()) != null)
125         {
126             if (reader.isNullValue())
127             {
128                 output.add(generateSpan(null, type));
129                 continue;
130             }
131             switch (type)
132             {
133                 case BOOL:
134                     output.add(generateSpan(reader.booleanValue(), type));
135                     break;
136                 case INT:
137                     output.add(generateSpan(reader.bigIntegerValue(), type));
138                     break;
139                 case FLOAT:
140                     output.add(generateSpan(reader.doubleValue(), type));
141                     break;
142                 case DECIMAL:
143                     output.add(generateSpan(reader.bigDecimalValue(), type));
144                     break;
145                 case TIMESTAMP:
146                     output.add(generateSpan(reader.timestampValue(), type));
147                     break;
148                 case STRING:
149                     output.add(generateSpan(reader.stringValue(), type));
150                     break;
151                 case SYMBOL:
152                     output.add(generateSpan(reader.symbolValue(), type));
153                     break;
154                 case BLOB:
155                 case CLOB:
156                     output.add(generateSpan(reader.newBytes(), type));
157                     break;
158                 case LIST:
159                 case SEXP:
160                 case STRUCT:
161                     // 'expected' not used for containers because there is no
162                     // corresponding *Value() method on the reader. Instead,
163                     // byte and position comparisons will be used, and the
164                     // containers' values will be recursively checked.
165                     output.add(generateSpan(null, type));
166                     reader.stepIn();
167                     generateSpans(output);
168                     reader.stepOut();
169                     break;
170                 default:
171                     throw new IllegalStateException("unexpected type: " + type);
172             }
173         }
174     }

176     private void assertSpan(SpanTester tester) throws Exception
177     {
178 
179         IonType type = tester.expectedType;
180         seekableReader.hoist(tester.span); // seeks back to the given span
181         assertEquals(type, reader.next());
182 
183         Object expected = tester.expected;
184         if (reader.isNullValue())
185         {
186             assertEquals(expected, null);
187         }
188         else
189         {
190             OffsetSpan valueSpan = (OffsetSpan)spanProvider.valueSpan();
191             switch(type)
192             {
193                 case BOOL:
194                     assertEquals(expected, reader.booleanValue());
195                     break;
196                 case INT:
197                     assertEquals(expected, reader.bigIntegerValue());
198                     break;
199                 case FLOAT:
200                     assertEquals(expected, reader.doubleValue());
201                     break;
202                 case DECIMAL:
203                     assertEquals(expected, reader.bigDecimalValue());
204                     break;
205                 case TIMESTAMP:
206                     assertEquals(expected, reader.timestampValue());
207                     break;
208                 case STRING:
209                     // A common use case will be to pass strings along without
210                     // decoding. This tests that case.
211                     assertArrayEquals(((String)expected).getBytes("UTF-8"), valueBytes(valueSpan));
212                     assertEquals(expected, reader.stringValue());
213                     break;
214                 case SYMBOL:
215                     // SymbolTokenImpl does not override .equals
216                     SymbolToken expectedToken = (SymbolToken)expected;
217                     SymbolToken actualToken = reader.symbolValue();
218                     assertEquals(expectedToken.getSid(), actualToken.getSid());
219                     assertEquals(expectedToken.getText(), actualToken.getText());
220 
221                     break;
222                 case BLOB:
223                 case CLOB:
224                     assertArrayEquals((byte[])expected, reader.newBytes());
225                     break;
226                 case STRUCT:
227                 case LIST:
228                 case SEXP:
229                     reader.stepIn();
230                     if (reader.next() != null)
231                     {
232                         // The start position of the container's value span should
233                         // be the same as the start position of its first element's
234                         // seekable span.
235                         long expectedValueStart = valueSpan.getStartOffset();
236                         long expectedValueEnd = ((OffsetSpan) seekableReader.currentSpan()).getStartOffset();
237 
238                         // skips any nop pad to get at the actual value start
239                         expectedValueStart += countNopPad((int) expectedValueStart);
240 
241                         if (reader.isInStruct())
242                         {
243                             // In structs, however, value spans will start before
244                             // the first value's field name SID (VarUInt - 7
245                             // bits per byte, hence division by 0x80).
246                             expectedValueStart += (reader.getFieldNameSymbol().getSid() / 0x80) + 1;
247                         }
248 
249                         assertEquals(expectedValueStart, expectedValueEnd);
250                     }
251 
252                     reader.stepOut();
253 
254                     break;
255                 default:
256                     throw new IllegalStateException("unexpected type: " + type);
257             }
258             assertArrayEquals(tester.expectedBytes, valueBytes(valueSpan));
259             // All spans over the same value, no matter where they started, should finish at
260             // the same position.
261             assertEquals(((OffsetSpan)tester.span).getFinishOffset(), valueSpan.getFinishOffset());
262         }
263     }
264 
265     private int countNopPad(int start) throws IOException
266     {
267         int index = start;
268         int len = 0;
269 
270         if (reader.isInStruct())
271         {
272             // In structs nop pads have a sid before then
273             index += (reader.getFieldNameSymbol().getSid() / 0x80) + 1;
274             len += index - start;
275         }
276 
277         int td = reader.getCurrentBuffer()[index] & 0xFF;
278         int tid = _Private_IonConstants.getTypeCode(td);
279         int typeLen = _Private_IonConstants.getLowNibble(td);
280 
281         if(tid == _Private_IonConstants.tidNull && typeLen != _Private_IonConstants.lnIsNull){
282             if(typeLen == _Private_IonConstants.lnIsVarLen) {
283                 len += reader.readVarUInt();
284             }
285             else {
286                 len += 1;
287             }
288 
289             return len;
290         }
291 
292         return 0;
293     }

295     private static byte[] readFileAsBytes(File file) throws IOException
296     {
297         FileInputStream in = new FileInputStream(file);
298         byte[] data = new byte[(int)file.length()];
299         in.read(data);
300         in.close();
301         return data;
302     }

304     /**
305      * Retrieve a span for each value. Seek back to them in any order and assert
306      * that the values can be retrieved both through raw inspection of the
307      * underlying buffer and through the high-level IonReader APIs.
308      */
309     @Test
310     public void testSpans() throws Exception
311     {
312         // Seeking currently only works over byte-backed IonReaders -- not InputStreams
313         reader = (IonReaderBinaryUserX) IonReaderBuilder.standard().build(readFileAsBytes(myTestFile));
314         spanProvider = reader.asFacet(RawValueSpanProvider.class);
315         seekableReader = reader.asFacet(SeekableReader.class);
316 
317         List<SpanTester> spans = new ArrayList<SpanTester>();
318         generateSpans(spans);
319         Collections.shuffle(spans); // spans can be revisited in any order
320         for (SpanTester span : spans)
321         {
322             assertSpan(span);
323         }
324     }
