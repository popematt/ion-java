77      @Test
78      public void testStandardCallback()
79          throws IOException
80      {
81          // Write to a StringWriter for testing
82          StringWriter out = new StringWriter();
83          IonReader ionReader = system().newReader(input);
84  
85          _Private_IonTextWriterBuilder builder = (_Private_IonTextWriterBuilder)
86              IonTextWriterBuilder.standard();
87          IonWriter ionWriter = builder
88              .withCallbackBuilder(new TestMarkupCallback.Builder())
89              .build(out);
90  
91          write(ionReader, ionWriter);
92          assertEquals("Markup callback with standard Ion Writer, " +
93                       "error with data:\n" + input + "\n",
94                       sExpected, out.toString());
95      }

97      @Test
98      public void testPrettyCallback()
99          throws IOException
100     {
101         // Write to a StringWriter for testing
102         StringWriter out = new StringWriter();
103         IonReader ionReader = system().newReader(input);
104         _Private_IonTextWriterBuilder builder = (_Private_IonTextWriterBuilder)
105             IonTextWriterBuilder.pretty();
106         IonWriter ionWriter = builder
107             .withCallbackBuilder(new TestMarkupCallback.Builder())
108             .build(out);
109         write(ionReader, ionWriter);
110         assertEquals("Markup callback with pretty printing Ion Writer," +
111                      " error with data:\n" + input + "\n",
112                      pExpected, out.toString());
113     }

115     @Test
116     public void testEscaping() throws IOException
117     {
118         String input = "These should be escaped <>&";
119         String expected = "<><><>&&&<><><>\"These should be escaped &lt;&gt;&amp;\"";
120         StringWriter out = new StringWriter();
121         _Private_IonTextWriterBuilder builder = (_Private_IonTextWriterBuilder)
122             IonTextWriterBuilder.standard();
123         IonWriter ionWriter = builder
124             .withCallbackBuilder(new EscapingCallback.Builder())
125             .build(out);
126         ionWriter.writeString(input);
127         ionWriter.finish();
128         assertEquals("Escaping failed, with data:\n" + input + "\n",
129                      expected, out.toString());
130     }
131 
132     public void write(IonReader ionReader, IonWriter ionWriter)
133         throws IOException
134     {
135         assertNotNull("ionReader is null", ionReader);
136         assertNotNull("ionWriter is null", ionWriter);
137 
138         ionWriter.writeValues(ionReader);
139 
140         ionWriter.close();
141         ionReader.close();
142     }
