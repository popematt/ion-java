116     protected IonWriter createWriter(final OutputStream out) throws IOException
117     {
118         return new IonRawBinaryWriter(
119             BlockAllocatorProviders.basicProvider(),
120             11,
121             out,
122             WriteValueOptimization.NONE,
123             StreamCloseMode.NO_CLOSE,
124             StreamFlushMode.NO_FLUSH,
125             preallocationMode,
126             true
127         );
128     }

130     private void writeIVMIfPossible() throws IOException
131     {
132         final IonWriter delegate = writer.getDelegate();
133         if (delegate instanceof IonRawBinaryWriter)
134         {
135             ((IonRawBinaryWriter) delegate).writeIonVersionMarker();
136         }
137     }

139     /** sub-classes can provide additional checks on a value as part of {@link #assertValue(String)}. */
140     protected void additionalValueAssertions(final IonValue value) {}
141 
142     protected final void assertValue(final String literal) throws IOException
143     {
144         writer.finish();
145         final byte[] data = writer.getBytes();
146         final IonValue actual;
147         try {
148             actual = system().singleValue(data);
149         } catch (final Exception e) {
150             throw new IonException("Bad generated data:\n" + hexDump(data), e);
151         }
152         final IonValue expected = system().singleValue(literal);
153         assertEquals(expected, actual);
154 
155         additionalValueAssertions(actual);
156 
157         // prepare for next value
158         writer.reset();
159         writeIVMIfPossible();
160     }
