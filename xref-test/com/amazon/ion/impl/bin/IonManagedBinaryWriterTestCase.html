40      private static final Map<String, Integer> SHARED_SYMBOL_LOCAL_SIDS ;
41      static
42      {
43          final Map<String, Integer> sidMap = new HashMap<String, Integer>();
44  
45          for (final SymbolToken token : Symbols.systemSymbols())
46          {
47              sidMap.put(token.getText(), token.getSid());
48          }
49          int sid = SystemSymbols.ION_1_0_MAX_ID + 1;
50          for (final List<String> symbolList : SHARED_SYMBOLS)
51          {
52              for (final String symbol : symbolList) {
53                  sidMap.put(symbol, sid);
54                  sid++;
55              }
56          }
57          SHARED_SYMBOL_LOCAL_SIDS = unmodifiableMap(sidMap);
58      }

75      private void checkSymbolTokenAgainstImport(final SymbolToken token)
76      {
77          final Integer sid = SHARED_SYMBOL_LOCAL_SIDS.get(token.getText());
78          if (sid != null)
79          {
80              assertEquals(sid.intValue(), token.getSid());
81          }
82      }

84      @Override
85      protected void additionalValueAssertions(final IonValue value)
86      {
87          for (final SymbolToken token : value.getTypeAnnotationSymbols()) {
88              checkSymbolTokenAgainstImport(token);
89          }
90          final IonType type = value.getType();
91          if (type == IonType.SYMBOL && !value.isNullValue())
92          {
93              checkSymbolTokenAgainstImport(((IonSymbol) value).symbolValue());
94          }
95          else if (IonType.isContainer(type))
96          {
97              for (final IonValue child : ((IonContainer) value))
98              {
99                  additionalValueAssertions(child);
100             }
101         }
102     }

120     @Override
121     protected IonWriter createWriter(final OutputStream out) throws IOException
122     {
123         final IonMutableCatalog catalog = ((IonMutableCatalog) system().getCatalog());
124 
125         final List<SymbolTable> symbolTables = new ArrayList<SymbolTable>();
126         int i = 1;
127         for (final List<String> symbols : SHARED_SYMBOLS) {
128             final SymbolTable table = system().newSharedSymbolTable("test_" + (i++), 1, symbols.iterator());
129             symbolTables.add(table);
130             catalog.putTable(table);
131         }
132 
133         final _Private_IonManagedBinaryWriterBuilder builder = _Private_IonManagedBinaryWriterBuilder
134             .create(_Private_IonManagedBinaryWriterBuilder.AllocatorMode.POOLED)
135             .withImports(importedSymbolResolverMode, symbolTables)
136             .withPreallocationMode(preallocationMode)
137             .withFloatBinary32Enabled();
138 
139         if (lstAppendMode.isEnabled()) {
140             builder.withLocalSymbolTableAppendEnabled();
141         } else {
142             builder.withLocalSymbolTableAppendDisabled();
143         }
144 
145         final IonWriter writer = builder.newWriter(out);
146 
147         final SymbolTable locals = writer.getSymbolTable();
148         assertEquals(14, locals.getImportedMaxId());
149 
150         return writer;
151     }
