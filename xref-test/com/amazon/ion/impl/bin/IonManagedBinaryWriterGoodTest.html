25          testdataFiles(GLOBAL_SKIP_LIST,

30  
31      public void setTestFile(File file)

33          myTestFile = file;
34      }
35  
36      @Test
37      public void allGoodFiles() throws Exception {
38          byte[] testData = _Private_Utils.loadFileBytes(myTestFile);
39          IonReader reader = IonReaderBuilder.standard().build(testData);
40          writer.writeValues(reader);
41          reader.close();
42  
43          writer.finish();
44          final byte[] data = writer.getBytes();
45          final IonValue actual;
46          try {
47              actual = system().getLoader().load(data);
48          } catch (final Exception e) {
49              throw new IonException("Bad generated data:\n" + hexDump(data), e);
50          }
51          final IonValue expected = system().getLoader().load(testData);
52          assertEquals(expected, actual);
53  
54          additionalValueAssertions(actual);
55      }
