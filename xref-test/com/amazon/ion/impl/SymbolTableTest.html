254     public void testLocalSymbolTableAppendImportBoundary() throws IOException
255     {
256         String text =
257             LocalSymbolTablePrefix +
258                 "{" +
259                 "  symbols:[ \"s11\"]" +
260                 "}\n" +
261                 "1\n" +
262             LocalSymbolTablePrefix +
263                 "{" +
264                 "  imports:" + ION_SYMBOL_TABLE + "," +
265                 "  symbols:[ \"s21\"]" +
266                 "}\n" +
267                 "null";
268 
269         IonDatagram datagram = loader().load(text);
270 
271         SymbolTable symbolTable = datagram.get(0).getSymbolTable();
272         assertSame(symbolTable, datagram.get(1).getSymbolTable());
273         symbolTable.intern("o1");
274         symbolTable.intern("a1");
275 
276         // new symbols don't influence SIDs for existing symbols; they are appended
277         checkSymbol("s11", systemMaxId() + 1, symbolTable);
278         checkSymbol("s21", systemMaxId() + 2, symbolTable);
279         checkSymbol("o1", systemMaxId() + 3, symbolTable);
280         checkSymbol("a1", systemMaxId() + 4, symbolTable);
281 
282         ByteArrayOutputStream out = new ByteArrayOutputStream();
283         IonWriter writer = IonBinaryWriterBuilder.standard().build(out);
284         datagram.writeTo(writer);
285         writer.close();
286         IonDatagram roundtripped = loader().load(out.toByteArray());
287         assertTrue(Equivalence.ionEquals(datagram, roundtripped));
288     }

290     @Test
291     public void testMutateDomAfterLocalSymbolTableAppend() throws IOException {
292         String text =
293                 LocalSymbolTablePrefix +
294                 "{" +
295                 "   imports:[{name:\"foo\", version:1, max_id:1}], " +
296                 "   symbols:[\"s1\", \"s2\"]" +
297                 "}\n" +
298                 "$10\n" + // Symbol with unknown text from "foo"
299                 "$11\n" + // s1
300                 LocalSymbolTablePrefix +
301                 "{" +
302                 "   imports:" + ION_SYMBOL_TABLE + "," +
303                 "   symbols:[\"s3\"]" +
304                 "}\n" +
305                 "$12"; // s2
306         IonDatagram datagram = loader().load(text);
307         datagram.add(1, system().newSymbol("abc"));
308         datagram.add(system().newSymbol(new SymbolTokenImpl(null, 13))); // s3
309         datagram.add(system().newSymbol(new SymbolTokenImpl(null, 10))); // Symbol with unknown text from "foo"
310         ByteArrayOutputStream textOutput = new ByteArrayOutputStream();
311         ByteArrayOutputStream binaryOutput = new ByteArrayOutputStream();
312         List<String> fooSymbols = Collections.singletonList("bar");
313         SymbolTable fooTable = system().newSharedSymbolTable("foo", 1, fooSymbols.iterator());
314         IonWriter textWriter = IonTextWriterBuilder.standard().withImports(fooTable).build(textOutput);
315         datagram.writeTo(textWriter);
316         textWriter.close();
317         IonWriter binaryWriter = IonBinaryWriterBuilder.standard().withImports(fooTable).build(binaryOutput);
318         datagram.writeTo(binaryWriter);
319         binaryWriter.close();
320         SimpleCatalog catalog = new SimpleCatalog();
321         catalog.putTable(fooTable);
322         IonSystem system = IonSystemBuilder.standard().withCatalog(catalog).build();
323         IonDatagram textRoundtrip = system.getLoader().load(textOutput.toByteArray());
324         IonDatagram binaryRoundtrip = system.getLoader().load(binaryOutput.toByteArray());
325         assertEquals(textRoundtrip, binaryRoundtrip);
326         assertEquals("bar", ((IonSymbol)textRoundtrip.get(0)).stringValue());
327         assertEquals("abc", ((IonSymbol)textRoundtrip.get(1)).stringValue());
328         assertEquals("s1", ((IonSymbol)textRoundtrip.get(2)).stringValue());
329         assertEquals("s2", ((IonSymbol)textRoundtrip.get(3)).stringValue());
330         assertEquals("s3", ((IonSymbol)textRoundtrip.get(4)).stringValue());
331         assertEquals("bar", ((IonSymbol)textRoundtrip.get(0)).stringValue());
332     }

564     public void testLocalTableWithMissingImport()
565         throws IOException
566     {
567         // Use a big symtab to get beyond any default allocation within the
568         // dummy symtab.  This was done to trap a bug in LocalSymbolTable.
569         ArrayList<String> syms = new ArrayList<String>();
570         int maxId = 50;
571         for (int i = 1; i <= maxId; i++)
572         {
573             syms.add("S" + i);
574         }
575 
576         SymbolTable table =
577             system().newSharedSymbolTable("T", 1, syms.iterator());
578 
579         SimpleCatalog catalog = (SimpleCatalog) system().getCatalog();
580         catalog.putTable(table);
581 
582         final int import1id = systemMaxId() + 1;
583         final int import2id = systemMaxId() + 2;
584 
585         final int local1id = systemMaxId() + maxId + 1;
586         final int local2id = local1id + 1;
587 
588 
589         String text =
590             LocalSymbolTablePrefix +
591             "{" +
592             "  symbols:[ \"local1\", \"local2\" ]," +
593             "  imports:[{name:'''T''', version:1," +
594             "            max_id:" + maxId + "}]," +
595             "}\n" +
596             "local1 local2 S1 S2";
597         byte[] binary = encode(text);
598 
599         // Remove the imported table before decoding the binary.
600         assertSame(table, catalog.removeTable("T", 1));
601 
602         IonDatagram dg = loader().load(binary);
603         checkSymbol("local1", local1id, dg.get(0));
604         checkSymbol("local2", local2id, dg.get(1));
605         checkUnknownSymbol(import1id, dg.get(2));
606         checkUnknownSymbol(import2id, dg.get(3));
607 
608         SymbolTable st = dg.get(3).getSymbolTable();
609         checkLocalTable(st);
610         checkUnknownSymbol("S1", import1id, st);
611         checkUnknownSymbol("S2", import2id, st);
612         assertEquals(local1id - 1, st.getImportedMaxId());
613 
614         SymbolTable dummy = checkFirstImport("T", 1, new String[maxId], st);
615         assertTrue(dummy.isSubstitute());
616         checkUnknownSymbol("S1", import1id, dummy);
617         checkUnknownSymbol("S2", import2id, dummy);
618 
619         SymbolTable[] importedTables = st.getImportedTables();
620         assertEquals(1, importedTables.length);
621     }

1184                                         String[] expectedSymbols,
1185                                         SymbolTable actual)

1190

1193         assertEquals(name, actual.getName());

1195         assertEquals(0, actual.getImportedMaxId());

1197         assertEquals("symbol count",
1198                      expectedSymbols.length, actual.getMaxId());
1199 
1200         Set<String> foundSymbols = new HashSet<String>();
1201 
1202         Iterator<String> iter = actual.iterateDeclaredSymbolNames();
1203         for (int i = 0; i < expectedSymbols.length; i++)
1204         {
1205             int sid = i+1;

1207 
1208             assertTrue(iter.hasNext());
1209             assertEquals(text, iter.next());
1210 
1211             boolean dupe = text != null && !foundSymbols.add(text);
1212 
1213             checkSymbol(text, sid, dupe, actual);
1214         }
1215 
1216         assertFalse(iter.hasNext());
1217 
1218         checkUnknownSymbol(" not defined ", UNKNOWN_SYMBOL_ID, actual);
1219     }
